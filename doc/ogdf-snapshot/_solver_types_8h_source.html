123 
124 class Clause;
125 typedef RegionAllocator<uint32_t>::Ref CRef;
126 
127 #ifdef _MSC_VER
128 #pragma warning ( push )
129 #pragma warning ( disable : 4200 )
130 #endif
131 
132 class Clause {
133  struct {
134  unsigned mark : 2;
135  unsigned learnt : 1;
136  unsigned has_extra : 1;
137  unsigned reloced : 1;
138  unsigned size : 27; } header;
139  union { Lit lit; float act; uint32_t abs; CRef rel; } data[0];
140 
141  friend class ClauseAllocator;
142 
143  // NOTE: This constructor cannot be used directly (doesn't allocate enough memory).
144  template<class V>
145  Clause(const V& ps, bool use_extra, bool learnt) {
146  header.mark = 0;
147  header.learnt = learnt;
148  header.has_extra = use_extra;
149  header.reloced = 0;
150  header.size = ps.size();
151 
152  for (int i = 0; i < ps.size(); i++)
153  data[i].lit = ps[i];
154 
155  if (header.has_extra){
156  if (header.learnt)
157  data[header.size].act = 0;
158  else
159  calcAbstraction(); }
160  }
161 
162 public:
163  void calcAbstraction() {
164  assert(header.has_extra);
165  uint32_t abstraction = 0;
166  for (int i = 0; i < size(); i++)
167  abstraction |= 1 << (var(data[i].lit) & 31);
168  data[header.size].abs = abstraction; }
169 
170 
171  int size () const { return header.size; }
172  void shrink (int i) { assert(i <= size()); if (header.has_extra) data[header.size-i] = data[header.size]; header.size -= i; }
173  void pop () { shrink(1); }
174  bool learnt () const { return header.learnt; }
175  bool has_extra () const { return header.has_extra; }
176  uint32_t mark () const { return header.mark; }
177  void mark (uint32_t m) { header.mark = m; }
178  const Lit& last () const { return data[header.size-1].lit; }
179 
180  bool reloced () const { return header.reloced; }
181  CRef relocation () const { return data[0].rel; }
182  void relocate (CRef c) { header.reloced = 1; data[0].rel = c; }
183 
184  // NOTE: somewhat unsafe to change the clause in-place! Must manually call 'calcAbstraction' afterwards for
185  // subsumption operations to behave correctly.
186  Lit& operator [] (int i) { return data[i].lit; }
187  Lit operator [] (int i) const { return data[i].lit; }
188  operator const Lit* (void) const { return reinterpret_cast<const Lit*>(data); }
189 
190  float& activity () { assert(header.has_extra); return data[header.size].act; }
191  uint32_t abstraction () const { assert(header.has_extra); return data[header.size].abs; }
192 
193  Lit subsumes (const Clause& other) const;
194  void strengthen (Lit p);
195 };

206 const CRef CRef_Undef = RegionAllocator<uint32_t>::Ref_Undef;
207 class ClauseAllocator : public RegionAllocator<uint32_t>
208 {
209  static int clauseWord32Size(int size, bool has_extra){
210  return (sizeof(Clause) + (sizeof(Lit) * (size + (int)has_extra))) / sizeof(uint32_t); }
211  public:
212  bool extra_clause_field;
213 
214  ClauseAllocator(uint32_t start_cap) : RegionAllocator<uint32_t>(start_cap), extra_clause_field(false){}
215  ClauseAllocator() : extra_clause_field(false){}
216 
217  void moveTo(ClauseAllocator& to){
218  to.extra_clause_field = extra_clause_field;
219  RegionAllocator<uint32_t>::moveTo(to); }
220 
221  template<class Lits>
222  CRef alloc(const Lits& ps, bool learnt = false)
223  {
224  assert(sizeof(Lit) == sizeof(uint32_t));
225  assert(sizeof(float) == sizeof(uint32_t));
226  bool use_extra = learnt | extra_clause_field;
227 
228  CRef cid = RegionAllocator<uint32_t>::alloc(clauseWord32Size(ps.size(), use_extra));
229  new (lea(cid)) Clause(ps, use_extra, learnt);
230 
231  return cid;
232  }
233 
234  // Deref, Load Effective Address (LEA), Inverse of LEA (AEL):
235  Clause& operator[](Ref r) { return (Clause&)RegionAllocator<uint32_t>::operator[](r); }
236  const Clause& operator[](Ref r) const { return (const Clause&)RegionAllocator<uint32_t>::operator[](r); }
237  Clause* lea (Ref r) { return reinterpret_cast<Clause*>(RegionAllocator<uint32_t>::lea(r)); }
238  const Clause* lea(Ref r) const { return reinterpret_cast<const Clause*>(RegionAllocator<uint32_t>::lea(r)); }
239  Ref ael (const Clause* t) { return RegionAllocator<uint32_t>::ael((const uint32_t*)t); }
240 
241  void free(CRef cid)
242  {
243  Clause& c = operator[](cid);
244  RegionAllocator<uint32_t>::free(clauseWord32Size(c.size(), c.has_extra()));
245  }
246 
247  void reloc(CRef& cr, ClauseAllocator& to)
248  {
249  Clause& c = operator[](cr);
250 
251  if (c.reloced()) { cr = c.relocation(); return; }
252 
253  cr = to.alloc(c, c.learnt());
254  c.relocate(cr);
255 
256  // Copy extra data-fields:
257  // (This could be cleaned-up. Generalize Clause-constructor to be applicable here instead?)
258  to[cr].mark(c.mark());
259  if (to[cr].learnt()) to[cr].activity() = c.activity();
260  else if (to[cr].has_extra()) to[cr].calcAbstraction();
261  }

302 void OccLists<Idx,Vec,Deleted>::cleanAll()

312 template<class Idx, class Vec, class Deleted>
313 void OccLists<Idx,Vec,Deleted>::clean(const Idx& idx)
314 {
315  Vec& vec = occs[toInt(idx)];
316  int i, j;
317  for (i = j = 0; i < vec.size(); i++)
318  if (!deleted(vec[i]))
319  vec[j++] = vec[i];
320  vec.shrink(i - j);
321  dirty[toInt(idx)] = 0;
322 }

367 /*_________________________________________________________________________________________________
368 |
369 | subsumes : (other : const Clause&) -> Lit
370 |
371 | Description:
372 | Checks if clause subsumes 'other', and at the same time, if it can be used to simplify 'other'
373 | by subsumption resolution.
374 |
375 | Result:
376 | lit_Error - No subsumption or simplification
377 | lit_Undef - Clause subsumes 'other'
378 | p - The literal p can be deleted from 'other'
379 |________________________________________________________________________________________________@*/
380 inline Lit Clause::subsumes(const Clause& other) const
381 {
382  //if (other.size() < size() || (extra.abst & ~other.extra.abst) != 0)
383  //if (other.size() < size() || (!learnt() && !other.learnt() && (extra.abst & ~other.extra.abst) != 0))
384  assert(!header.learnt); assert(!other.header.learnt);
385  assert(header.has_extra); assert(other.header.has_extra);
386  if (other.header.size < header.size || (data[header.size].abs & ~other.data[other.header.size].abs) != 0)
387  return lit_Error;
388 
389  Lit ret = lit_Undef;
390  const Lit* c = static_cast<const Lit*>(*this);
391  const Lit* d = static_cast<const Lit*>(other);
392 
393  for (unsigned i = 0; i < header.size; i++) {
394  // search for c[i] or ~c[i]
395  for (unsigned j = 0; j < other.header.size; j++)
396  if (c[i] == d[j])
397  goto ok;
398  else if (ret == lit_Undef && c[i] == ~d[j]){
399  ret = c[i];
400  goto ok;
401  }
402 
403  // did not find it
404  return lit_Error;
405  ok:;
406  }
407 
408  return ret;
409 }
