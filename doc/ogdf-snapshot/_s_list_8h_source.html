1323 template<class E>
1324 void SListPure<E>::bucketSort(int l, int h, BucketFunc<E> &f)
1325 {
1326  // if less than two elements, nothing to do
1327  if (m_head == m_tail) return;
1328 
1329  Array<SListElement<E> *> head(l,h,nullptr), tail(l,h);
1330 
1331  SListElement<E> *pX;
1332  for (pX = m_head; pX; pX = pX->m_next) {
1333  int i = f.getBucket(pX->m_x);
1334  if (head[i])
1335  tail[i] = (tail[i]->m_next = pX);
1336  else
1337  head[i] = tail[i] = pX;
1338  }
1339 
1340  SListElement<E> *pY = nullptr;
1341  for (int i = l; i <= h; i++) {
1342  pX = head[i];
1343  if (pX) {
1344  if (pY)
1345  pY->m_next = pX;
1346  else
1347  m_head = pX;
1348  pY = tail[i];
1349  }
1350  }
1351 
1352  m_tail = pY;
1353  pY->m_next = nullptr;
1354 }
