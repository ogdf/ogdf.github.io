974 {
975  // if less than two elements, nothing to do
976  if (m_head == m_tail) return;

978  int l, h;
979  l = h = f.getBucket(m_head->m_x);
980 
981  SListElement<E> *pX;
982  for(pX = m_head->m_next; pX; pX = pX->m_next)

984  int i = f.getBucket(pX->m_x);
985  if (i < l) l = i;
986  if (i > h) h = i;

992 template<class E>
993 void SListPure<E>::bucketSort(int l, int h, BucketFunc<E> &f)
994 {
995  // if less than two elements, nothing to do
996  if (m_head == m_tail) return;
997 
998  Array<SListElement<E> *> head(l,h,nullptr), tail(l,h);
999 
1000  SListElement<E> *pX;
1001  for (pX = m_head; pX; pX = pX->m_next) {
1002  int i = f.getBucket(pX->m_x);
1003  if (head[i])
