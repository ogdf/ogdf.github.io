43 template<typename T>
44 class BlowupComponents
45 {
46 protected:
47  // To represent Gamma(X) [the set of all components in the blowup graph], we give
48  // - terminals, source and target the component id 0
49  // - all other nodes a component id > 0. Nodes with same id belong to the same component.
50  // Note that this is also fine for 2-components with only one edge, since this
51  // is a core edge and hence a dummy node is inserted.
52  NodeArray<int> componentId;
53  // We also save lists of terminals for each component in the specified array buffer.
54  ArrayBuffer<ArrayBuffer<node>> componentTerminals;
55  // To efficiently traverse the found component, we save the edge from the root of the component
56  ArrayBuffer<edge> componentRootEdge;
57  // Finally a component -> cost array.
58  ArrayBuffer<T> componentCost;
59 
60  int maxId; // == the size of the arraybuffers
61 
63  void initializeComponent(edge rootEdge, const BlowupGraph<T> &blowupGraph)
64  {
65  List<node> queue;
66  const node start = rootEdge->target();
67  queue.pushBack(start);
68  componentRootEdge.push(rootEdge);
69  componentTerminals.push(ArrayBuffer<node>());
70  auto &terms = componentTerminals[maxId];
71  if (!blowupGraph.getOriginal(start)) { // start node is a core edge
72  componentCost.push(blowupGraph.getCost(rootEdge));
73  } else {
74  componentCost.push(0);
75  }
76  T &cost = componentCost[maxId];
77  ++maxId;
78  while (!queue.empty()) {
79  const node v = queue.popBackRet();
80  componentId[v] = maxId;
81  for (adjEntry adj : v->adjEntries) {
82  const node w = adj->twinNode();
83  if (componentId[w] < 0) {
84  // count coreEdge cost only once
85  if (blowupGraph.getOriginal(v)) { // v is no core edge
86  cost += blowupGraph.getCost(adj->theEdge());
87  }
88  if (blowupGraph.isTerminal(w)) {
89  terms.push(w);
90  } else {
91  queue.pushBack(w);
92  }
93  }
94  }
95  }
96 #ifdef OGDF_STEINER_TREE_GOEMANS_BLOWUP_COMPONENTS_LOGGING
97  std::cout
98  << " * component with terminals " << terms
99  << " starting with edge " << rootEdge
100  << " having cost " << cost
101  << " and capacity " << blowupGraph.getCapacity(rootEdge) << std::endl;
102 #endif
103  }
104 
105 public:
107  BlowupComponents(const BlowupGraph<T> &blowupGraph)
108  : componentId(blowupGraph.getGraph(), -1)
109  , maxId(0)
110  {
111  componentId[blowupGraph.getSource()] = 0;
112  componentId[blowupGraph.getPseudotarget()] = 0;
113  componentId[blowupGraph.getTarget()] = 0;
114 
115 #ifdef OGDF_STEINER_TREE_GOEMANS_BLOWUP_COMPONENTS_LOGGING
116  std::cout << "Finding components in blowup graph:" << std::endl;
117 #endif
118  for (node t : blowupGraph.terminals()) {
119  for (adjEntry rootAdj : t->adjEntries) {
120  const edge rootEdge = rootAdj->theEdge();
121  if (rootEdge->source() != t) {
122  continue;
123  }
124  const node r = rootAdj->twinNode();
125  OGDF_ASSERT(r == rootEdge->target());
126  if (componentId[r] < 0) {
127  initializeComponent(rootEdge, blowupGraph);
128  }
129  }
130  }
131 
132  // now set terminals to id 0 [XXX: why dont we keep -1?]
133  for (node t : blowupGraph.terminals()) {
134  componentId[t] = 0;
135  }
136  }
