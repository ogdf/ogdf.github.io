184 template<typename T>
185 void MinSteinerTreeMehlhorn<T>::calculateCompleteGraph(const EdgeWeightedGraph<T> &wG,
186  const List<node> &terminals,
187  const Voronoi<T> &voronoi,
188  EdgeArray<edge> &bridges,
189  EdgeWeightedGraphCopy<T> &completeTerminalGraph)
190 {
191  completeTerminalGraph.createEmpty(wG);
192 
193  for (node v : terminals) {
194  completeTerminalGraph.newNode(v);
195  }
196  if (completeTerminalGraph.numberOfNodes() <= 1) {
197  bridges.init(completeTerminalGraph);
198  return;
199  }
200 
201  // extract complete graph edges
202  List<MehlhornTriple> triples;
203  for(edge e : wG.edges) {
204  MehlhornTriple triple;
205  triple.u = voronoi.seed(e->source());
206  triple.v = voronoi.seed(e->target());
207  if (triple.u != triple.v) {
208  triple.value = voronoi.distance(e->source()) + voronoi.distance(e->target()) + wG.weight(e);
209  triple.bridge = e;
210  triples.pushBack(triple);
211  }
212  }
213 
214  MehlhornTripleBucketMaxFunc mtbMax;
215  MehlhornTripleBucketMinFunc mtbMin;
216 
217  triples.bucketSort(0, wG.maxNodeIndex(), mtbMax);
218  triples.bucketSort(0, wG.maxNodeIndex(), mtbMin);
219 
220  int currentSource = triples.front().u->index();
221  int currentTarget = triples.front().v->index();
222  ListConstIterator<MehlhornTriple> minTriple = triples.begin();
223 
224  bridges.init(completeTerminalGraph);
225 
226  for (ListConstIterator<MehlhornTriple> mtIt = triples.begin().succ(); mtIt.valid(); ++mtIt) {
227  if (((*mtIt).u->index() == currentSource && (*mtIt).v->index() == currentTarget)
228  || ((*mtIt).u->index() == currentTarget && (*mtIt).v->index() == currentSource)) {
229  if ((*mtIt).value < (*minTriple).value) {
230  minTriple = mtIt;
231  }
232  } else {
233  // add new direct edge
234  edge tmp = completeTerminalGraph.newEdge(completeTerminalGraph.copy((*minTriple).u),
235  completeTerminalGraph.copy((*minTriple).v), (*minTriple).value);
236  bridges[tmp] = (*minTriple).bridge;
237 
238  currentSource = (*mtIt).u->index();
239  currentTarget = (*mtIt).v->index();
240  minTriple = mtIt;
241  }
242  }
243  // insert last triple
244  if (minTriple.valid()) {
245  edge tmp = completeTerminalGraph.newEdge(completeTerminalGraph.copy((*minTriple).u),
246  completeTerminalGraph.copy((*minTriple).v), (*minTriple).value);
247  bridges[tmp] = (*minTriple).bridge;
248  }
249 }

251 template<typename T>
252 void MinSteinerTreeMehlhorn<T>::reinsertShortestPaths(EdgeWeightedGraphCopy<T> &completeTerminalGraph,
253  const Voronoi<T> &voronoi,
254  const NodeArray<edge> &mstPred,
255  const EdgeArray<edge> &bridges,
256  EdgeWeightedGraphCopy<T> &finalSteinerTree,
257  const EdgeWeightedGraph<T> &wG)
258 {
259  for(node u : completeTerminalGraph.nodes) {
260  if (mstPred[u] != nullptr) {
261  edge bridge = bridges[mstPred[u]];
262  node v = bridge->source();
263  node w = bridge->target();
264  insertPath(v, voronoi, finalSteinerTree, wG);
265  insertPath(w, voronoi, finalSteinerTree, wG);
266  edge e = finalSteinerTree.newEdge(finalSteinerTree.copy(v), finalSteinerTree.copy(w), wG.weight(bridge));
267  finalSteinerTree.setEdge(bridge, e);
268  }
269  }

272 template<typename T>
273 void MinSteinerTreeMehlhorn<T>::insertPath(node u,
274  const Voronoi<T> &voronoi,
275  EdgeWeightedGraphCopy<T> &finalSteinerTree,
276  const EdgeWeightedGraph<T> &wG)
277 {
278  node currentSource;
279  node currentTarget = finalSteinerTree.copy(u);
280  if (!currentTarget) {
281  currentTarget = finalSteinerTree.newNode(u);
282  }
283  edge e = voronoi.predecessorEdge(u);
284  edge newE;
285  while (e && finalSteinerTree.chain(e).empty()) { // e is not in ST yet
286  if ((currentSource = finalSteinerTree.copy(e->opposite(finalSteinerTree.original(currentTarget)))) == nullptr) {
287  currentSource = finalSteinerTree.newNode(e->opposite(finalSteinerTree.original(currentTarget)));
288  }
289  if (finalSteinerTree.original(currentSource) == e->source()) {
290  newE = finalSteinerTree.newEdge(currentSource, currentTarget, wG.weight(e));
291  } else {
292  newE = finalSteinerTree.newEdge(currentTarget, currentSource, wG.weight(e));
293  }
294  finalSteinerTree.setEdge(e, newE);
295  currentTarget = currentSource;
296  e = voronoi.predecessorEdge(finalSteinerTree.original(currentTarget));
297  }
298 }
