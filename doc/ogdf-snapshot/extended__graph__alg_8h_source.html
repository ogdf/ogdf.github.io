282 template<typename T>
283 void computeMinST(node s, const Graph &G, const EdgeArray<T> &weight, NodeArray<edge> &pred)
284 {
285  PrioritizedMapQueue<node, T> pq(G); // priority queue of front vertices
286 
287  // insert start node
288  T tmp(0);
289  pq.push(s, tmp);
290 
291  // extract the nodes again along a minimum ST
292  NodeArray<bool> processed(G, false);
293  pred.init(G, nullptr);
294 
295  while (!pq.empty()) {
296  const node v = pq.topElement();
297  pq.pop();
298  processed[v] = true;
299  for (adjEntry adj = v->firstAdj(); adj; adj = adj->succ()) {
300  const node w = adj->twinNode();
301  const edge e = adj->theEdge();
302  if (pred[w] == nullptr && w != s) {
303  tmp = weight[e];
304  pq.push(w, tmp);
305  pred[w] = e;
306  } else
307  if (!processed[w]
308  && weight[e] < pq.priority(w)) {
309  pq.decrease(w, weight[e]);
310  pred[w] = e;
311  }
312  }
313  }
314 }

327 T computeMinST(node s, const Graph &G, const EdgeArray<T> &weight, NodeArray<edge> &pred, EdgeArray<bool> &isInTree)
328 {
329  computeMinST(s, G, weight, pred);
330 
331  // now just compute isInTree and total weight
332  int rootcount = 0;
333  T treeWeight = 0;
334  isInTree.init(G, false);
335  for (node v = G.firstNode(); v; v = v->succ()) {
336  if (!pred[v]) {
337  ++rootcount;
338  } else {
339  isInTree[pred[v]] = true;
340  treeWeight += weight[pred[v]];
341  }
342  }
343  OGDF_ASSERT(rootcount == 1); // is connected
344 
345  return treeWeight;

358 template<typename T>
359 T makeMinimumSpanningTree(Graph &G, const EdgeArray<T> &weight)
360 {
361  T total(0);
362  List< Prioritized<edge, T> > sortEdges;
363  for (edge e = G.firstEdge(); e; e = e->succ()) {
364  sortEdges.pushBack(Prioritized<edge,T>(e, weight[e]));
365  }
366  sortEdges.quicksort();
367 
368  // now let's do Kruskal's algorithm
369  NodeArray<int> setID(G);
370  DisjointSets<> uf(G.numberOfNodes());
371  for (node v = G.firstNode(); v; v = v->succ()) {
372  setID[v] = uf.makeSet();
373  }
374 
375  for (ListConstIterator< Prioritized<edge,T> > it = sortEdges.begin(); it.valid(); ++it) {
376  const edge e = (*it).item();
377  const int v = setID[e->source()];
378  const int w = setID[e->target()];
379  if (uf.find(v) != uf.find(w)) {
380  uf.link(uf.find(v), uf.find(w));
381  total += weight[e];
382  } else {
383  G.delEdge(e);
384  }
385  }
386  return total;
387 }
