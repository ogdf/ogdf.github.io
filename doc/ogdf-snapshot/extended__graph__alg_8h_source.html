279 template<typename T>
280 void computeMinST(node s, const Graph &G, const EdgeArray<T> &weight, NodeArray<edge> &pred)
281 {
282  PrioritizedMapQueue<node, T> pq(G); // priority queue of front vertices
283 
284  // insert start node
285  T tmp(0);
286  pq.push(s, tmp);
287 
288  // extract the nodes again along a minimum ST
289  NodeArray<bool> processed(G, false);
290  pred.init(G, nullptr);
291 
292  while (!pq.empty()) {
293  const node v = pq.topElement();
294  pq.pop();
295  processed[v] = true;
296  for (adjEntry adj = v->firstAdj(); adj; adj = adj->succ()) {
297  const node w = adj->twinNode();
298  const edge e = adj->theEdge();
299  if (pred[w] == nullptr && w != s) {
300  tmp = weight[e];
301  pq.push(w, tmp);
302  pred[w] = e;
303  } else
304  if (!processed[w]
305  && weight[e] < pq.priority(w)) {
306  pq.decrease(w, weight[e]);
307  pred[w] = e;
308  }
309  }
310  }
311 }

323 template<typename T>
324 T computeMinST(node s, const Graph &G, const EdgeArray<T> &weight, NodeArray<edge> &pred, EdgeArray<bool> &isInTree)
325 {
326  computeMinST(s, G, weight, pred);
327 
328  // now just compute isInTree and total weight
329  int rootcount = 0;
330  T treeWeight = 0;
331  isInTree.init(G, false);
332  for (node v = G.firstNode(); v; v = v->succ()) {
333  if (!pred[v]) {
334  ++rootcount;
335  } else {
336  isInTree[pred[v]] = true;
337  treeWeight += weight[pred[v]];
338  }
339  }
340  OGDF_ASSERT(rootcount == 1); // is connected
341 
342  return treeWeight;
343 }

354 template<typename T>
355 T makeMinimumSpanningTree(Graph &G, const EdgeArray<T> &weight)
356 {
357  T total(0);
358  Array<Prioritized<edge, T>> sortEdges(G.numberOfEdges());
359  int i = 0;
360  for (edge e : G.edges) {
361  sortEdges[i++] = Prioritized<edge,T>(e, weight[e]);
362  }
363  sortEdges.quicksort();
364 
365  // now let's do Kruskal's algorithm
366  NodeArray<int> setID(G);
367  DisjointSets<> uf(G.numberOfNodes());
368  for (node v : G.nodes) {
369  setID[v] = uf.makeSet();
370  }
371 
372  for (auto prioEdge : sortEdges) {
373  const edge e = prioEdge.item();
374  const int v = setID[e->source()];
375  const int w = setID[e->target()];
376  if (uf.find(v) != uf.find(w)) {
377  uf.link(uf.find(v), uf.find(w));
378  total += weight[e];
379  } else {
380  G.delEdge(e);
381  }
382  }
383  return total;
384 }
