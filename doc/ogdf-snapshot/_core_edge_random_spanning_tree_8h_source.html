56  {
57  // Let's do Kruskal's algorithm without weights but on a randomly permuted edge list.
58  // We virtually contract all terminals in the union-find data structure.
59  NodeArray<int> setID(graph, -1);
60  isInTree.init(graph, false);
61  DisjointSets<> uf(graph.numberOfNodes() - terminals.size() + 1);
62 
63  int contractedID = uf.makeSet();
64  OGDF_ASSERT(contractedID >= 0);
65  for (node t : terminals) {
66  setID[t] = contractedID;
67  }
68  for (node v : graph.nodes) {
69  if (setID[v] < 0) {
70  setID[v] = uf.makeSet();
71  }
72  }
73 
74  // obtain a random edge permutation
75  ArrayBuffer<edge> edgePermutation;
76  for (edge e : graph.edges) {
77  edgePermutation.push(e);
78  }
79  edgePermutation.permute(m_rng);
80 
81  // add edges if we do not close a cycle
82  for (edge e : edgePermutation) {
83  const int v = setID[e->source()];
84  const int w = setID[e->target()];
85  if (uf.find(v) != uf.find(w)) {
86  isInTree[e] = true;
87  uf.link(uf.find(v), uf.find(w));
