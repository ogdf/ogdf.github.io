580

604  if (nodePtr->getNodeInfo()->userStructInfo()->m_pertLeafCount < maxPertLeafCount)

607  // [[nodePtr]] is not the root, the counts of the pertinent
608  // children of [[nodePtr]]'s parent are
609  // updated. In case that all pertinent children of the parent of
610  // [[nodePtr]] have a valid $[w,h,a]$-number, it is possible to compute

616  nodePtr->parent()->getNodeInfo()->userStructInfo()->m_notVisitedCount--;
617  if (!nodePtr->parent()->getNodeInfo()->userStructInfo()->m_notVisitedCount)
618  {

620  archiv.push(nodePtr->parent());
621  }
622  }

624  {

626  nodePtr->status(PQNodeRoot::FULL);
627  nodePtr->getNodeInfo()->userStructInfo()->m_w = 1;
628  nodePtr->getNodeInfo()->userStructInfo()->m_h = 0;
629  nodePtr->getNodeInfo()->userStructInfo()->m_a = 0;
630  if (nodePtr->getNodeInfo()->userStructInfo()->m_pertLeafCount < maxPertLeafCount)
631  fullChildren(nodePtr->parent())->pushFront(nodePtr);
632  }
633  else
634  {

636  // The $[w,h,a]$ number of $P$- or $Q$-nodes is computed identically.
637  // This is done calling the function [[sumPertChild]].
638  nodePtr->getNodeInfo()->userStructInfo()->m_w = sumPertChild(nodePtr);

643  // is trivial. It also updates the list of full nodes of the parent
644  // of [[nodePtr]].
645  nodePtr->status(PQNodeRoot::FULL);

649  nodePtr->getNodeInfo()->userStructInfo()->m_h = 0;
650  nodePtr->getNodeInfo()->userStructInfo()->m_a = 0;
651  }
652  else

655  // nontrivial for both $P$- and $Q$-nodes and is performed in the
656  // function [[haNumPnode]] for $P$-nodes and in the
657  // function [[haNumQnode]] for $Q$-nodes.
658  // This chunk also updates the partial children stack of the parent.
659  nodePtr->status(PQNodeRoot::PARTIAL);

662  partialChildren(nodePtr->parent())->pushFront(nodePtr);
663 
664  if (nodePtr->type() == PQNodeRoot::PNode)
665  haNumPnode(nodePtr);
666  else
667  haNumQnode(nodePtr);
668  }
669  }
670 
671  }

674  //[[nodePtr]] and finds the minimum of the $h$- and $a$-number of
675  //[[m_pertinentRoot]]. In case that the minimum is equal to $0$, the
676  //[[m_pertinentRoot]] stays of type $B$. Otherwise the type is selected
677  //according to the minimum.
678  this->m_pertinentRoot = nodePtr;
679  if (this->m_pertinentRoot->getNodeInfo()->userStructInfo()->m_h <
680  this->m_pertinentRoot->getNodeInfo()->userStructInfo()->m_a)
681  countDeletedLeaves = this->m_pertinentRoot->getNodeInfo()->userStructInfo()->m_h;

684

688  this->m_pertinentRoot->getNodeInfo()->userStructInfo()->m_a)
689  this->m_pertinentRoot->getNodeInfo()->userStructInfo()->m_deleteType = H_TYPE;
690  else
691  this->m_pertinentRoot->getNodeInfo()->userStructInfo()->m_deleteType = A_TYPE;
692  }
693 
694  findMinWHASequence(archiv,eliminatedKeys);
695 
696  return countDeletedLeaves;
697 }

733 
734  //a pointer to the second child of type $h$ of [[nodePtr]].
735  PQNode<T, whaInfo*, Y> *hChild2 = 0;
736 
737  //a pointer to the child of type $a$ of [[nodePtr]].

754  frontier, the node must therefore keep all its pertinent leaves

769  */

773  eliminatedKeys.pushBack(nodePtr->getKey());

782  */

802  the default is type $b$). It furthermore marks all partial children to

805  marked to be of type $h$.
806  */

808  markPertinentChildren(nodePtr, PQNodeRoot::FULL, B_TYPE);
809  if (nodePtr->getNodeInfo()->userStructInfo()->m_hChild1)

1242  \[\max_{i\in Par(\mbox{[[nodePtr]]})}1\left\{(w_i - h_i)\right\}.\]
1243  Since this can be simultaneously it also computes
1244  \[\max_{i\in Par(\mbox{[[nodePtr]]})}2\left\{(w_i - h_i)\right\}\]

1268  }
