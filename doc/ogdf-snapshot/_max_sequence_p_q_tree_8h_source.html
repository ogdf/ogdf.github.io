394 template<class T,class Y>
395 bool MaxSequencePQTree<T,Y>::Bubble(SListPure<PQLeafKey<T,whaInfo*,Y>*> &leafKeys)
396 {
397  // Queue for storing all pertinent nodes that still have
398  // to be processed.
399  Queue<PQNode<T,whaInfo*,Y>*> processNodes;
400 
401  /*
402  Enter the [[Full]] leaves into the queue [[processNodes]].
403  In a first step the pertinent leaves have to be identified in the tree
404  and entered on to the queue [[processNodes]]. The identification of
405  the leaves can be done with the help of a pointer stored in every
406  [[PQLeafKey]] (see \ref{PQLeafKey}) in constant time for every element.
407  */
408  SListIterator<PQLeafKey<T,whaInfo*,Y>*> it;
409  for (it = leafKeys.begin(); it.valid(); ++it)
410  {
411  PQNode<T,whaInfo*,Y>* checkLeaf = (*it)->nodePointer();
412  processNodes.append(checkLeaf);
413  cleanUp.pushBack(checkLeaf);
414  if (!checkLeaf->getNodeInfo()) // if leaf does not have an information
415  // class for storing the [wha]-number
416  // allocate one.
417  {
418  whaInfo *newInfo = new whaInfo;
419  PQNodeKey<T,whaInfo*,Y> *infoPtr = new PQNodeKey<T,whaInfo*,Y>(newInfo);
420  checkLeaf->setNodeInfo(infoPtr);
421  infoPtr->setNodePointer(checkLeaf);
422  }
423  checkLeaf->getNodeInfo()->userStructInfo()->m_notVisitedCount = 1;
424  checkLeaf->mark(PQNodeRoot::PQNodeMark::Queued);
425  }
426 
427  /*
428  For every node in [[processNodes]], its father is detected using the
429  function [[GetParent]] \ref{GetParent}. The father is placed onto the
430  queue as well, if [[_nodePtr]] is its first child, that is popped from
431  the queue. In that case, the father is marked as [[Queued]] to
432  prevent the node from queue more than once. In any case, the number
433  of pertinent children of the father is updated. This is an important
434  number for computing the maximal pertinent sequence.
435  */
436  while (!processNodes.empty())
437  {
438  PQNode<T,whaInfo*,Y>* nodePtr = processNodes.pop();
439  nodePtr->parent(GetParent(nodePtr));
440  if (nodePtr->parent() &&
441  !nodePtr->parent()->getNodeInfo())
442  // if the parent does not have an information
443  // class for storing the [wha]-number
444  // allocate one.
445  {
446  whaInfo *newInfo = new whaInfo;
447  PQNodeKey<T,whaInfo*,Y> *infoPtr = new PQNodeKey<T,whaInfo*,Y>(newInfo);
448  nodePtr->parent()->setNodeInfo(infoPtr);
449  infoPtr->setNodePointer(nodePtr->parent());
450  }
451  if (nodePtr != this->m_root)
452  {
453  if (nodePtr->parent()->mark() == PQNodeRoot::PQNodeMark::Unmarked)
454  {
455  processNodes.append(nodePtr->parent());
456  cleanUp.pushBack(nodePtr->parent());
457  nodePtr->parent()->mark(PQNodeRoot::PQNodeMark::Queued);
458  }
459  nodePtr->parent()->getNodeInfo()->userStructInfo()->m_notVisitedCount++;
460  int childCount = nodePtr->parent()->pertChildCount();
461  nodePtr->parent()->pertChildCount(++childCount);
462  }
463  }
464 
465 
466  /*
467  This chunk belongs to the function [[Bubble]]. After doing some
468  cleaning up and resetting the flags that have been left at the
469  pertinent nodes during the first bubble up, this chunk computes the
470  maximal pertinent sequence by calling the function [[determineMinRemoveSequence]].
471  It then removes
472  all leaves from the tree that have been marked as [[Eliminated]] and
473  possibly some internal nodes of the $PQ$-tree and stores pointers of
474  type [[leafKey]] for the pertinent leaves in the maximal sequence in the
475  array [[survivedElements]].
476  */
477 
478  SListIterator<PQNode<T,whaInfo*,Y>*> itn;
479  for (itn = cleanUp.begin(); itn.valid(); ++itn)
480  (*itn)->mark(PQNodeRoot::PQNodeMark::Unmarked);
481 
482  return true;

555 template<class T,class Y>
556 int MaxSequencePQTree<T,Y>::determineMinRemoveSequence(
557  SListPure<PQLeafKey<T,whaInfo*,Y>*> &leafKeys,
558  SList<PQLeafKey<T,whaInfo*,Y>*> &eliminatedKeys)
559 {
560  PQNode<T,whaInfo*,Y> *nodePtr = nullptr; // dummy
561 
562  //Number of leaves that have to be deleted
563  int countDeletedLeaves = 0;
564  //Number of pertinent leaves
565  int maxPertLeafCount = 0;
566 
567  // A queue storing the nodes whose $[w,h,a]$-number has to be computed next.
568  // A node is stored in [[processNodes]], if for all of its children the
569  // [w,h,a]-number has been computed.
570  Queue<PQNode<T,whaInfo*,Y>*> processNodes;
571 
572  // A stack storing all nodes where a $[w,h,a]$-number
573  // has been computed. This is necessary for a valid cleanup.
574  ArrayBuffer<PQNode<T,whaInfo*,Y>*> archiv;
575 
576  // Compute a valid parent pointer for every pertinent node.
577  Bubble(leafKeys);
578 
579  // Get all pertinent leaves and stores them on [[processNodes]]
580  // and [[_archiv]].
581  SListIterator<PQLeafKey<T,whaInfo*,Y>*> it;
582  for (it = leafKeys.begin(); it.valid(); ++it)
583  {
584  PQNode<T, whaInfo*, Y> *checkLeaf = (*it)->nodePointer();
585  checkLeaf->getNodeInfo()->userStructInfo()->m_pertLeafCount = 1;
586  checkLeaf->getNodeInfo()->userStructInfo()->m_notVisitedCount--;
587  processNodes.append(checkLeaf);
588  archiv.push(checkLeaf);
589 
590  maxPertLeafCount++;
591  }
592 
593  while (!processNodes.empty())
594  {
595  nodePtr = processNodes.pop();
596  // Compute the $[w,h,a]$ number of [[nodePtr]]. Computing this number is
597  // trivial for leaves and full nodes. When considering a partial node, the
598  // computation has to distinguish between $P$- and $Q$-nodes.
599  if (nodePtr->getNodeInfo()->userStructInfo()->m_pertLeafCount < maxPertLeafCount)
600  {
601  // In case that the actual node, depicted by the pointer
602  // [[nodePtr]] is not the root, the counts of the pertinent
603  // children of [[nodePtr]]'s parent are
604  // updated. In case that all pertinent children of the parent of
605  // [[nodePtr]] have a valid $[w,h,a]$-number, it is possible to compute
606  // the $[w,h,a]$-number of parent. Hence the parent is placed onto
607  // [[processNodes]]and [[_archiv]].
608  nodePtr->parent()->getNodeInfo()->userStructInfo()->m_pertLeafCount =
609  nodePtr->parent()->getNodeInfo()->userStructInfo()->m_pertLeafCount
610  + nodePtr->getNodeInfo()->userStructInfo()->m_pertLeafCount;
611  nodePtr->parent()->getNodeInfo()->userStructInfo()->m_notVisitedCount--;
612  if (!nodePtr->parent()->getNodeInfo()->userStructInfo()->m_notVisitedCount)
613  {
614  processNodes.append(nodePtr->parent());
615  archiv.push(nodePtr->parent());
616  }
617  }
618  if (nodePtr->type() == PQNodeRoot::PQNodeType::Leaf)
619  {
620  // Compute the $[w,h,a]$-number of a leaf. The computation is trivial.
621  nodePtr->status(PQNodeRoot::PQNodeStatus::Full);
622  nodePtr->getNodeInfo()->userStructInfo()->m_w = 1;
623  nodePtr->getNodeInfo()->userStructInfo()->m_h = 0;
624  nodePtr->getNodeInfo()->userStructInfo()->m_a = 0;
625  if (nodePtr->getNodeInfo()->userStructInfo()->m_pertLeafCount < maxPertLeafCount)
626  fullChildren(nodePtr->parent())->pushFront(nodePtr);
627  }
628  else
629  {
630  // [[nodePtr]] is a $P$- or $Q$-node
631  // The $[w,h,a]$ number of $P$- or $Q$-nodes is computed identically.
632  // This is done calling the function [[sumPertChild]].
633  nodePtr->getNodeInfo()->userStructInfo()->m_w = sumPertChild(nodePtr);
634 
635  if (fullChildren(nodePtr)->size() == nodePtr->childCount())
636  {
637  // computes the $h$- and $a$-numbers of a full node. The computation
638  // is trivial. It also updates the list of full nodes of the parent
639  // of [[nodePtr]].
640  nodePtr->status(PQNodeRoot::PQNodeStatus::Full);
641  if (nodePtr->getNodeInfo()->userStructInfo()->m_pertLeafCount
642  < maxPertLeafCount)
643  fullChildren(nodePtr->parent())->pushFront(nodePtr);
644  nodePtr->getNodeInfo()->userStructInfo()->m_h = 0;
645  nodePtr->getNodeInfo()->userStructInfo()->m_a = 0;
646  }
647  else
648  {
649  // computes the $[w,h,a]$-number of a partial node. The computation is
650  // nontrivial for both $P$- and $Q$-nodes and is performed in the
651  // function [[haNumPnode]] for $P$-nodes and in the
652  // function [[haNumQnode]] for $Q$-nodes.
653  // This chunk also updates the partial children stack of the parent.
654  nodePtr->status(PQNodeRoot::PQNodeStatus::Partial);
655  if (nodePtr->getNodeInfo()->userStructInfo()->m_pertLeafCount <
656  maxPertLeafCount)
657  partialChildren(nodePtr->parent())->pushFront(nodePtr);
658 
659  if (nodePtr->type() == PQNodeRoot::PQNodeType::PNode)
660  haNumPnode(nodePtr);
661  else
662  haNumQnode(nodePtr);
663  }
664  }
665  }
666 
667  // Determine the root of the pertinent subtree, which the last processed node
668  //[[nodePtr]] and finds the minimum of the $h$- and $a$-number of
669  //[[m_pertinentRoot]]. In case that the minimum is equal to $0$, the
670  //[[m_pertinentRoot]] stays of type $B$. Otherwise the type is selected
671  //according to the minimum.
672  this->m_pertinentRoot = nodePtr;
673  if (this->m_pertinentRoot->getNodeInfo()->userStructInfo()->m_h <
674  this->m_pertinentRoot->getNodeInfo()->userStructInfo()->m_a)
675  countDeletedLeaves = this->m_pertinentRoot->getNodeInfo()->userStructInfo()->m_h;
676  else
677  countDeletedLeaves = this->m_pertinentRoot->getNodeInfo()->userStructInfo()->m_a;
678 
679  if (countDeletedLeaves > 0)
680  {
681  if (this->m_pertinentRoot->getNodeInfo()->userStructInfo()->m_h <
682  this->m_pertinentRoot->getNodeInfo()->userStructInfo()->m_a)
683  this->m_pertinentRoot->getNodeInfo()->userStructInfo()->m_deleteType = whaType::H;
684  else
685  this->m_pertinentRoot->getNodeInfo()->userStructInfo()->m_deleteType = whaType::A;
686  }
687 
688  findMinWHASequence(archiv,eliminatedKeys);
689 
690  return countDeletedLeaves;
691 }

693 template<class T, class Y>
694 void MaxSequencePQTree<T, Y>::findMinWHASequence(
695  ArrayBuffer<PQNode<T, whaInfo*, Y>*> &archiv,
696  SList<PQLeafKey<T, whaInfo*, Y>*> &eliminatedKeys)
697 {
698  //a pointer to the first child of type $h$ of [[nodePtr]].
699  PQNode<T, whaInfo*, Y> *hChild1 = nullptr;
700 
701  //a pointer to the second child of type $h$ of [[nodePtr]].
702  PQNode<T, whaInfo*, Y> *hChild2 = nullptr;
703 
704  //a pointer to the child of type $a$ of [[nodePtr]].
705  PQNode<T, whaInfo*, Y> *aChild = nullptr;
706 
707  // pertinent sibling of hChild1
708  PQNode<T, whaInfo*, Y> *hChild2Sib = nullptr;
709 
710  while (!archiv.empty())
711  {
712  //counts the number of children of [[nodePtr]].
713  int childCount = 0;
714 
715  //a pointer to a pertinent node of the $PQ$-tree that is currently examined.
716  PQNode<T, whaInfo*, Y> *nodePtr = archiv.popRet();
717 
718  /*
719  Check if [[nodePtr]] is a full node whose delete type is either of
720  type $h$ or type $a$. Since there are no empty leaves in its
721  frontier, the node must therefore keep all its pertinent leaves
722  in its frontier and is depicted to be of type $b$.
723  */
724  if (nodePtr->status() == PQNodeRoot::PQNodeStatus::Full &&
725  (nodePtr->getNodeInfo()->userStructInfo()->m_deleteType == whaType::H ||
726  nodePtr->getNodeInfo()->userStructInfo()->m_deleteType == whaType::A))
727  {
728  nodePtr->getNodeInfo()->userStructInfo()->m_deleteType = whaType::B;
729  this->m_pertinentNodes->pushFront(nodePtr);
730  }
731 
732  /*
733  Check if [[nodePtr]] is a leaf whose delete type is either of type $w$ or
734  $b$. In case it is of type $w$, the leaf has to be removed from the
735  tree to gain reducibility of the set $S$.
736  */
737  else if (nodePtr->type() == PQNodeRoot::PQNodeType::Leaf)
738  {
739  if (nodePtr->getNodeInfo()->userStructInfo()->m_deleteType == whaType::W)
740  eliminatedKeys.pushBack(nodePtr->getKey());
741  else
742  this->m_pertinentNodes->pushFront(nodePtr);
743  }
744 
745  /*
746  Manage the case of [[nodePtr]] being either a partial $P$-node, a partial
747  $Q$-node, or a full $P$- or $Q$-node, where in the latter case the
748  delete type of [[nodePtr]] is of type $b$.
749  */
750  else
751  switch (nodePtr->getNodeInfo()->userStructInfo()->m_deleteType)
752  {
753  case whaType::B:
754  this->m_pertinentNodes->pushFront(nodePtr);
755  break;
756 
757  case whaType::W:
758  markPertinentChildren(nodePtr, PQNodeRoot::PQNodeStatus::Pertinent, whaType::W);
759  nodePtr->pertChildCount(0);
760  this->m_pertinentNodes->pushFront(nodePtr);
761  break;
762 
763  case whaType::H:
764  if (nodePtr->type() == PQNodeRoot::PQNodeType::PNode)
765  {
766  /*
767  [[nodePtr]] is a $P$-node of type $h$. Mark all full
768  children of [[nodePtr]] to be of type $b$ (by doing nothing, since
769  the default is type $b$). It furthermore marks all partial children to
770  be of type $w$ except for the child stored in [[hChild1]] of the
771  information class of type [[whaInfo*]] of [[nodePtr]]. This child is
772  marked to be of type $h$.
773  */
774  markPertinentChildren(nodePtr, PQNodeRoot::PQNodeStatus::Partial, whaType::W);
775  markPertinentChildren(nodePtr, PQNodeRoot::PQNodeStatus::Full, whaType::B);
776  if (nodePtr->getNodeInfo()->userStructInfo()->m_hChild1)
777  {
778  hChild1 = (PQNode<T, whaInfo*, Y>*)
779  nodePtr->getNodeInfo()->userStructInfo()->m_hChild1;
780  hChild1->getNodeInfo()->userStructInfo()->m_deleteType = whaType::H;
781  if (hChild1->getNodeInfo()->userStructInfo()->m_h <
782  hChild1->getNodeInfo()->userStructInfo()->m_w)
783  childCount = 1;
784  }
785  nodePtr->pertChildCount(nodePtr->pertChildCount() +
786  childCount - partialChildren(nodePtr)->size());
787  }
788  else
789  {
790  /*
791  [[nodePtr]] is a $Q$-node. Mark all pertinent children
792  to be of type $w$, except for the full children between the
793  [[hChild1]] and the endmost child of [[nodePtr]]. These full
794  children are marked $b$ while [[hChild1]] is marked to be of type
795  $h$. Setting the type of children to $b$ or $h$ is hidden in the
796  function call [[setHchild]] \label{setHChild}.
797  */
798  markPertinentChildren(nodePtr, PQNodeRoot::PQNodeStatus::Pertinent, whaType::W);
799  hChild1 = (PQNode<T, whaInfo*, Y>*)
800  nodePtr->getNodeInfo()->userStructInfo()->m_hChild1;
801  nodePtr->pertChildCount(setHchild(hChild1));
802  }
803  this->m_pertinentNodes->pushFront(nodePtr);
804  break;
805 
806  case whaType::A:
807  if (nodePtr->type() == PQNodeRoot::PQNodeType::PNode)
808  {
809  /*
810  [[nodePtr]] is a $P$-node of type $a$.
811  Distinguish two main cases, based on the existence of a
812  child of [[nodePtr]] stored in [[aChild]] of the information
813  class of type [[_whaInfo]] of [[nodePtr]].
814  \begin{enumerate}
815  \item
816  If [[aChild]] is not empty, the chunk marks all full
817  and partial children of [[nodePtr]] to be of type $w$ and
818  marks the child [[aChild]] to be of type $a$.
819  \item
820  If [[aChild]] is empty, the chunk
821  marks all full children of [[nodePtr]] to be of type $b$
822  (by doing nothing, since the default is type $b$).
823  It furthermore marks all partial children to be of type $w$
824  except for the children stored in [[hChild1]] and
825  [[hChild2]] of the information class of type [[whaInfo*]] of
826  [[nodePtr]] which are marked to be of type $h$. Observe that
827  we have to distinguish the cases where both, [[_hChild]] and
828  [[hChild2]] are available, just [[_h_Child1]] is available or
829  none of the nodes exist.
830  \end{enumerate}
831  */
832  if (nodePtr->getNodeInfo()->userStructInfo()->m_aChild)
833  {
834  markPertinentChildren(nodePtr, PQNodeRoot::PQNodeStatus::Pertinent, whaType::W);
835  aChild = (PQNode<T, whaInfo*, Y>*)
836  nodePtr->getNodeInfo()->userStructInfo()->m_aChild;
837  aChild->getNodeInfo()->userStructInfo()->m_deleteType = whaType::A;
838  nodePtr->pertChildCount(1);
839  }
840  else
841  {
842  markPertinentChildren(nodePtr, PQNodeRoot::PQNodeStatus::Full, whaType::B);
843  markPertinentChildren(nodePtr, PQNodeRoot::PQNodeStatus::Partial, whaType::W);
844  if (nodePtr->getNodeInfo()->userStructInfo()->m_hChild1)
845  {
846  hChild1 = (PQNode<T, whaInfo*, Y>*)
847  nodePtr->getNodeInfo()->userStructInfo()->m_hChild1;
848  hChild1->getNodeInfo()->userStructInfo()->m_deleteType = whaType::H;
849  if (hChild1->getNodeInfo()->userStructInfo()->m_h <
850  hChild1->getNodeInfo()->userStructInfo()->m_w)
851  childCount = 1;
852  }
853  if (nodePtr->getNodeInfo()->userStructInfo()->m_hChild2)
854  {
855  hChild2 = (PQNode<T, whaInfo*, Y>*)
856  nodePtr->getNodeInfo()->userStructInfo()->m_hChild2;
857  hChild2->getNodeInfo()->userStructInfo()->m_deleteType = whaType::H;
858  if (hChild2->getNodeInfo()->userStructInfo()->m_h <
859  hChild2->getNodeInfo()->userStructInfo()->m_w)
860  childCount++;
861  }
862  nodePtr->pertChildCount(nodePtr->pertChildCount() + childCount -
863  partialChildren(nodePtr)->size());
864  }
865  }
866  else
867  {
868  /*
869  [[nodePtr]] is a $Q$-node of type $a$.
870  Distinguish two main cases, based on the existence of a child of

911  }

915  this chunk cleans up the information needed during the

917  */

1105 template<class T, class Y>
1106 void MaxSequencePQTree<T, Y>::haNumPnode(PQNode<T, whaInfo*, Y> *nodePtr)
1107 {
1108  /* Variables:
1109  * - \a sumParW = sum_{i in Par(\p nodePtr)} w_i, where
1110  * Par(\p nodePtr) denotes the set of partial children of \p nodePtr.
1111  * - \a sumMax1 = max_{i in Par(\p nodePtr)}1 {(w_i - h_i)}
1112  * where Par(\p nodePtr) denotes the set of partial children of
1113  * - \p nodePtr and max 1 the first maximum.
1114  * - \a sumMax2 = max_{i in Par(\p nodePtr)}2 {(w_i - h_i)}
1115  * where Par(\p nodePtr) denotes the set of partial children of
1116  * \p nodePtr and max2 the second maximum.
1117  * - \a currentNode
1118  * - \a hChild1 is a pointer to the \a hChild1 of \p nodePtr.
1119  * - \a hChild2 is a pointer to the \a hChild2 of \p nodePtr.
1120  * - \a aChild is a pointer to the \a aChild of \p nodePtr.
1121  */
1122  int sumParW = 0;
1123  int sumMax1 = 0;
1124  int sumMax2 = 0;
1125  PQNode<T,whaInfo*,Y> *currentNode = nullptr;
1126  PQNode<T,whaInfo*,Y> *hChild1 = nullptr;
1127  PQNode<T,whaInfo*,Y> *hChild2 = nullptr;
1128  PQNode<T,whaInfo*,Y> *aChild = nullptr;
1129 
1130  /*
1131  Computes the $h$-number
1132  \[ h = \sum_{i \in Par(\mbox{[[nodePtr]]})}w_i - \max_{i\in
1133  Par(\mbox{[[nodePtr]]})}1\left\{(w_i - h_i)\right\}\]
1134  of the $P$-node [[nodePtr]].
1135  This is done by scanning all partial children stored in the
1136  [[partialChildrenStack]] of [[nodePtr]] summing up the $w_i$ for every
1137  $i \in Par(\mbox{[[nodePtr]]})$ and detecting
1138  \[\max_{i\in Par(\mbox{[[nodePtr]]})}1\left\{(w_i - h_i)\right\}.\]
1139  Since this can be simultaneously it also computes
1140  \[\max_{i\in Par(\mbox{[[nodePtr]]})}2\left\{(w_i - h_i)\right\}\]
1141  which is needed to determine the $a$-number.
1142  After successfully determining the $h$-number, the [[hChild1]] and
1143  [[hChild2]] of [[nodePtr]] are set.
1144  */
1145 
1146  ListIterator<PQNode<T,whaInfo*,Y>*> it;
1147  for (it = partialChildren(nodePtr)->begin(); it.valid(); ++it)
1148  {
1149  currentNode = (*it);
1150  sumParW = sumParW + currentNode->getNodeInfo()->userStructInfo()->m_w;
1151  int sumHelp = currentNode->getNodeInfo()->userStructInfo()->m_w -
1152  currentNode->getNodeInfo()->userStructInfo()->m_h;
1153  if (sumMax1 <= sumHelp)
1154  {
1155  sumMax2 = sumMax1;
1156  hChild2 = hChild1;
1157  sumMax1 = sumHelp;
1158  hChild1 = currentNode;
1159  }
1160  else if (sumMax2 <= sumHelp)
1161  {
1162  sumMax2 = sumHelp;
1163  hChild2 = currentNode;
1164  }
1165  }
1166  nodePtr->getNodeInfo()->userStructInfo()->m_hChild1 = hChild1;
1167  nodePtr->getNodeInfo()->userStructInfo()->m_hChild2 = hChild2;
1168  nodePtr->getNodeInfo()->userStructInfo()->m_h = sumParW - sumMax1;
1169 
1170  /*
1171  Compute the $a$-number of the $P$-node [[nodePtr]] where
1172  \[ a = \min \{ \alpha_1, \alpha_2\}\]
1173  such that
1174  \[\alpha_1 = \sum_{i \in P(\mbox{[[nodePtr]]})}w_i - \max_{i\in
1175  P(\mbox{[[nodePtr]]})}\left\{(w_i - h_i)\right\} \]
1176  which can be computed calling the function [[alpha1beta1Number]] and
1177  \[{\alpha}_2 \sum_{i \in Par(\mbox{[[nodePtr]]})}w_i -
1178  \max_{i\in Par(\mbox{[[nodePtr]]})}1\left\{(w_i - h_i)\right\}
1179  - \max_{i\in Par(\mbox{[[nodePtr]]})}2\left\{(w_i - h_i)\right\}\]
1180  This chunk uses two extra variables
1181  \begin{description}
1182  \item[[alpha1]] $ = \alpha_1$.
1183  \item[[alpha2]] $ = \alpha_2$.
1184  \end{description}
1185  */
1186  int alpha2 = sumParW - sumMax1 - sumMax2;
1187  int alpha1 = alpha1beta1Number(nodePtr,&aChild);
1188 
1189  if (alpha1 <= alpha2)
1190  {
1191  nodePtr->getNodeInfo()->userStructInfo()->m_a = alpha1;
1192  nodePtr->getNodeInfo()->userStructInfo()->m_aChild = aChild;
1193  }
1194  else
1195  {
1196  nodePtr->getNodeInfo()->userStructInfo()->m_a = alpha2;
1197  nodePtr->getNodeInfo()->userStructInfo()->m_aChild = nullptr;
1198  }
1199 }

1215 void MaxSequencePQTree<T,Y>::hNumQnode(
1216  PQNode<T,whaInfo*,Y> *nodePtr,
1217  int sumAllW)
1218 {
1219  /* Variables:
1220  * - \a sumLeft = sum_{i in P(\p nodePtr)} w_i - sum_{i
1221  * in P_L(\p nodePtr)}(w_i - h_i), where
1222  * P_L(\p nodePtr) denotes the maximal consecutive sequence
1223  * of pertinent children on the left side of the Q-node
1224  * \p nodePtr such that only the rightmost node in
1225  * P_L(\p nodePtr) may be partial.
1226  * - \a sumRight = sum_{i in P(\p [nodePtr)}w_i - sum_{i
1227  * in P_L(\p nodePtr)}(w_i - h_i), where
1228  * P_L(\p nodePtr) denotes the maximal consecutive sequence
1229  * of pertinent children on the left side of the Q-node
1230  * \p nodePtr such that only the rightmost node in
1231  * P_L(\p nodePtr) may be partial.
1232  * - \a fullLabel
1233  * - \a aChild is a pointer to the a-child of \p nodePtr.
1234  * - \a leftChild is a pointer to the left endmost child of \p nodePtr.
1235  * - \a rightChild is a pointer to the right endmost child of \p nodePtr.
1236  * - \a holdSibling is a pointer to a child of \p nodePtr, needed
1237  * to proceed through sequences of pertinent children.
1238  * - \a checkSibling is a pointer to a currently examined child of \p nodePtr.
1239  */
1240  int sumLeft = 0;
1241  int sumRight = 0;
1242  bool fullLabel = true;
1243  PQNode<T,whaInfo*,Y> *leftChild = nullptr;
1244  PQNode<T,whaInfo*,Y> *rightChild = nullptr;
1245  PQNode<T,whaInfo*,Y> *holdSibling = nullptr;
1246  PQNode<T,whaInfo*,Y> *checkSibling = nullptr;
1247 
1248  //Compute the $h$-number of the $Q$-node [[nodePtr]]
1249 
1250  //Get endmost children of the $Q$-node [[nodePtr]].
1251  leftChild = nodePtr->getEndmost(0);
1252  rightChild = nodePtr->getEndmost(leftChild);
1253  OGDF_ASSERT(leftChild);
1254  OGDF_ASSERT(rightChild);
1255 
1256  /*
1257  Check the left
1258  side of the $Q$-node [[nodePtr]] for the maximal consecutive sequence
1259  of full nodes, including at most one partial child at the end of the sequence.
1260 
1261  The variable [[fullLabel]] is [[true]] as long as the [[while]]-loop
1262  has not detected an partial <b>or</b> empty child (see case [[if
1263  (leftChild->status() != Full)]]. Observe that the
1264  construction of the [[while]]-loop examines the last child if it is a
1265  partial child as well (see case [[if (leftChild->status() !=
1266  Empty)]] where in the computation in [[sumLeft]] we take advantage
1267  of the fact, that the $h$-number of a full child is zero).
1268  */
1269  while (fullLabel)
1270  {
1271  if (leftChild->status() != PQNodeRoot::PQNodeStatus::Full)
1272  fullLabel = false;
1273  if (leftChild->status() != PQNodeRoot::PQNodeStatus::Empty)
1274  {
1275  sumLeft = sumLeft +
1276  leftChild->getNodeInfo()->userStructInfo()->m_w -
1277  leftChild->getNodeInfo()->userStructInfo()->m_h;
1278  checkSibling = leftChild->getNextSib(holdSibling);
1279  if (checkSibling == nullptr)
1280  fullLabel = false;
1281  holdSibling = leftChild;
1282  leftChild = checkSibling;
1283  }
1284  }
1285 
1286  /*
1287  Check the right
1288  side of the $Q$-node [[nodePtr]] for the maximal consecutive sequence
1289  of full nodes, including at most one partial child at the end of the sequence.
1290 
1291  The variable [[fullLabel]] is [[true]] as long as the [[while]]-loop
1292  has not detected an partial <b>or</b> empty child (see case [[if
1293  (leftChild->status() != Full)]]. Observe that the
1294  construction of the [[while]]-loop examines the last child if it is a
1295  partial child as well (see case [[if (leftChild->status() !=
1296  Empty)]] where in the computation in [[sumLeft]] we take advantage
1297  of the fact, that the $h$-number of a full child is zero).
1298  */
1299  holdSibling = nullptr;
1300  checkSibling = nullptr;
1301  fullLabel = true;
1302  while (fullLabel)
1303  {
1304  if (rightChild->status() != PQNodeRoot::PQNodeStatus::Full)
1305  fullLabel = false;
1306  if (rightChild->status() != PQNodeRoot::PQNodeStatus::Empty)
1307  {
1308  sumRight = sumRight +
1309  rightChild->getNodeInfo()->userStructInfo()->m_w -
1310  rightChild->getNodeInfo()->userStructInfo()->m_h;
1311 
1312  checkSibling = rightChild->getNextSib(holdSibling);
1313 
1314  if (checkSibling == nullptr)
1315  fullLabel = false;
1316 
1317  holdSibling = rightChild;
1318  rightChild = checkSibling;
1319  }
1320  }
1321 
1322  /*
1323  After computing the number of pertinent leaves that stay in the $PQ$-tree
1324  when keeping either the left pertinent or the right pertinent side of
1325  the $Q$-node in the tree, this chunk chooses the side where the
1326  maximum number of leaves stay in the tree.
1327  Observe that we have to case the fact, that on both sides of the
1328  $Q$-node [[nodePtr]] no pertinent children are.
1329  */
1330  leftChild = nodePtr->getEndmost(0);
1331  rightChild = nodePtr->getEndmost(leftChild);
1332  if (sumLeft == 0 && sumRight == 0)
1333  {
1334  nodePtr->getNodeInfo()->userStructInfo()->m_h = sumAllW;
1335  nodePtr->getNodeInfo()->userStructInfo()->m_hChild1 = nullptr;
1336  }
1337  else if (sumLeft < sumRight)
1338  {
1339  nodePtr->getNodeInfo()->userStructInfo()->m_h = sumAllW - sumRight;
1340  nodePtr->getNodeInfo()->userStructInfo()->m_hChild1 = rightChild;
1341  }
1342  else
1343  {
1344  nodePtr->getNodeInfo()->userStructInfo()->m_h = sumAllW - sumLeft;
1345  nodePtr->getNodeInfo()->userStructInfo()->m_hChild1 = leftChild;
1346  }
1347 }

1355  * - \a beta1 = sum_{i in P(\p nodePtr} w_i - max_{i in P(\p nodePtr)}{(w_i =
1356  * a_i)}, where $P(\p nodePtr) denotes the set of all pertinent
1357  * children of the Q-node \p nodePtr. Depicts the a-number if just one

1361  * sequence of pertinent children of the Q-node \p nodePtr such that all
1362  * nodes in P_A(\p nodePtr) except for the leftmost and rightmost ones are

1369  * - \a actualNode pointer to a child of the Q-node. It is the

1372  * sequence. Needed to process all nodes stored in \a sequence.
1373  * - \a lastChild is a pointer to the endmost child of the Q-node
1374  * that is opposite to the endmost child, where this chunk starts
1375  * processing the sequence of children.

1377  * the nodes of a consecutive sequence that is actually processed.

1380  int beta1 = alpha1beta1Number(nodePtr,&aChild);
1381  int beta2 = 0;
1382  int aSum = 0;
1383  int aHoldSum = 0;
1384  bool endReached = 0;
1385  PQNode<T,whaInfo*,Y> *leftMost = nullptr;
1386  PQNode<T,whaInfo*,Y> *leftSib = nullptr;
1387  PQNode<T,whaInfo*,Y> *leftMostHold = nullptr;
1388  PQNode<T,whaInfo*,Y> *leftSibHold = nullptr;
1389  PQNode<T,whaInfo*,Y> *actualNode = nullptr;
1390  PQNode<T,whaInfo*,Y> *currentNode = nullptr;
1391  PQNode<T,whaInfo*,Y> *lastChild = nullptr;
1392  PQNode<T,whaInfo*,Y> *holdSibling = nullptr;
1393  PQNode<T,whaInfo*,Y> *checkSibling = nullptr;
