93  void call(const EdgeArray<T> &weights, const EdgeArray<T> &flow, const node source, const node target)
94  {
95  m_frontCutCount = 0;
96  m_backCutCount = 0;
97  m_totalCount = weights.graphOf()->numberOfNodes();
98 
99  List<node> queue;
100  m_nodeSet.init(*weights.graphOf(), NO_CUT);
101  queue.pushBack(source);
102  m_nodeSet[source] = FRONT_CUT;
103 
104  // front cut
105  while(!queue.empty()) {
106  const node v = queue.popFrontRet();
107  for(adjEntry adj : v->adjEntries) {
108  const node w = adj->twinNode();
109  const edge e = adj->theEdge();
110  if(m_nodeSet[w] == NO_CUT
111  && ((e->source() == v && m_et.get().less(flow[e], weights[e]))
112  || (e->target() == v && m_et.get().greater(flow[e], (T) 0)))) {
113  queue.pushBack(w);
114  m_nodeSet[w] = FRONT_CUT;
115  m_frontCutCount++;
116  }
117  }
118  }
119 
120  // back cut
121  queue.pushBack(target);
122  m_nodeSet[target] = BACK_CUT;
123 
124  while(!queue.empty()) {
125  const node v = queue.popFrontRet();
126  for(adjEntry adj : v->adjEntries) {
127  const node w = adj->twinNode();
128  const edge e = adj->theEdge();
129  if(m_nodeSet[w] == NO_CUT

131  || (e->source() == v && m_et.get().greater(flow[e], (T) 0)))) {
132  queue.pushBack(w);
