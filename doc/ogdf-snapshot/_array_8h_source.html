534  template<class COMPARER>
535  static void quicksortInt(E *pL, E *pR, const COMPARER &comp) {
536  size_t s = pR-pL;
537 
538  // use insertion sort for small instances
539  if (s < maxSizeInsertionSort) {
540  for (E *pI = pL+1; pI <= pR; pI++) {
541  E v = *pI;
542  E *pJ = pI;
543  while (--pJ >= pL && comp.less(v,*pJ)) {
544  *(pJ+1) = *pJ;
545  }
546  *(pJ+1) = v;
547  }
548  return;
549  }
550 
551  E *pI = pL, *pJ = pR;
552  E x = *(pL+(s>>1));
553 
554  do {
555  while (comp.less(*pI,x)) pI++;
556  while (comp.less(x,*pJ)) pJ--;
557  if (pI <= pJ) std::swap(*pI++,*pJ--);
558  } while (pI <= pJ);
559 
560  if (pL < pJ) quicksortInt(pL,pJ,comp);
561  if (pI < pR) quicksortInt(pI,pR,comp);
562  }

569 template<class E, class INDEX>
570 void Array<E, INDEX>::expandArray(INDEX add)
571 {
572  INDEX sOld = size(), sNew = sOld + add;
573 
574  // expand allocated memory block
575  if (m_pStart != nullptr) {
576  // if the element type is trivially copiable, just use realloc
577 #if defined(__GNUC__) && !defined(__clang__) && __GNUC__ < 5
578  // g++ 4.8/4.9 does not have is_trivially_copyable, but
579  // clang 3.5 (which is also __GNUC__ < 5) has it
580  if (std::has_trivial_copy_assign<E>::value) {
581 #else
582  if (std::is_trivially_copyable<E>::value) {
583 #endif
584  E *p = static_cast<E *>( realloc(m_pStart, sNew*sizeof(E)) );
585  if (p == nullptr) OGDF_THROW(InsufficientMemoryException);
586  m_pStart = p;
587 
588  // otherwise allocate new block, move elements, and free old block
589  } else {
590  E *p = static_cast<E *>( malloc(sNew*sizeof(E)) );
591  if (p == nullptr) OGDF_THROW(InsufficientMemoryException);
592 
593  for (int i = 0; i < min(sOld,sNew); ++i) {
594  new (&p[i]) E(std::move(m_pStart[i]));
595  }
596 
597  deconstruct();
598  m_pStart = p;
599  }
600 
601  } else {
602  m_pStart = static_cast<E *>( malloc(sNew*sizeof(E)) );
603  if (m_pStart == nullptr) OGDF_THROW(InsufficientMemoryException);
604  }
605 
606  m_vpStart = m_pStart - m_low;
607  m_pStop = m_pStart + sNew;
608  m_high += add;
609 }
