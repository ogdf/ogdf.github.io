711  template<class COMPARER>
712  static void quicksortInt(E *pL, E *pR, const COMPARER &comp) {
713  size_t s = pR-pL;
714 
715  // use insertion sort for small instances
716  if (s < maxSizeInsertionSort) {
717  for (E *pI = pL+1; pI <= pR; pI++) {
718  E v = *pI;
719  E *pJ = pI;
720  while (--pJ >= pL && comp.less(v,*pJ)) {
721  *(pJ+1) = *pJ;
722  }
723  *(pJ+1) = v;
724  }
725  return;
726  }
727 
728  E *pI = pL, *pJ = pR;
729  E x = *(pL+(s>>1));
730 
731  do {
732  while (comp.less(*pI,x)) pI++;
733  while (comp.less(x,*pJ)) pJ--;
734  if (pI <= pJ) std::swap(*pI++,*pJ--);
735  } while (pI <= pJ);
736 
737  if (pL < pJ) quicksortInt(pL,pJ,comp);
738  if (pI < pR) quicksortInt(pI,pR,comp);
739  }

744 // enlarges storage for array and moves old entries
745 template<class E, class INDEX>
746 void Array<E, INDEX>::expandArray(INDEX add)
747 {
748  INDEX sOld = size(), sNew = sOld + add;
749 
750  // expand allocated memory block
751  if (m_pStart != nullptr) {
752  // if the element type is trivially copiable, just use realloc
753 #if defined(__GNUC__) && !defined(__clang__) && __GNUC__ < 5
754  // g++ 4.8/4.9 does not have is_trivially_copyable, but
755  // clang 3.5 (which is also __GNUC__ < 5) has it
756  if (std::has_trivial_copy_assign<E>::value) {
757 #else
758  if (std::is_trivially_copyable<E>::value) {
759 #endif
760  E *p = static_cast<E *>( realloc(m_pStart, sNew*sizeof(E)) );
761  if (p == nullptr) OGDF_THROW(InsufficientMemoryException);
762  m_pStart = p;
763 
764  // otherwise allocate new block, move elements, and free old block
765  } else {
766  E *p = static_cast<E *>( malloc(sNew*sizeof(E)) );
767  if (p == nullptr) OGDF_THROW(InsufficientMemoryException);
768 
769  for (int i = 0; i < min(sOld,sNew); ++i) {
770  new (&p[i]) E(std::move(m_pStart[i]));
771  }
772 
773  deconstruct();
774  m_pStart = p;
775  }
776 
777  } else {
778  m_pStart = static_cast<E *>( malloc(sNew*sizeof(E)) );
779  if (m_pStart == nullptr) OGDF_THROW(InsufficientMemoryException);
780  }
781 
782  m_vpStart = m_pStart - m_low;
783  m_pStop = m_pStart + sNew;
784  m_high += add;
785 }
