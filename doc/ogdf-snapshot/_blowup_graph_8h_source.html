77  /* witness set (for some component and specified K (core edge set))
78  *
79  * W(e) = { e } if e is a core edge
80  * W(e) =
81  * take the path P of loss edges from u to e in the component
82  * add every incident core edge of u
83  * that is: if we root the loss connected component at the terminal,
84  * W(e) = all core edges incident to the subtree below e
85  *
86  * Need fast lookups for:
87  * (1) |W(f)|
88  * (2) all loss edges f such that given core edge e is in W(f)
89  * (3) all loss edges f such that W(f) is a subset of a given basis -> for each e \in B: do (2)
90  *
91  * Data Structures:
92  * - witnessCard[e] = |W(e)|
93  * - witness[v_e] = { f | e \in W(f) }
94  * (Note that core edges are given as nodes.)
95  * Also note that we can save it all much sparser.
96  */
97  EdgeArray<int> m_witnessCard;
98  NodeArray<ArrayBuffer<edge>> m_witness;

140  List<node> queueC;
141  queueT.pushBack(start);
142  queueC.pushBack(copy[v]);
143  while (!queueT.empty()) {
144  const adjEntry inAdj = queueT.popFrontRet();
145  const node wT = inAdj->twinNode();

229  int updateSourceAndTargetArcCapacities(const node v) {
230  int delta = 0;
231  int capSource = 0;
232  int capTarget = -getLCM();
233  adjEntry adj2;
234  for (adjEntry adj = v->firstAdj(); adj; adj = adj2) {
235  adj2 = adj->succ();
236  if (adj->twinNode() == getSource()) {
237  // remove arcs from the source
238  m_graph.delEdge(adj->theEdge());
239  }
240  else
241  if (adj->twinNode() == getPseudotarget()) {
242  // remove arcs to the pseudotarget
243  delta -= getCapacity(adj->theEdge());
244  m_graph.delEdge(adj->theEdge());
245  }
246  else {
247  // compute y_v for the contraction node
248  capTarget += getCapacity(adj->theEdge());
249  // compute s->v capacity
250  if (v != adj->theEdge()->target()) { // outgoing edge
251  capSource += getCapacity(adj->theEdge());
252  }
253  }
254  }
255  OGDF_ASSERT(capTarget >= 0);
256  if (capTarget > 0) {
257  newEdge(v, getPseudotarget(), 0, capTarget);
258  }
259  if (capSource > 0) {
260  newEdge(getSource(), v, 0, capSource);
261  }
262 
263  return delta + capTarget;
264  }

290  void initCoreWitness() {
291  m_witnessCard.init(m_graph, 0);
292  m_witness.init(m_graph);
293 
294  // compute set of core edges
295  EdgeArray<bool> isLossEdge;
296  m_ceModule.call(m_graph, terminals(), isLossEdge);
297 
298  // add nodes for core edges and be able to map them
299  EdgeArray<node> splitMap(m_graph, nullptr);
300  ArrayBuffer<edge> coreEdges;
301  for (edge e : m_graph.edges) {
302  if (!isLossEdge[e]) {
303  splitMap[e] = m_graph.newNode();
304  coreEdges.push(e);
305  }
306  }
307 
308  // traverse losses from all terminals to find witness sets
309  NodeArray<adjEntry> pred(m_graph, nullptr);
310  for (node t : terminals()) {
311  ArrayBuffer<node> stack;
312  stack.push(t);
313  while (!stack.empty()) {
314  // for each node v "below" an edge e in the traversal:
315  // add all incident core edges vw to W(e)
316  const node v = stack.popRet();
317  for (adjEntry adj : v->adjEntries) {
318  const edge e = adj->theEdge();
319  const node w = adj->twinNode();
320  if (!pred[v] || w != pred[v]->theNode()) { // do not look at backward arcs in the tree
321  if (isLossEdge[e]) {
322  stack.push(w);
323  pred[w] = adj;
324  } else {
325  for (node x = v; pred[x]; x = pred[x]->theNode()) {
326  addWitness(splitMap[e], pred[x]->theEdge());
327  }
328  }
329  }
330  }
331  }
332  }
333 
334  // finally replace core edges by nodes
335  for (edge e : coreEdges) {
336  const T w = getCost(e);
337  const node x = splitMap[e];
338  const int cap = getCapacity(e);
339  OGDF_ASSERT(x);
340  newEdge(e->source(), x, w, cap);
341  newEdge(x, e->target(), w, cap);
342  // the cost of a core edge node is hence the weight of one incident edge; also keep capacity
343  m_graph.delEdge(e);
344  addCore(x);
345  }
346  }

349  void makeCWCopy(const HashArray<edge,edge> &edgeMap) {
350  for (HashConstIterator<edge,edge> pair = edgeMap.begin(); pair.valid(); ++pair) {
351  const edge eO = pair.key();
352  const edge eC = pair.info();
353  const node vO = eO->target();
354  const node vC = eC->target();
355  m_witnessCard[eC] = m_witnessCard[eO]; // copy witness cardinality
356  if (vC == vO) { // target is a terminal, so it cannot be a core edge
357  continue;
358  }
359  for (ListIterator<node> it = m_coreEdges.begin(); it.valid(); ++it) {
360  if (*it == vO) { // vO is a core edge
361  m_coreEdges.insertAfter(vC, it); // make vC a core edge
362 
363  // copy witness sets
364  // XXX: do we need this or are witness sets computed after the loop again?!
365  for (edge e : m_witness[vO]) {
366  m_witness[vC].push(edgeMap[e]);
367  }
368  break;
369  }
370  }
371  }

515  void removeBasis(node v, std::function<void(edge)> newRootFunction) {
516  ArrayBuffer<node> cleanup;
517  cleanup.push(v->firstAdj()->twinNode());
518  cleanup.push(v->lastAdj()->twinNode());
519  OGDF_ASSERT(v->degree() == 2);
520  OGDF_ASSERT(v->firstAdj()->twinNode() != v->lastAdj()->twinNode());
521  m_graph.delNode(v);
522 
523  while (!cleanup.empty()) {
524  v = cleanup.popRet();
525  if (!isTerminal(v)) {
526  OGDF_ASSERT(v->degree() >= 1);
527  if (v->degree() == 1) { // v is a pendant node, delete
528  cleanup.push(v->firstAdj()->twinNode());
529  m_graph.delNode(v);
530  } else
531  if (v->indeg() == 0) { // v has no incoming edge, fix
532  const node w = v->firstAdj()->twinNode();
533  const edge e = v->firstAdj()->theEdge();
534  m_graph.reverseEdge(e);
535  OGDF_ASSERT(e->source() == w);
536  if (!isTerminal(w)) {
537  cleanup.push(w);
538  // when w is cleaned up, it must not go back to v
539  if (w->firstAdj()->theEdge() == e) {
540  // move first adjacency entries of w away (w->v is not first anymore)
541  m_graph.moveAdjAfter(w->firstAdj(), w->lastAdj());
542  }
543  } else {
544  // a terminal means we have a new root
545  newRootFunction(e);
546  }
547  }
548  }
