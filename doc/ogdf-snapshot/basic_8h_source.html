166  inline double randomDoubleNormal(double m, double sd)
167  {
168  double x1, y1, w;
169 
170  do {
171  double rndVal = randomDouble(0,1);
172  x1 = 2.0 * rndVal - 1.0;
173  rndVal = randomDouble(0,1);
174  double x2 = 2.0 * rndVal - 1.0;
175  w = x1*x1 + x2*x2;
176  } while (w >= 1.0);
177 
178  w = sqrt((-2.0 * log(w))/w) ;
179  y1 = x1*w;
180 
181  return m + y1 * sd;
182  }
