203  inline double randomDoubleNormal(double m, double sd)
204  {
205  double x1, y1, w;
206 
207  do {
208  double rndVal = randomDouble(0,1);
209  x1 = 2.0 * rndVal - 1.0;
210  rndVal = randomDouble(0,1);
211  double x2 = 2.0 * rndVal - 1.0;
212  w = x1*x1 + x2*x2;
213  } while (w >= 1.0);
214 
215  w = sqrt((-2.0 * log(w))/w) ;
216  y1 = x1*w;
217 
218  return(m + y1*sd);
