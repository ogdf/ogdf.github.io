77  const DWMData dataOf(const List<node> &key) const
78  {
79  OGDF_ASSERT(key.size() > 1);
80  if (key.size() == 2) {
81  NodePairs nodepairs(0, 0, NodePair(key.front(), key.back()));

88  {
89  OGDF_ASSERT(key.size() > 1);
90  if (key.size() == 2) {
91  return m_distance[key.front()][key.back()];

97  void insertSorted(List<node> &list, node v) const
98  {
99  for (ListIterator<node> it = list.begin(); it.valid(); ++it) {
100  OGDF_ASSERT((*it)->index() != v->index());
101  if (v->index() < (*it)->index()) {
102  list.insertBefore(v, it);
103  return;

105  }

121  void call(int restricted)
122  {
123  SubsetEnumerator<node> terminalSubset(m_terminals);
124  for (terminalSubset.begin(2, restricted-1); terminalSubset.valid(); terminalSubset.next()) {
125  List<node> terminals;
126  terminalSubset.list(terminals);
127  SubsetEnumerator<node> subset(terminals);
128  // populate "split"
129  NodeArray<DWMData> split(m_G);
130  auto computeSplit = [&](node v) {
131  OGDF_ASSERT(!terminalSubset.hasMember(v));
132  DWMData best;
133  for (subset.begin(1, terminals.size()-1); subset.valid(); subset.next()) {
134  List<node> list1, list2;
135  subset.list(list1, list2);
136  insertSorted(list1, v);
137  insertSorted(list2, v);
138  T tmp = costOf(list1) + costOf(list2);
139  if (tmp < best.cost) {
140  best = dataOf(list1);
141  DWMData data = dataOf(list2);
142  best.cost += data.cost;
143  int bsize = best.nodepairs.size();
144  int dsize = data.nodepairs.size();
145  best.nodepairs.grow(dsize);
146  for (int i = 0; i < dsize; ++i) { // copy
147  best.nodepairs[bsize + i] = data.nodepairs[i];
148  }
149  }
150  }
151  split[v] = best;
152  };
153  // compute partial solutions
154  auto computePartialSolutions = [&](node v) {
155  List<node> newTerminals(terminals);
156  insertSorted(newTerminals, v);
157  if (m_map.member(newTerminals)) { // already defined
158  return; // abort
159  }
160  DWMData best;
161  for (node w : m_G.nodes) {
162  T dist = m_distance[v][w];
163  if (v == w) {
164  dist = 0;
165  }
166  if (terminalSubset.hasMember(w)) {
167  // we attach edge vw to tree containing terminal w
168  T tmp;
169  tmp = costOf(terminals) + dist;
170  if (tmp < best.cost) {
171  best = dataOf(terminals);
172  best.cost += dist;
173  best.nodepairs.grow(1, NodePair(v,w));
174  }
175  } else {
176  // we attach edge vw to tree split[w]
177  T tmp;
178  if (split[w].nodepairs.size() == 0) {
179  computeSplit(w);
180  }
181  tmp = split[w].cost + dist;
182  if (tmp < best.cost) {
183  best = split[w];
184  if (v != w) {
185  best.cost += dist;
186  best.nodepairs.grow(1, NodePair(v,w));
187  }
188  }
189  }
190  }
191  m_map.fastInsert(newTerminals, best);
192  };
193  if (terminalSubset.size() != restricted - 1) {
194  for (node v : m_G.nodes) {
195  if (!terminalSubset.hasMember(v)) {
196  computePartialSolutions(v);
197  }
198  }
199  } else { // maximal terminal subset
200  for (node v : m_terminals) { // save time by only adding terminals instead of all nodes
201  if (!terminalSubset.hasMember(v)) {
202  computePartialSolutions(v);
203  }
204  }
205  }
206  }
207  }

210  T getSteinerTreeFor(const List<node> &terminals, EdgeWeightedGraphCopy<T> &tree)
211  {
212  T cost(0);
213  DWMData data = dataOf(terminals);
214  tree.createEmpty(m_G);
215  for (auto nodepair : data.nodepairs) {
216  node uO = nodepair.source;
217  node vO = nodepair.target;
218  node uC = tree.copy(uO);
219  node vC = tree.copy(vO);
220  if (uC == nullptr) {
221  uC = tree.newNode(uO);
222  }
223  if (vC == nullptr) {
224  vC = tree.newNode(vO);
225  }
226  const T dist = m_distance[uO][vO];
227  tree.newEdge(uC, vC, dist);
228  cost += dist;
229  }
230  OGDF_ASSERT(isTree(tree));
231  return cost;
232  }

248  int hash(const List<node> &key) const
249  {
250  ListConstIterator<node> it = key.begin();
251  int hash = (*it)->index();
252  for (; it.valid(); ++it) {
253  hash = (hash * m_random + (*it)->index()) % c_prime;
