158  virtual void onWellSeparatedPair(int a, int b)
159  {
160  // force vector
161  double delta[Dim];
162  double force;
163  double force_prime;
164 
165  double dist = computeDeltaAndDistance<Dim>(m_treeForce.m_nodeData[a].centerOfMass,
166  m_treeForce.m_nodeData[b].centerOfMass,
167  delta);
168  m_forceFunc(dist, force, force_prime);
169 #if 0
170  m_forceFunc(m_treeForce.m_nodeData[a].centerOfMass, m_treeForce.m_nodeData[b].centerOfMass, force, force_prime);
171 #endif
172 
173  // compute the force vector for each dim
174  for (int d = 0; d < Dim; d++) {
175  m_treeForce.m_nodeData[a].force[d] += force * delta[d] / dist * m_treeForce.m_nodeData[b].mass;
176  m_treeForce.m_nodeData[b].force[d] -= force * delta[d] / dist * m_treeForce.m_nodeData[a].mass;
177  };
178 
179  if (UseForcePrime) {
180  m_treeForce.m_nodeData[a].force_prime += force_prime * m_treeForce.m_nodeData[b].mass;
181  m_treeForce.m_nodeData[b].force_prime += force_prime * m_treeForce.m_nodeData[a].mass;
182  }
183 
184 
185  }

284 void DTreeForce<Dim>::computeForces(ForceFunc forceFunc)
285 {
286  // reset all point forces
287  resetPointForces();
288 
289  if (numPoints() <= 1)

294 
295  // now do the bottom up phase
296  bottomUpPhase(tree().rootIndex());
297

302 
303  // finally, push down the forces

308 template<int Dim>
309 void DTreeForce<Dim>::bottomUpPhase(int curr)
310 {
311  // reset the force and the center of mass
312  for (int d = 0; d < Dim; d++) {
313  m_nodeData[curr].force[d] = 0.0;
314  m_nodeData[curr].force_prime = 0.0;
315  m_nodeData[curr].centerOfMass[d] = 0.0;
316  }
317 
318  // reset the mass
319  m_nodeData[curr].mass = 0.0;
320 
321  // if this is an inner node
322  if (tree().numChilds(curr)) {
323  // iterate over the children
324  for (int i = 0; i < tree().numChilds(curr); i++) {
325  // the child index
326  const int child = tree().child(curr, i);
327 
328  // compute the size of the subtree
329  bottomUpPhase(child);
330 
331  // node curr
332  for (int d = 0; d < Dim; d++) {
333  // sum up the center of mass coordinates
334  m_nodeData[curr].centerOfMass[d] += m_nodeData[child].centerOfMass[d] * m_nodeData[child].mass;
335  }
336 
337  // add the mass
338  m_nodeData[curr].mass += m_nodeData[child].mass;
339  };
340  } // else it is a leaf
341  else {
342  // and the remaining points
343  for (int i = 0; i < tree().numPoints(curr); ++i) {
344  int pointIndex = tree().point(curr, i);
345  // node curr
346  for (int d = 0; d < Dim; d++) {
347  m_nodeData[curr].centerOfMass[d] += position(pointIndex, d) * mass(pointIndex);
348  }
349 
350  m_nodeData[curr].mass += mass(pointIndex);
351  }
352  }
353  // node curr
354  for (int d = 0; d < Dim; d++) {
355  m_nodeData[curr].centerOfMass[d] /= m_nodeData[curr].mass;
356  }
357 }

360 void DTreeForce<Dim>::topDownPhase(int curr)
361 {
362  // if this is an inner node
363  if (tree().numChilds(curr)) {
364  // iterate over the children
365  for (int i = 0; i < tree().numChilds(curr); i++) {
366  // the child index
367  const int child = tree().child(curr, i);
368 
369  // node curr
370  for (int d = 0; d < Dim; d++) {
371  // push the force down to the child

373  }
374

376

378  topDownPhase(child);
379  };
380  } // else it is a leaf
381  else {
