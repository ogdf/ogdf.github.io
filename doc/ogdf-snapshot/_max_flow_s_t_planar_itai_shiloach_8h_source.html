197  bool findUppermostPath(const edge saturatedEdge)
198  {
199  node v, u;
200  adjEntry initialAdj;
201 
202  if (saturatedEdge == nullptr) {
203  v = *this->m_s;
204  u = *this->m_t;
205  initialAdj = m_commonFaceAdj;
206  } else {
207  v = saturatedEdge->source();
208  u = saturatedEdge->target();
209  initialAdj = saturatedEdge->adjSource()->cyclicSucc();
210  }
211 
212  OGDF_ASSERT(u != v);
213  OGDF_ASSERT(v != *this->m_t);
214  OGDF_ASSERT(u != *this->m_s);
215 
216  for(adjEntry adj = initialAdj;
217  m_edgeCounter[v] < v->degree();
218  adj = adj->cyclicSucc())
219  {
220  m_edgeCounter[v]++;
221  edge e = adj->theEdge();
222 
223  bool visited = m_visited[e];
224  m_visited[e] = true;
225 
226  if(!visited
227  && e->target() != v
228  && m_status[e->target()] != NT_DONE)
229  {
230  EdgePathType pathType = getPathType(e);
231  OGDF_ASSERT(pathType != EPT_UNKNOWN);
232 
233  if(pathType == EPT_NO_PATH) {
234  // extend the path
235  appendEdge(e);
236  adj = e->adjTarget();
237  v = e->target();
238  } else if(pathType == EPT_TARGET_PATH) {
239  // merge path
240  node w = e->target();
241 
242  // remove tangling target path
243  edge f;
244  while(m_pred[w] != nullptr) {
245  f = m_pred[w];
246  w = f->source();
247  dropEdge(f);
248  }
249 
250  m_status[w] = NT_DONE;
251  appendEdge(e);
252 
253  return true;
254  } else {
255  // remove source path cycle
256  OGDF_ASSERT(pathType == EPT_SOURCE_PATH);
257  node w = e->target();
258 
259  node formerSource;
260  while(e->source() != w) {
261  formerSource = e->source();
262  if(e->target() != w) {
263  dropEdge(e);
264  }
265  e = m_pred[formerSource]->adjSource()->theEdge();
266  }
267 
268  dropEdge(e);
269  adj = e->adjSource();
270  v = e->source();
271  u = e->target();
272  }
273  }
274  }
275 
276  // v is a deadend
277  if(v == *this->m_s) {
278  return false;
279  } else {
280  edge e = m_pred[v];
281  dropEdge(e);
282  return findUppermostPath(e);
283  }

311  void dropEdge(const edge e) {
312  node v = e->target();
313  OGDF_ASSERT(m_pred[v] == e);
314  OGDF_ASSERT(m_status[v] == NT_PATH);
315 
316  // update path predecessor
317  m_pred[v] = nullptr;
318  m_status[v] = v == *this->m_t ? NT_PATH : NT_DONE;
319 
320  // remove edge from priority queue
321  TCap modCap = m_prioritizedEdges->priority(e);
322  m_prioritizedEdges->decrease(e, TCap(-1));
323 #ifdef OGDF_DEBUG
324  edge f = m_prioritizedEdges->topElement();
325 #endif
326  m_prioritizedEdges->pop();
327 
328  // determine the flow on this edge
329  (*this->m_flow)[e] = m_partialFlow - (modCap - (*this->m_cap)[e]);
330 
331  OGDF_ASSERT(e == f);
332  }

340  EdgePathType getPathType(const edge e) const {
341  node v = e->source();
342  node w = e->target();
343 
344  EdgePathType result = m_status[w] == NT_PATH ? EPT_UNKNOWN : EPT_NO_PATH;
345 
346  while(result == EPT_UNKNOWN) {
347  if(v == e->target() || w == *this->m_s) {
348  result = EPT_SOURCE_PATH;
349  } else if(m_pred[v] == nullptr || m_pred[w] == nullptr) {
350  result = EPT_TARGET_PATH;
351  } else if(m_pred[w]->source() == *this->m_s) {
352  result = EPT_SOURCE_PATH;
353  } else {
354  OGDF_ASSERT(w != m_pred[w]->source());
355  OGDF_ASSERT(v != m_pred[v]->source());
356 
357  w = m_pred[w]->source();
358  v = m_pred[v]->source();
359  }
360  }
361 
362  return result;
363  }

375  {
376  OGDF_ASSERT(v != w);
377 
378  for(adjEntry adjV = v->firstAdj(); adjV != nullptr; adjV = adjV->succ()) {
