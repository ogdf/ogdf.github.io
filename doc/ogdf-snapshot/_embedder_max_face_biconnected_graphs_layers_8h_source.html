507 void EmbedderMaxFaceBiconnectedGraphsLayers<T>::embed(
508  Graph& G,
509  adjEntry& adjExternal,
510  const NodeArray<T>& nodeLength,
511  const EdgeArray<T>& edgeLength,
512  const node& n /* = 0*/)
513 {
514  //Base cases (SPQR-Tree-implementatioin would crash with these inputs):
515  OGDF_ASSERT(G.numberOfNodes() >= 2);
516  if (G.numberOfEdges() <= 2)
517  {
518  edge e = G.firstEdge();
519  adjExternal = e->adjSource();
520  return;
521  }
522 
523  //First step: calculate maximum face and edge lengths for virtual edges
524  StaticSPQRTree spqrTree(G);
525  NodeArray< EdgeArray<T> > edgeLengthSkel;
526  compute(G, nodeLength, edgeLength, &spqrTree, edgeLengthSkel);
527 
528  //Second step: Embed G
529  T biggestFace = -1;
530  node bigFaceMu;
531  if (n == nullptr)
532  {
533  for(node mu : spqrTree.tree().nodes)
534  {
535  //Expand all faces in skeleton(mu) and get size of the largest of them:
536  T sizeMu = largestFaceInSkeleton(spqrTree, mu, nodeLength, edgeLengthSkel);
537  if (sizeMu > biggestFace)
538  {
539  biggestFace = sizeMu;
540  bigFaceMu = mu;
541  }
542  }
543  }
544  else
545  {
546  node* mus = new node[n->degree()];
547  int i = 0;
548  for(adjEntry adj : n->adjEntries) {
549  mus[i] = spqrTree.skeletonOfReal(adj->theEdge()).treeNode();
550  bool alreadySeenMu = false;
551  for (int j = 0; j < i && !alreadySeenMu; j++)
552  {
553  if (mus[i] == mus[j])
554  alreadySeenMu = true;
555  }
556  if (alreadySeenMu)
557  {
558  i++;
559  continue;
560  }
561  else
562  {
563  //Expand all faces in skeleton(mu) containing n and get size of
564  //the largest of them:
565  T sizeInMu = largestFaceContainingNode(spqrTree, mus[i], n, nodeLength, edgeLengthSkel);
566  if (sizeInMu > biggestFace)
567  {
568  biggestFace = sizeInMu;
569  bigFaceMu = mus[i];
570  }
571 
572  i++;
573  }
574  }
575  delete[] mus;
576  }
577 
578  bigFaceMu = spqrTree.rootTreeAt(bigFaceMu);
579 
580  NodeArray<T> thickness(spqrTree.tree());
581  bottomUpThickness(spqrTree, bigFaceMu, thickness, nodeLength, edgeLengthSkel);
582 
583  NodeArray< List<adjEntry> > newOrder(G);
584  NodeArray<bool> treeNodeTreated(spqrTree.tree(), false);
585  ListIterator<adjEntry> it;
586  adjExternal = nullptr;
587  NodeArray< ListIterator<adjEntry> > adjBeforeNodeArraySource(spqrTree.tree());
588  NodeArray< ListIterator<adjEntry> > adjBeforeNodeArrayTarget(spqrTree.tree());
589  expandEdge(spqrTree, treeNodeTreated, bigFaceMu, nullptr, nodeLength,
590  edgeLengthSkel, thickness, newOrder, adjBeforeNodeArraySource,
591  adjBeforeNodeArrayTarget, 0, 0, adjExternal, n);
592 
593  for(node v : G.nodes)
594  G.sort(v, newOrder[v]);
595 }
