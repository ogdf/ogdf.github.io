119  double angle(GenericPoint q, GenericPoint r) const
120  {
121  const double dx1 = q.m_x - m_x, dy1 = q.m_y - m_y;
122  const double dx2 = r.m_x - m_x, dy2 = r.m_y - m_y;
123 
124  // two vertices on the same place!
125  if ((dx1 == 0 && dy1 == 0)
126  || (dx2 == 0 && dy2 == 0)) {
127  return 0.0;
128  }
129 
130  const double norm = (dx1*dx1+dy1*dy1)*(dx2*dx2+dy2*dy2);
131  const double cosphi = (dx1*dx2+dy1*dy2) / sqrt(norm);
132  if (cosphi >= 1.0) return 0;
133  if (cosphi <= -1.0) return Math::pi;
134 
135  double phi = acos(cosphi);
136 
137  if (dx1*dy2 < dy1*dx2) phi = -phi;
138  if (phi < 0) phi += 2*Math::pi;
139 
140  return phi;
141  }
142 
144  double angleDegrees(GenericPoint q, GenericPoint r) const
145  {
146  return angle(q, r)*360.0/(2*Math::pi);
147  }
