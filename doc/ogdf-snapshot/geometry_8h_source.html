118  double angle(GenericPoint q, GenericPoint r) const
119  {
120  const double dx1 = q.m_x - m_x, dy1 = q.m_y - m_y;
121  const double dx2 = r.m_x - m_x, dy2 = r.m_y - m_y;
122 
123  // two vertices on the same place!
124  if ((dx1 == 0 && dy1 == 0)
125  || (dx2 == 0 && dy2 == 0)) {
126  return 0.0;
127  }
128 
129  const double norm = (dx1*dx1+dy1*dy1)*(dx2*dx2+dy2*dy2);
130  const double cosphi = (dx1*dx2+dy1*dy2) / sqrt(norm);
131  if (cosphi >= 1.0) return 0;
132  if (cosphi <= -1.0) return Math::pi;
133 
134  double phi = acos(cosphi);
135 
136  if (dx1*dy2 < dy1*dx2) phi = -phi;
137  if (phi < 0) phi += 2*Math::pi;
138 
139  return phi;
140  }
141 
143  double angleDegrees(GenericPoint q, GenericPoint r) const
144  {
145  return angle(q, r)*360.0/(2*Math::pi);
146  }
