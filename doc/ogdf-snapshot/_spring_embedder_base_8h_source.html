86  virtual void call(GraphAttributes &GA) override {
87  const Graph &G = GA.constGraph();
88  if(G.empty())
89  return;
90 
91  // all edges straight-line
92  GA.clearAllBends();
93 
94  GraphCopy GC;
95  GC.createEmpty(G);
96 
97  // compute connected component of G
98  NodeArray<int> component(G);
99  int numCC = connectedComponents(G,component);
100 
101  // intialize the array of lists of nodes contained in a CC
102  Array<List<node> > nodesInCC(numCC);
103 
104  for(node v : G.nodes)
105  nodesInCC[component[v]].pushBack(v);
106 
107  EdgeArray<edge> auxCopy(G);
108  Array<DPoint> boundingBox(numCC);
109 
110  for(int i = 0; i < numCC; ++i)
111  {
112  GC.initByNodes(nodesInCC[i],auxCopy);
113  makeSimpleUndirected(GC);
114 
115  const int n = GC.numberOfNodes();
116 
117  // special case: just one node
118  if(n == 1) {
119  node vOrig = GC.original(GC.firstNode());
120  GA.x(vOrig) = GA.y(vOrig) = 0;
121  boundingBox[i] = DPoint(0,0);
122  continue;
123  }
124 
125  callMaster(GC, GA, boundingBox[i]);
126  //std::cout << "avg. force: " << master.avgDisplacement() << std::endl;
127  //std::cout << "max. force: " << master.maxDisplacement() << std::endl;
128  }
129 
130  Array<DPoint> offset(numCC);
131  TileToRowsCCPacker packer;
132  packer.call(boundingBox,offset,m_pageRatio);
133 
134  // The arrangement is given by offset to the origin of the coordinate
135  // system. We still have to shift each node and edge by the offset
136  // of its connected component.
137 
138  for(int i = 0; i < numCC; ++i)
139  {
140  const List<node> &nodes = nodesInCC[i];
141 
142  const double dx = offset[i].m_x;
143  const double dy = offset[i].m_y;
144 
145  // iterate over all nodes in ith CC
146  ListConstIterator<node> it;
147  for(node v : nodes) {
148  GA.x(v) += dx;
149  GA.y(v) += dy;
150  }
151  }
152  }
