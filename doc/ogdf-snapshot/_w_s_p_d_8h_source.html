97  void addWSP(NodeID a, NodeID b)
98  {
99  // get the index of a free element
100  uint32_t e_index = m_numPairs++;
101 
102  // get the pair entry
103  WSPDPairInfo& e = pairInfo(e_index);
104 
105  // (a,b) is the pair we are adding
106  e.a = a;
107  e.b = b;
108 
109  // get the node info
110  WSPDNodeInfo& aInfo = nodeInfo(a);
111  WSPDNodeInfo& bInfo = nodeInfo(b);
112 
113  // if a is part of at least one pair
114  if (aInfo.numWSNodes)
115  {
116  // adjust the links
117  WSPDPairInfo& a_e = pairInfo(aInfo.lastEntry);
118  // check which one is a
119  if (a==a_e.a)
120  a_e.a_next = e_index;
121  else
122  a_e.b_next = e_index;
123  } else
124  {
125  // this pair is the first for a => set the firstEntry link
126  aInfo.firstEntry = e_index;
127  }
128 
129  // same for b: if a is part of at least one pair
130  if (bInfo.numWSNodes)
131  {
132  // adjust the links
133  WSPDPairInfo& b_e = pairInfo(bInfo.lastEntry);
134  // check which one is b
135  if (b==b_e.a)
136  b_e.a_next = e_index;
137  else
138  b_e.b_next = e_index;
139  } else
140  {
141  // this pair is the first for b => set the firstEntry link
142  bInfo.firstEntry = e_index;
143  }
144  // and the lastEntry link
145  aInfo.lastEntry = e_index;
146  bInfo.lastEntry = e_index;
147  // one more pair for each node
148  aInfo.numWSNodes++;
149  bInfo.numWSNodes++;
150  }
