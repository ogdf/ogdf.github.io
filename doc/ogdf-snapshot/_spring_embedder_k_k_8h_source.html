173  bool finished(double maxdelta)
174  {
175  if (m_prevEnergy == startVal) //first step
176  {
177  m_prevEnergy = maxdelta;
178  return false;
179  }
180 
181  double diff = m_prevEnergy - maxdelta; //energy difference
182  if (diff < 0.0) diff = -diff;
183  //#ifdef OGDF_DEBUG
184  // cout << "Finished(): maxdelta: "<< maxdelta<<" diff/prev: "<<diff / m_prevEnergy<<"\n";
185  //#endif
186  //check if we want to stop
187  bool done = (maxdelta < m_tolerance);// || (diff / m_prevEnergy) < m_tolerance);
188 
189  m_prevEnergy = maxdelta; //save previous energy level
190  m_prevLEnergy = startVal; //reset energy level for local node decision
191 
192  return done;
193  }//finished
195  bool finishedNode(double deltav)
196  {
197  if (m_prevLEnergy == startVal)
198  {
199  m_prevLEnergy = deltav;
200  return deltav == 0.0;//<m_ltolerance; //locally stable
201  }
202  //#ifdef OGDF_DEBUG
203  // cout << "Local delta: "<<deltav<<"\n";
204  //#endif
205  double diff = m_prevLEnergy - deltav;
206  //check if we want to stop
207  bool done = (deltav == 0.0 || (diff / m_prevLEnergy) < m_ltolerance);
208 
209  m_prevLEnergy = deltav; //save previous energy level
210 
211  return done;
212  }//finishedNode
