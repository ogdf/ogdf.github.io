1204  clientSibLeft(checkNode)->mark() == PQNodeRoot::PQNodeMark::Blocked)
1205  blockedSiblings++;
1206  }
1207 
1208 
1209  else
1210  // checkNode is not endmost child of
1211  // a QNode. It has not a valid
1212  // pointer to its parent.
1213  {
1214  if (clientSibLeft(checkNode)->mark() == PQNodeRoot::PQNodeMark::Unblocked)
1215  // checkNode is adjacent to an
1216  // unblocked node. Take its parent.
1217  {
1218  checkNode->mark(PQNodeRoot::PQNodeMark::Unblocked);
1219  checkNode->m_parent = clientSibLeft(checkNode)->m_parent;
1220  }
1221  else if (clientSibLeft(checkNode)->mark() == PQNodeRoot::PQNodeMark::Blocked)
1222  blockedSiblings++;
1223 
1224  if (clientSibRight(checkNode)->mark() == PQNodeRoot::PQNodeMark::Unblocked)
1225  // checkNode is adjacent to an
1226  // unblocked node. Take its parent.
1227  {
1228  checkNode->mark(PQNodeRoot::PQNodeMark::Unblocked);
1229  checkNode->m_parent = clientSibRight(checkNode)->m_parent;
1230  }
1231  else if (clientSibRight(checkNode)->mark() == PQNodeRoot::PQNodeMark::Blocked)
1232  blockedSiblings++;
1233  }
1234  }
1235 
1236  else
1237  // checkNode is son of a PNode
1238  // and children of P_NODEs
1239  // cannot be blocked.
1240  checkNode->mark(PQNodeRoot::PQNodeMark::Unblocked);
1241 
1242  if (checkNode->mark() == PQNodeRoot::PQNodeMark::Unblocked)
1243  {
1244  PQNode<T,X,Y>* parent = checkNode->m_parent;
1245 
1246  /*
1247  Get maximal consecutive set of blocked siblings.
1248  This chunk belongs to the procedure [[bubble]].
1249  The node [[checkNode]] is [[Unblocked]].
1250  If the parent of [[checkNode]] is a $Q$-Node, then we check the
1251  siblings [[checkSib]] of [[checkNode]] whether they are
1252  [[Blocked]]. If they are blocked, they have to be marked
1253  [[Unblocked]] since they are adjacent to the [[Unblocked]] node
1254  [[checkNode]]. We then have to proceed with the siblings of
1255  [[checkSib]] in order to find [[Blocked]] nodes
1256  adjacent to [[checkSib]]. This is repeated until no [[Blocked]]
1257  nodes are found any more.
1258 
1259  Observe that while running through the children of the $Q$-Node
1260  (referred by the pointer [[parent]]), their parent pointers,
1261  as well as the [[pertChildCount]] of [[parent]] are updated.
1262  Furthermore we reduce simultaneously the count [[numBlocked]].
1263  */
1264  if (blockedSiblings > 0)
1265  {
1266  if (clientSibLeft(checkNode) != nullptr)
1267  {
1268  checkSib = clientSibLeft(checkNode);
1269  PQNode<T,X,Y>* oldSib = checkNode;
1270  while (checkSib->mark() == PQNodeRoot::PQNodeMark::Blocked)
1271  {
1272  checkSib->mark(PQNodeRoot::PQNodeMark::Unblocked);
1273  checkSib->m_parent = parent;
1274  numBlocked--;
1275  parent->m_pertChildCount++;
1276  PQNode<T,X,Y>* holdSib = clientNextSib(checkSib,oldSib);
1277  oldSib = checkSib;
1278  checkSib = holdSib;
1279  // Blocked node as endmost child of a QNode.
1280  }
1281  }
1282 
1283  if (clientSibRight(checkNode) != nullptr)
1284  {
1285  checkSib = clientSibRight(checkNode);
1286  PQNode<T,X,Y>* oldSib = checkNode;
1287  while (checkSib->mark() == PQNodeRoot::PQNodeMark::Blocked)
1288  {
1289  checkSib->mark(PQNodeRoot::PQNodeMark::Unblocked);
1290  checkSib->m_parent = parent;
1291  numBlocked--;
1292  parent->m_pertChildCount++;
1293  PQNode<T,X,Y>* holdSib = clientNextSib(checkSib,oldSib);
1294  oldSib = checkSib;
1295  checkSib = holdSib;
1296  // Blocked node as endmost child of a QNode.
1297  }
1298  }
1299  }
1300 
1301  /*
1302  Process parent of [[checkNode]]
1303  After processing the siblings of the [[Unblocked]] [[checkNode]]
1304  the parent has to be processed. If [[checkNode]] is the root
1305  of the tree we do nothing except setting the flag [[offTheTop]].
1306  If it is not the root and [[parent]] has not been placed onto the
1307  queue [[processNodes]], the [[parent]] is placed on to
1308  [[processNodes]].
1309 
1310  Observe that the number [[blockCount]] is updated. Since
1311  [[checkNode]] was [[Unblocked]] all pertinent nodes adjacent
1312  to that node became [[Unblocked]] as well. Therefore the number
1313  of blocks is reduced by the number of [[Blocked]] siblings of
1314  [[checkNode]].
1315  */
1316  if (parent == nullptr)
1317  // checkNode is root of the tree.
1318  offTheTop = 1;
1319  else
1320  // checkNode is not the root.
1321  {
1322  parent->m_pertChildCount++;
1323  if (parent->mark() == PQNodeRoot::PQNodeMark::Unmarked)
1324  {
1325  processNodes.append(parent);
1326  m_pertinentNodes->pushFront(parent);
1327  parent->mark(PQNodeRoot::PQNodeMark::Queued);
1328  }
1329  }
1330 
1331  blockCount -= blockedSiblings;
1332  blockedSiblings = 0;
1333  } else {
1334  /*
1335  Process blocked [[checkNode]]
1336  Since [[checkNode]] is [[Blocked]], we cannot continue
1337  processing at this point in the Tree. We have to wait until
1338  this node becomes unblocked. So only the variables
1339  [[blockCount]] and [[numBlocked]] are updated.
1340  */
1341  blockCount += 1 - blockedSiblings;
1342  numBlocked++;
1343  }
1344  }
1345 
1346  if (blockCount == 1)
1347  {
1348  /*
1349  If [[blockCount]] $= 1$ enter [[m_pseudoRoot]] to the tree
1350  In case that the root of the pertinent subtree is a $Q$-node
1351  with empty children on both sides and the pertinent children
1352  in the middle, it is possible that the $PQ$-tree is reducible.
1353  But since the sequence of pertinent children of the $Q$-node is
1354  blocked, the procedure is not able to find the parent of its
1355  pertinent children. This is due to the fact that the interior
1356  children of a $Q$-node do not have a valid parent pointer.
1357 
1358  So the root of the pertinent subtree is not known, hence cannot be
1359  entered into the processing queue used in the function call [[Reduce]]
1360  (see \ref{Reduce}). To solve this problem a special node only designed
1361  for this cases is used: [[m_pseudoRoot]]. It simulates the root of the
1362  pertinent subtree. This works out well, since for this node the only
1363  possible template maching is [[templateQ3]] (see \ref{templateQ3}),
1364  where no pointers to the endmost children of a $Q$-node are used.
1365  */
1366  while (!blockedNodes.empty())
1367  {
1368  PQNode<T,X,Y>* checkNode = blockedNodes.popRet();
1369  if (checkNode->mark() == PQNodeRoot::PQNodeMark::Blocked)
1370  {
1371  checkNode->mark(PQNodeRoot::PQNodeMark::Unblocked);
1372  checkNode->m_parent = m_pseudoRoot;
1373  m_pseudoRoot->m_pertChildCount++;
1374  OGDF_ASSERT(!checkNode->endmostChild());
1375  //Blocked node as endmost child of a QNode.
1376  }
1377  }
1378  }
1379 
1380  return true;
1381 }

1383 template<class T,class X,class Y>
1384 bool PQTree<T,X,Y>::checkChain(
1385  PQNode<T,X,Y> *nodePtr,
1386  PQNode<T,X,Y> *firstFull,
1387  PQNode<T,X,Y> **seqStart,
1388  PQNode<T,X,Y> **seqEnd)
1389 {
1390  /* Variables:
1391  * - \a fullCount counts the number of children that are
1392  * discovered by the function checkChain(). This is necessary, since
1393  * checkChain() is used by two template matching functions
1394  * templateQ2() and templateQ3() where in the latter case the
1395  * pointer \a firstFull may point to any full child in the front of the Q-node
1396  * \a nodePtr.
1397  * - \a notFull is set 1 when an empty child is encountered.
1398  * - \a checkNode is the actual node that is examined.
1399  * - \a leftNext is the next node that has to be examined on the
1400  * left side of \a firstFull.
1401  * - \a leftOld is the node that has been examined right before
1402  * \a checkNode on the left side of \a firstFull.
1403  * - \a rightNext is the next node that has to be examined on the
1404  * right side of \a firstFull. Not needed when checkChain() was
1405  * called by templateQ2().
1406  * - \a rightOld is the node that has been examined right before
1407  * \a checkNode on the right side of \a firstFull.
1408  * Not needed when checkChain() was called by templateQ2().
1409  */
1410  bool notFull = false;
1411  int fullCount = nodePtr->fullChildren->size();
1412  fullCount--; // firstFull does have a Full label.
1413 
1414  /*
1415  Start at the [[firstFull]] child sweeping through the full children on
1416  the left side of [[firstfull]]. It stops as soon as a nonfull child is
1417  detected. The last found full child is stored in [[seqEnd]].
1418  */
1419  PQNode<T,X,Y> *leftNext = clientSibLeft(firstFull);
1420  (*seqEnd) = firstFull;
1421  if (leftNext != nullptr)
1422  {
1423  if (leftNext->status() == PQNodeRoot::PQNodeStatus::Full) {
1424  fullCount--;
1425 
1426  PQNode<T,X,Y> *leftOld = firstFull;
1427  PQNode<T,X,Y> *checkNode = leftNext;
1428 
1429  while (fullCount > 0 && !notFull)
1430  // There are still full children to be
1431  // counted, and no empty child has been
1432  // encountered on this side.
1433  {
1434  leftNext = clientNextSib(checkNode,leftOld);
1435  if (leftNext != nullptr && leftNext->status() == PQNodeRoot::PQNodeStatus::Full)
1436  fullCount--;
1437  else
1438  notFull = true;
1439  leftOld = checkNode;
1440  checkNode = leftNext;
1441  }
1442 
1443  if (checkNode != nullptr && checkNode->status() == PQNodeRoot::PQNodeStatus::Full)
1444  (*seqEnd) = checkNode;
1445 
1446  else {
1447  //searching consecutive sequence in Q2 or Q3.
1448  OGDF_ASSERT(leftOld != nullptr);
1449  OGDF_ASSERT(leftOld->status() == PQNodeRoot::PQNodeStatus::Full);
1450  (*seqEnd) = leftOld;
1451  }
1452 
1453  } else {
1454  (*seqEnd) = firstFull;
1455  }
1456  }
1457 
1458  /*
1459  Start at the [[firstFull]] child sweeping through the full children on
1460  the right side of [[firstfull]]. It stops as soon as a nonfull child is
1461  detected.
1462  */
1463  notFull = false;
1464  PQNode<T,X,Y> *rightNext = clientSibRight(firstFull);
1465  (*seqStart) = firstFull;
1466  if(rightNext != nullptr)
1467  {
1468  if (rightNext->status() == PQNodeRoot::PQNodeStatus::Full) {
1469  fullCount--;
1470 
1471  PQNode<T,X,Y> *rightOld = firstFull;
1472  PQNode<T,X,Y> *checkNode = rightNext;
1473 
1474  while (fullCount > 0 && !notFull)
1475  // There are still full children to be
1476  // counted, and no empty child has been
1477  // encountered on this side.
1478  {
1479  rightNext = clientNextSib(checkNode,rightOld);
1480  if (rightNext != nullptr && rightNext->status() == PQNodeRoot::PQNodeStatus::Full)
1481  fullCount--;
1482  else
1483  notFull = true;
1484  rightOld = checkNode;
1485  checkNode = rightNext;
1486  }
1487  if (checkNode != nullptr && checkNode->status() == PQNodeRoot::PQNodeStatus::Full)
1488  (*seqStart) = checkNode;
1489 
1490  else {
1491  OGDF_ASSERT(rightOld != nullptr);
1492  OGDF_ASSERT(rightOld->status() == PQNodeRoot::PQNodeStatus::Full);
1493  (*seqStart) = rightOld;
1494  //searching consecutive seqeuence in Q2 or Q3.
1495  }
1496 
1497  } else {
1498  (*seqStart) = firstFull;
1499  }
1500  }
1501 
1502  if (firstFull == (*seqEnd))
1503  {
1504  PQNode<T,X,Y> *checkNode = (*seqEnd);
1505  (*seqEnd) = (*seqStart);
1506  (*seqStart) = checkNode;
1507  }
1508 
1509  if (fullCount == 0)
1510  // All full children occupy a consecutive
1511  // sequence.
1512  return true;
1513  else
1514  return false;
1515 }

1542 template<class T,class X,class Y>
1543 void PQTree<T,X,Y>::Cleanup()
1544 {
1545  /* In order to free the allocated memory, all nodes of the
1546  * tree have to be deleted, hence there destructors are called.
1547  * In order to achieve this, we start at the root of the tree and go down the
1548  * tree to the leaves for reaching every node. When a node is processed,
1549  * (besides the #m_root, this will always be the node \a checkNode)
1550  * the pointers of all its children are stored in a queue \a helpqueue and
1551  * then the processed node is deleted.
1552  *
1553  * The use of a queue \a helpqueue is a must, since the nodes do not
1554  * have pointers to all of their children, as the children mostly do
1555  * not have a pointer to their parent.
1556  *
1557  * It might look weird at the first glance that the function Cleanup()
1558  * calls the function emptyAllPertinentNodes(), but if some nodes were removed
1559  * during a reduction, they were stored in the stack #m_pertinentNodes.
1560  * These nodes have to be deleted as well
1561  * which is provided by the function emptyAllPertinentNodes().
1562  */
1563  PQNode<T,X,Y>* nextSon = nullptr;
1564  PQNode<T,X,Y>* lastSon = nullptr;
1565 
1566  Queue<PQNode<T,X,Y>*> helpqueue;
1567 
1568  if (m_root != nullptr)
1569  {
1570  emptyAllPertinentNodes();
1571  PQNode<T,X,Y>* oldSib = nullptr;
1572 
1573  /*
1574  Process the [[m_root]] of the [[PQTree]]. Before deleting [[m_root]],
1575  pointers to all its children are stored in the queue [[helpqueue]].
1576  */
1577  if (m_root->type() == PQNodeRoot::PQNodeType::PNode)
1578  {
1579  if (m_root->m_referenceChild != nullptr)
1580  {
1581  PQNode<T,X,Y>* firstSon = m_root->m_referenceChild;
1582  helpqueue.append(firstSon);
1583 
1584  if (firstSon->m_sibRight != nullptr)
1585  nextSon = firstSon->m_sibRight;
1586  while (nextSon != firstSon)
1587  {
1588  helpqueue.append(nextSon);
1589  nextSon = nextSon->m_sibRight;
1590  }
1591  }
1592  }
1593  else if (m_root->type() == PQNodeRoot::PQNodeType::QNode)
1594  {
1595  PQNode<T,X,Y>* firstSon = m_root->m_leftEndmost;
1596  helpqueue.append(firstSon);
1597 
1598  lastSon = m_root->m_rightEndmost;
1599  helpqueue.append(lastSon);
1600 
1601  nextSon = lastSon->getNextSib(oldSib);
1602  oldSib = lastSon;
1603  while (nextSon != firstSon)
1604  {
1605  helpqueue.append(nextSon);
1606  PQNode<T,X,Y>* holdSib = nextSon->getNextSib(oldSib);
1607  oldSib = nextSon;
1608  nextSon = holdSib;
1609  }
1610  }
1611 
1612 
1613  CleanNode(m_root);
1614  delete m_root;
1615 
1616  while (!helpqueue.empty())
1617  {
1618  PQNode<T,X,Y>* checkNode = helpqueue.pop();
1619 
1620  /*
1621  Process an arbitrary node [[checkNode]] of the [[PQTree]].
1622  Before deleting [[checkNode]],
1623  pointers to all its children are stored in the queue [[helpqueue]].
1624  */
1625  if (checkNode->type() == PQNodeRoot::PQNodeType::PNode)
1626  {
1627  if (checkNode->m_referenceChild != nullptr)
1628  {
1629  PQNode<T,X,Y>* firstSon = checkNode->m_referenceChild;
1630  helpqueue.append(firstSon);
1631 
1632  if (firstSon->m_sibRight != nullptr)
1633  nextSon = firstSon->m_sibRight;
1634  while (nextSon != firstSon)
1635  {
1636  helpqueue.append(nextSon);
1637  nextSon = nextSon->m_sibRight;
1638  }
1639  }
1640  }
1641  else if (checkNode->type() == PQNodeRoot::PQNodeType::QNode)
1642  {
1643  oldSib = nullptr;
1644 
1645  PQNode<T,X,Y>*firstSon = checkNode->m_leftEndmost;
1646  helpqueue.append(firstSon);
1647 
1648  lastSon = checkNode->m_rightEndmost;
1649  helpqueue.append(lastSon);
1650 
1651  nextSon = lastSon->getNextSib(oldSib);
1652  oldSib = lastSon;
1653  while (nextSon != firstSon)
1654  {
1655  helpqueue.append(nextSon);
1656  PQNode<T,X,Y>* holdSib = nextSon->getNextSib(oldSib);
1657  oldSib = nextSon;
1658  nextSon = holdSib;
1659  }
1660  }
1661 
1662  CleanNode(checkNode);
1663  delete checkNode;
1664  }
1665  }
1666 
1667  CleanNode(m_pseudoRoot);
1668  delete m_pseudoRoot;
1669 
1670  delete m_pertinentNodes;
1671 
1672  m_root = nullptr;
1673  m_pertinentRoot = nullptr;
1674  m_pseudoRoot = nullptr;
1675  m_pertinentNodes = nullptr;
1676 
1677  m_numberOfLeaves = 0;
1678  m_identificationNumber = 0;
1679 }

1752 template<class T,class X,class Y>
1753 PQNode<T,X,Y>* PQTree<T,X,Y>::createNodeAndCopyFullChildren(
1754  List<PQNode<T,X,Y>*> *fullNodes)
1755 {
1756  /* Variables:
1757  * - \a newNode stores the adress of the new allocated P-node or
1758  * the adress of the only full child.
1759  * - \a oldSon is a variable used for adding the full nodes as
1760  * children to the new P-node.
1761  * - \a firstSon stores the adress of the first detected full
1762  * child. It is needed for adding the full nodes as
1763  * children to the new P-node.
1764  * - \a checkSon is a variable used for adding the full nodes as
1765  * children to the new P-node.
1766  * - \a newPQnode is used for proper allocation of the new P-node.
1767  */
1768  PQNode<T,X,Y> *newNode = nullptr;
1769 
1770  if (fullNodes->size() == 1)
1771  {
1772  /*
1773  There is just one full child. So no new $P$-node is created. The
1774  full child is copied as child to the [[partialChild]].
1775  */
1776  newNode = fullNodes->popFrontRet();
1777  removeChildFromSiblings(newNode);
1778  }
1779 
1780  else
1781  {
1782  /*
1783  This chunk belongs to the function [[createNodeAndCopyFullChildren]].
1784  There is more than one full child, so a new $P$-node is created.
1785  This chunk first allocates the memory for the new $P$-node that will
1786  be stored in [[newNode]]. Then it pops the nodes out of the stack
1787  [[fullNodes]] and introduces them as sons of [[newNode]].
1788  Popping the nodes out of the stack implies at the same time
1789  that they are removed from the [[fullChildren]] stack of the
1790  $P$-node of their parent.
1791  */
1792  newNode = new PQInternalNode<T,X,Y>(m_identificationNumber++, PQNodeRoot::PQNodeType::PNode, PQNodeRoot::PQNodeStatus::Full);
1793  m_pertinentNodes->pushFront(newNode);
1794  newNode->m_pertChildCount = fullNodes->size();
1795  newNode->m_childCount = fullNodes->size();
1796 
1797  /*
1798  The first node is copied separately, since we need the pointer to it
1799  for setting the pointers to the siblings of the next full nodes.
1800  */
1801  PQNode<T,X,Y> *firstSon = fullNodes->popFrontRet();
1802  removeChildFromSiblings(firstSon);
1803  newNode->fullChildren->pushFront(firstSon);
1804  firstSon->m_parent = newNode;

1884 template<class T,class X,class Y>
1885 void PQTree<T,X,Y>::exchangeNodes(
1886  PQNode<T,X,Y> *oldNode,
1887  PQNode<T,X,Y> *newNode)
1888 {
1889  if (oldNode->m_referenceParent != nullptr)
1890  {
1891  /*
1892  The node [[oldNode]] is connected to its parent
1893  via the reference pointer of the doubly linked list. The [[newNode]]
1894  will be the new reference child and is linked via the reference
1895  pointers to the $P$-node.
1896  */
1897  oldNode->m_referenceParent->m_referenceChild = newNode;
1898  newNode->m_referenceParent = oldNode->m_referenceParent;
1899  oldNode->m_referenceParent = nullptr;
1900  }
1901 
1902  else if (oldNode->endmostChild())
1903  {
1904  /*
1905  The [[oldNode]] is endmost child of a Q-node. So its parent
1906  contains an extra pointer to [[oldNode]]. Link the parent of
1907  [[oldNode]] to [[newNode]] via this pointer and make [[newNode]]
1908  endmost child of its new parent.
1909  */
1910  if (oldNode->m_parent->m_leftEndmost == oldNode)
1911  oldNode->m_parent->m_leftEndmost = newNode;
1912  else if (oldNode->m_parent->m_rightEndmost == oldNode)
1913  oldNode->m_parent->m_rightEndmost = newNode;
1914  }
1915 
1916  if (oldNode->m_sibLeft == oldNode && oldNode->m_sibRight == oldNode) {
1917  /*
1918  Two possible cases are occured.
1919  \begin{enumerate}
1920  \item [[oldNode]] is the only child of a $P$-node. In order to
1921  implement the doubly linked list of children of the $P$-node, the sibling
1922  pointers of [[newNode]] are set to [[newNode]].
1923  \item [[oldNode]] is the [[m_root]] of the $PQ$-tree. Since
1924  by our definition of the $PQ$-tree the sibling pointers of the
1925  [[m_root]] point to the root itself, (i.e. to make sure that
1926  checking for the endmost child property is also valid for the root)
1927  the sibling pointers of [[newNode]] are set to [[newNode]] as well.
1928  \end{enumerate}
1929  */
1930  oldNode->m_sibLeft = nullptr;
1931  oldNode->m_sibRight = nullptr;
1932  if (oldNode->m_parent == nullptr)
1933  {
1934  newNode->m_sibLeft = newNode;
1935  newNode->m_sibRight = newNode;
1936  }
1937  else
1938  {
1939  newNode->m_sibLeft = newNode;
1940  newNode->m_sibRight = newNode;
1941  }
1942  }
1943  else
1944  {
1945  OGDF_ASSERT(!(oldNode->m_sibLeft == oldNode));
1946  //sibling pointers of old node are not compatible
1947  OGDF_ASSERT(!(oldNode->m_sibRight == oldNode));
1948  //sibling pointers of old node are not compatible.
1949  }
1950  /*
1951  Manage the exchange of [[oldNode]] and [[newNode]] according to
1952  [[oldNode]]'s siblings. The chunk checks both siblings of
1953  [[oldNode]] and resets the sibling pointers of [[oldNode]]'s siblings
1954  as well as the sibling pointers of [[newNode]].
1955  */
1956  if (oldNode->m_sibLeft != nullptr)
1957  {
1958  if (oldNode->m_sibLeft->m_sibRight == oldNode)
1959  oldNode->m_sibLeft->m_sibRight = newNode;
1960  else {
1961  // Sibling was not connected to child?
1962  OGDF_ASSERT(oldNode->m_sibLeft->m_sibLeft == oldNode);
1963  oldNode->m_sibLeft->m_sibLeft = newNode;
1964  }
1965  newNode->m_sibLeft = oldNode->m_sibLeft;
1966  oldNode->m_sibLeft = nullptr;
1967  }
1968 
1969  if (oldNode->m_sibRight != nullptr)
1970  {
1971  if (oldNode->m_sibRight->m_sibLeft == oldNode)
1972  oldNode->m_sibRight->m_sibLeft = newNode;
1973  else {
1974  // Sibling was not connected to child?
1975  OGDF_ASSERT(oldNode->m_sibRight->m_sibRight == oldNode);
1976  oldNode->m_sibRight->m_sibRight = newNode;
1977  }
1978  newNode->m_sibRight = oldNode->m_sibRight;
1979  oldNode->m_sibRight = nullptr;
1980  }
1981 
1982  newNode->m_parentType = oldNode->m_parentType;
1983  newNode->m_parent = oldNode->m_parent;
1984 }
