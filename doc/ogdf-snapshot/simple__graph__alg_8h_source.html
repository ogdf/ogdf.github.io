141 void makeParallelFree(Graph &G, EDGELIST &parallelEdges)
142 {
143  parallelEdges.clear();
144  if (G.numberOfEdges() <= 1) return;
145 
146  SListPure<edge> edges;
147  parallelFreeSort(G,edges);
148 
149  SListConstIterator<edge> it = edges.begin();
150  edge ePrev = *it++, e;
151  bool bAppend = true;
152  while(it.valid()) {
153  e = *it++;
154  if (ePrev->source() == e->source() && ePrev->target() == e->target()) {
155  G.delEdge(e);
156  if (bAppend) { parallelEdges.pushBack(ePrev); bAppend = false; }
157  } else {
158  ePrev = e; bAppend = true;
159  }
160  }
161 }

240 void getParallelFreeUndirected(const Graph &G, EdgeArray<EDGELIST> &parallelEdges)
241 {
242  if (G.numberOfEdges() <= 1) {
243  return;
244  }
245 
246  SListPure<edge> edges;
247  EdgeArray<int> minIndex(G), maxIndex(G);
248  parallelFreeSortUndirected(G,edges,minIndex,maxIndex);
249 
250  SListConstIterator<edge> it = edges.begin();
251  edge ePrev = *it++, e;
252  while (it.valid()) {
253  e = *it++;
254  if (minIndex[ePrev] == minIndex[e] && maxIndex[ePrev] == maxIndex[e]) {
255  parallelEdges[ePrev].pushBack(e);
256  } else {
257  ePrev = e;
258  }
259  }
260 }

281 template <class EDGELIST = SListPure<edge>>
282 void makeParallelFreeUndirected(
283  Graph &G,
284  EDGELIST *parallelEdges = nullptr,
285  EdgeArray<int> *cardPositive = nullptr,
286  EdgeArray<int> *cardNegative = nullptr)
287 {
288  if (parallelEdges != nullptr) { parallelEdges->clear(); }
289  if (cardPositive != nullptr) { cardPositive->fill(0); }
290  if (cardNegative != nullptr) { cardNegative->fill(0); }
291 
292  if (G.numberOfEdges() <= 1) {
293  return;
294  }
295 
296  EdgeArray<SListPure<edge>> parEdges(G);
297  getParallelFreeUndirected(G, parEdges);
298 
299  for (edge e : G.edges) {
300  for (edge parEdge : parEdges(e)) {
301  if (cardPositive != nullptr && e->source() == parEdge->source()) {
302  (*cardPositive)[e]++;
303  }
304  if (cardNegative != nullptr && e->source() == parEdge->target()) {
305  (*cardNegative)[e]++;
306  }
307  G.delEdge(parEdge);
308  if (parallelEdges != nullptr) {
309  parallelEdges->pushBack(e);
310  }
311  }
312  }
313 }
