146 template <class EDGELIST>
147 void makeParallelFree(Graph &G, EDGELIST &parallelEdges)
148 {
149  parallelEdges.clear();
150  if (G.numberOfEdges() <= 1) return;
151 
152  SListPure<edge> edges;
153  parallelFreeSort(G,edges);
154 
155  SListConstIterator<edge> it = edges.begin();
156  edge ePrev = *it++, e;
157  bool bAppend = true;
158  while(it.valid()) {
159  e = *it++;
160  if (ePrev->source() == e->source() && ePrev->target() == e->target()) {
161  G.delEdge(e);
162  if (bAppend) { parallelEdges.pushBack(ePrev); bAppend = false; }
163  } else {
164  ePrev = e; bAppend = true;
165  }
166  }
167 }
