142 void makeParallelFree(Graph &G, EDGELIST &parallelEdges)
143 {
144  parallelEdges.clear();
145  if (G.numberOfEdges() <= 1) return;
146 
147  SListPure<edge> edges;
148  parallelFreeSort(G,edges);
149 
150  SListConstIterator<edge> it = edges.begin();
151  edge ePrev = *it++, e;
152  bool bAppend = true;
153  while(it.valid()) {
154  e = *it++;
155  if (ePrev->source() == e->source() && ePrev->target() == e->target()) {
156  G.delEdge(e);
157  if (bAppend) { parallelEdges.pushBack(ePrev); bAppend = false; }
158  } else {
159  ePrev = e; bAppend = true;
160  }
161  }
162 }

241 void getParallelFreeUndirected(const Graph &G, EdgeArray<EDGELIST> &parallelEdges)
242 {

283 void makeParallelFreeUndirected(
