127 template<typename MNR_T, typename C_T>
128 inline MNR_T mortonNumber(C_T ix, C_T iy)
129 {
130  MNR_T x = (MNR_T)ix;
131  MNR_T y = (MNR_T)iy;
132  // bit length of the result
133  const unsigned int BIT_LENGTH = static_cast<unsigned int>(sizeof(MNR_T)) << 3;
134  // set all bits
135  MNR_T mask = 0x0;
136  mask = ~mask;
137 
138  for (unsigned int i = (BIT_LENGTH >> 1);i>0; i = i >> 1) {
139  // increase frequency
140  mask = mask ^ (mask << i);
141  x = (x | (x << i)) & mask;
142  y = (y | (y << i)) & mask;
143  }
144  return x | (y << 1);
145 }

151 {
152  // bit length of the coordinates
153  unsigned int BIT_LENGTH = static_cast<unsigned int>(sizeof(C_T)) << 3;
154  // set least significant bit

169 inline uint32_t mostSignificantBit(T n)
170 {
171  uint32_t BIT_LENGTH = static_cast<uint32_t>(sizeof(T)) << 3;
172  T mask = 0x1;
173  mask = mask << (BIT_LENGTH - 1);
