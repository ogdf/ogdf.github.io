171 
172  // 2. Copy block embedding into graph embedding and call recursively
173  // embedBlock for all cut vertices in bT
174  CombinatorialEmbedding CE(blockGraph);
175  face f = CE.leftFace(m_adjExternal);
176 
177  if (*pAdjExternal == nullptr) {
178  node on = pBCTree->original(mapNodeToH[m_adjExternal->theNode()]);
179 
180  for (adjEntry ae : on->adjEntries) {
181  if (ae->theEdge() == pBCTree->original(mapEdgeToH[m_adjExternal->theEdge()])) {
182  *pAdjExternal = ae->twin();
183  break;
184  }
185  }
186  }
187 
188  for (node nSG : blockGraph.nodes) {
189  node nH = mapNodeToH[nSG];
190  node nG = pBCTree->original(nH);
191  adjEntry ae = nSG->firstAdj();
192  ListIterator<adjEntry>* pAfter = pBCTree->bcproper(nG) == cT ? &after : new ListIterator<adjEntry>;
193 
194  if (pBCTree->typeOfGNode(nG) == BCTree::GNodeType::CutVertex) {
195  node cT2 = pBCTree->bcproper(nG);
196  bool doRecurse = true;
197 
198  if (cT2 == cT) {
199  node parent_bT_of_cT2 = nullptr;
200 
201  for (adjEntry adj : cT2->adjEntries) {
202  if (adj->theEdge()->source() == cT2) {
203  parent_bT_of_cT2 = adj->twinNode();
204  break;
205  }
206  }
207 
208  OGDF_ASSERT(parent_bT_of_cT2 != nullptr);
209 
210  if (treeNodeTreated[parent_bT_of_cT2]) {
211  doRecurse = false;
212  }
213  }
214 
215  // (if exists) find adjacency entry of nSG which lies on external face f:
216  for (adjEntry aeFace : f->entries) {
217  if (aeFace->theNode() == nSG) {
218  ae = aeFace->succ() == nullptr ? nSG->firstAdj() : aeFace->succ();
219  break;
220  }
221 
222  }
223 
224  if (doRecurse) {
225  for (adjEntry adj : cT2->adjEntries) {
226  node bT2 = adj->theEdge()->opposite(cT2);
227 
228  if (!treeNodeTreated[bT2]) {
229  embedBlock(bT2, cT2, *pAfter);
230  }
231  }
232  }
233  }
234 
235  // embed all edges of block bT:
236  bool after_ae = true;
237  for (adjEntry aeNode = ae; after_ae || aeNode != ae; aeNode = aeNode->succ() == nullptr ? nSG->firstAdj() : aeNode->succ()) {
238  edge e = pBCTree->original(mapEdgeToH[aeNode->theEdge()]);
239  if (nG == e->source()) {
240  if (pAfter->valid()) {
241  *pAfter = newOrder[nG].insertAfter(e->adjSource(), *pAfter);
242  } else {
243  *pAfter = newOrder[nG].pushBack(e->adjSource());
244  }
245  } else {
246  if (pAfter->valid()) {
247  *pAfter = newOrder[nG].insertAfter(e->adjTarget(), *pAfter);
248  } else {
249  *pAfter = newOrder[nG].pushBack(e->adjTarget());
250  }
251  }
252 
253  after_ae &= aeNode->succ() != nullptr;
254  }
255 
256  if (*pAfter != after) {
257  delete pAfter;
258  }
259  }
