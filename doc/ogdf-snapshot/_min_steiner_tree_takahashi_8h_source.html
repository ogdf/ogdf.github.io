159 T MinSteinerTreeTakahashi<T>::terminalDijkstra(const EdgeWeightedGraph<T> &wG,
160  EdgeWeightedGraphCopy<T> &intermediateTerminalSpanningTree, const node s, int numberOfTerminals,
161  const NodeArray<bool> &isTerminal)
162 {
163  NodeArray<edge> predecessor(wG, nullptr);
164  NodeArray<T> distance(wG, numeric_limits<T>::max());
165  distance[s] = 0;
166  NodeArray<T> bestDistance(wG, numeric_limits<T>::max());
167  bestDistance[s] = 0;
168  NodeArray<bool> isInQueue(wG, true);
169 
170  PrioritizedMapQueue<node, T> queue(wG); //priority queue
171  for (node v : wG.nodes) {
172  queue.push(v, distance[v]);
173  }
174 
175  T mstWeight = 0;
176  int terminalsFound = 1;
177  while (!queue.empty() && terminalsFound < numberOfTerminals) {
178  node v = queue.topElement();
179  queue.pop();
180  isInQueue[v] = false;
181  bestDistance[v] = distance[v];
182  if (isTerminal[v]
183  && distance[v] > 0) {
184  ++terminalsFound;
185  // insert path from new node to old tree
186  node tmpT = intermediateTerminalSpanningTree.newNode(v);
187  while (distance[v] > 0) {
188  distance[v] = 0;
189  queue.push(v, distance[v]);
190  isInQueue[v] = true;
191  const edge e = predecessor[v];
192  OGDF_ASSERT(e);
193  const node w = e->opposite(v);
194  node tmpS = intermediateTerminalSpanningTree.copy(w);
195  if (!tmpS) {
196  tmpS = intermediateTerminalSpanningTree.newNode(w);
197  }
198  edge tmpE;
199  if (e->target() == v) {
200  tmpE = intermediateTerminalSpanningTree.newEdge(tmpS, tmpT, wG.weight(e));
201  } else {
202  tmpE = intermediateTerminalSpanningTree.newEdge(tmpT, tmpS, wG.weight(e));
203  }
204  mstWeight += wG.weight(e);
205  intermediateTerminalSpanningTree.setEdge(e, tmpE);
206  tmpT = tmpS;
207  v = w;
208  }
209  } else { // !isTerminal[v] || distance[v] == 0
210  for(adjEntry adj : v->adjEntries) {
211  const node w = adj->twinNode();
212  const edge e = adj->theEdge();
213  if (distance[w] > distance[v] + wG.weight(e)
214  && bestDistance[w] >= distance[w]) {
215  distance[w] = distance[v] + wG.weight(e);
216  if (!isInQueue[w]) {
217  queue.push(w, distance[w]);
218  isInQueue[w] = true;
219  } else {
220  queue.decrease(w, distance[w]);
221  }
222  predecessor[w] = e;
223  }
224  }
225  }
226  }
227  return mstWeight;
228 }
