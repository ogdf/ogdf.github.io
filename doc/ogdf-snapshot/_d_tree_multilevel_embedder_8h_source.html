145 template<int Dim>
146 void DTreeMultilevelEmbedder<Dim>::call(const Graph& graph, NodeArray<NodeCoords>& resultCoords)
147 {
148  // we call this just in case
149  resultCoords.init(graph);
150 
151  // make sure the graph is connected
152  OGDF_ASSERT(isConnected(graph));
153 
154  // setup the multilevel step
155  GalaxyLevel* pLevelBegin = new GalaxyLevel(graph);
156 
157  // this is the coarsest level with at most m_levelMaxNumNodes
158  GalaxyLevel* pLevelEnd = pLevelBegin->buildLevelsUntil(m_levelMaxNumNodes);
159 
160  // this array will hold the layout information of the parent node in the coarser level.
161  // furthermore this will keep the final result.
162  NodeArray<NodeCoords>& parentPosition = resultCoords;
163 
164  double currNumIterations = m_numIterationsFinestLevel;
165  double currThreshold = m_thresholdFinestLevel;
166 
167  int numLevels = 0;
168  // loop from the coarsest to the finest level
169  for (GalaxyLevel* pCurrLevel = pLevelBegin; pCurrLevel; pCurrLevel = pCurrLevel->nextCoarser())
170  {
171  numLevels++;
172  currNumIterations *= m_numIterationsFactorPerLevel;
173  currThreshold *= m_thresholdFactorPerLevel;
174  }
175 
176  int currLevelIndex = numLevels - 1;
177  // now we loop from the coarsest to the finest level
178  for (GalaxyLevel* pCurrLevel = pLevelEnd; pCurrLevel; pCurrLevel = pCurrLevel->nextFiner())
179  {
180  currNumIterations /= m_numIterationsFactorPerLevel;
181  currThreshold /= m_thresholdFactorPerLevel;
182 
183  // new embedder instance for the current level
184  DTreeEmbedder<Dim>* pEmbedder = new DTreeEmbedder<Dim>(pCurrLevel->graph());
185 
186  // if this is coarsest one
187  if (pCurrLevel->isCoarsestLevel())
188  {
189  // we cannot init from the parent level, do it random
190  for (node v = pCurrLevel->graph().firstNode(); v; v = v->succ())
191  {
192  // for all dims
193  for (int d = 0; d < Dim; d++)
194  {
195  // set the position to some random value
196  pEmbedder->setPosition(v, d, randomDouble(-1.0, 1.0));
197  }
198  }
199  } else // if we have a parent level
200  {
201  // iterate over all nodes
202  for (node v = pCurrLevel->graph().firstNode(); v; v = v->succ())
203  {
204  // this is a node on the coarser level we already processed
205  node v_parent = pCurrLevel->parent(v);
206 
207  // now init the position from the parent.
208  for (int d = 0; d < Dim; d++)
209  {
210  // get the position of the parent
211  double offset = parentPosition[v_parent].coords[d] * m_scaleFactorPerLevel;
212 
213  // set v's position to the parents pos with some random
214  pEmbedder->setPosition(v, d, offset + randomDouble(-1.0, 1.0));
215  }
216  }
217  } // we have some proper initial coordinates for the nodes
218 
219 
220  if (m_useMultilevelWeights)
221  {
222  // we cannot init from the parent level, do it random
223  for (node v = pCurrLevel->graph().firstNode(); v; v = v->succ())
224  {
225  pEmbedder->setMass(v, pCurrLevel->weight(v));
226  }
227 
228  for (edge e = pCurrLevel->graph().firstEdge(); e; e = e->succ())
229  {
230  pEmbedder->setEdgeWeight(e, pCurrLevel->edgeWeight(e));
231  }
232  }
233 
234 
235  if (pCurrLevel->isCoarsestLevel())
236  {
237  int numIterations = std::min(std::max(m_minIterationsPerLevel, (int)m_numIterationsCoarsestLevel), m_maxIterationsPerLevel);
238 
239  pEmbedder->scaleNodes(3.0);
240  pEmbedder->doIterationsNewton(numIterations, currThreshold, RepForceFunctionNewton<Dim, 1>, AttrForceFunctionPow<Dim, 2>);
241  pEmbedder->scaleNodes(1.0 / 3.0);
242  pEmbedder->doIterationsNewton(numIterations, currThreshold, RepForceFunctionNewton<Dim, 2>, AttrForceFunctionPow<Dim, 2>);
243  }
244  else
245  {
246  int numIterations = std::min(std::max(m_minIterationsPerLevel, (int)currNumIterations), m_maxIterationsPerLevel);
247  pEmbedder->scaleNodes(3.0);
248  pEmbedder->doIterationsNewton(numIterations, currThreshold, RepForceFunctionNewton<Dim, 1>, AttrForceFunctionPow<Dim, 2>);
249  pEmbedder->scaleNodes(1.0 / 3.0);
250  pEmbedder->doIterationsNewton(numIterations, currThreshold, RepForceFunctionNewton<Dim, 2>, AttrForceFunctionPow<Dim, 2>);
251  }
252  // run the layout
253 
254 
255  // we now have to backup the positions before getting rid of the embedder
256  parentPosition.init(pCurrLevel->graph());
257 
258  // iterate over all nodes
259  for (node v = pCurrLevel->graph().firstNode(); v; v = v->succ())
260  {
261  // for all coords
262  for (int d = 0; d < Dim; d++)
263  {
264  parentPosition[v].coords[d] = pEmbedder->position(v, d);
265  }
266  }
267 
268  currLevelIndex--;
269  // now we can free the embedder instance
270  delete pEmbedder;
271  }
272 
273  // we are done with the layout. It is saved now in the parentposition nodearray.
274  // which is a reference to the result array
275 }
