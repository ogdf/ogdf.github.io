303  EdgeArray<edge>& eSG_to_eG,
304  NodeArray<node>& nG_to_nSG,
305  EdgeArray<edge>& eG_to_eSG)
306 {
307  node nSG = SG.newNode();
308  nodeLengthSG[nSG] = nodeLengthG[nG];
309  nG_to_nSG[nG] = nSG;
310  nSG_to_nG[nSG] = nG;
311  nodeVisited[nG] = true;
312 
313  for(adjEntry adj : nG->adjEntries) {
314  edge eG = adj->theEdge();
315  if (!nodeVisited[eG->source()])
316  recursion(SG, nodeVisited, edgeVisited, eG->source(),
317  nodeLengthG, nodeLengthSG, edgeLengthG, edgeLengthSG,
318  nSG_to_nG, eSG_to_eG, nG_to_nSG, eG_to_eSG);
319  else if (!nodeVisited[eG->target()])
320  recursion(SG, nodeVisited, edgeVisited, eG->target(),
321  nodeLengthG, nodeLengthSG, edgeLengthG, edgeLengthSG,
322  nSG_to_nG, eSG_to_eG, nG_to_nSG, eG_to_eSG);
323  if (!edgeVisited[eG])
324  {
325  edge eSG = SG.newEdge(nG_to_nSG[eG->source()], nG_to_nSG[eG->target()]);
326  edgeLengthSG[eSG] = edgeLengthG[eG];
327  eG_to_eSG[eG] = eSG;
328  eSG_to_eG[eSG] = eG;
329  edgeVisited[eG] = true;
