109  //Case1: both nodes in same cluster => connect the nodes in the

111  if (uAncestor == vAncestor)

116  else

120  //now only one node can be directly in rootcluster

122  if (uAncestor == CG.rootCluster())

124  m_repEdge[e] = RepGraph[uAncestor]->newEdge(
