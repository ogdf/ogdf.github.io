465  void convertSenseToBound(
466  double inf,
467  const char sense,
468  const double right,
469  const double range,
470  double& lower,
471  double& upper) const
472  {
473  switch (sense) {
474  case 'E':
475  lower = upper = right;
476  break;
477  case 'L':
478  lower = -inf;
479  upper = right;
480  break;
481  case 'G':
482  lower = right;
483  upper = inf;
484  break;
485  case 'R':
486  lower = right - range;
487  upper = right;
488  break;
489  case 'N':
490  lower = -inf;
491  upper = inf;
492  break;
493  }
494  }
