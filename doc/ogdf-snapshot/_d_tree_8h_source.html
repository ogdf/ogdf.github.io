233 template<typename IntType, int Dim>
234 void DTree<IntType, Dim>::prepareNodeLayer()
235 {
236  Node* leafLayer = m_nodes;
237  Node* innerLayer = m_nodes + m_numPoints;
238 
239  //int last = 0;
240  for (int i = 0; i<m_numPoints; )
241  {
242  Node& leaf = leafLayer[i];
243  Node& innerNode = innerLayer[i];
244  // i represents the current node on both layers
245  int j = i+1;
246 
247  // find the next morton number that differs or stop when j is equal to m_numPoints
248  while ((j < m_numPoints) &&
249  (m_mortonOrder[i] == m_mortonOrder[j]))
250  j++;
251  // j is the index of the next cell (node)
252 
253  // init the node on the leaf layer
254  leaf.firstPoint = i; //< node sits above the first point of the cell
255  leaf.numPoints = j-i; //< number of points with equal morton numbers (number of points in grid cell)
256  leaf.numChilds = 0; //< it's a leaf
257  leaf.level = 0; //< it's a leaf
258  leaf.next = j; //< this leaf hasnt been created yet but we use indices so its ok
259 
260  if (j < m_numPoints )
261  {
262  // Note: the n-th inner node is not needed because we only need n-1 inner nodes to cover n leaves
263  // if we reach the n-1-th inner node, the variable last is set for the last time
264  //last = i;
265  // init the node on the inner node layer
266  innerNode.child[0] = i; //< node sits above the first leaf
267  innerNode.child[1] = j; //< this leaf hasnt been created yet but we use indices so its ok
268  innerNode.numChilds = 2; //< every inner node covers two leaves in the beginning
269  innerNode.level = lowestCommonAncestorLevel<IntType, Dim>(m_mortonOrder[i].mortonNr, m_mortonOrder[j].mortonNr);
270  innerNode.next = m_numPoints + j; // the inner node layer is shifted by n
271  } else {
272  // no next for the last
273  innerLayer[i].next = 0;
274  // this is important to make the recursion stop
275  innerLayer[i].level = m_maxLevel + 1;
276  }
277 
278  // advance to the next cell
279  i = j;
280  };
281  // here we set the successor of the n-1-th inner node to zero to avoid dealing with the n-th inner node
282  //innerLayer[last].next = 0;
283 };

314 template<typename IntType, int Dim>
315 int DTree<IntType, Dim>::linkNodes(int curr, int maxLevel)
316 {
317  // while the subtree is smaller than maxLevel
318  while (node(curr).next && (node(node(curr).next).level < maxLevel))
319  {
320  // get next node in the chain
321  int next = node(curr).next;
322  // First case: same level => merge, discard next
323  if (node(curr).level == node(next).level)
324  {
325  mergeWithNext(curr);
326  } else // Second case: next is higher => become first child
327  if (node(curr).level < node(next).level)
328  {
329  // set the first child of next to the current node
330  node(next).child[0] = curr;
331 
332  // adjust the point info of the curr
333  adjustPointInfo(curr);
334 
335  // this is the only case where we advance curr
336  curr = next;
337  } else // Third case: next is smaller => construct a maximal subtree starting with next
338  {
339  int r = linkNodes(next, node(curr).level);
340  node(curr).child[node(curr).numChilds-1] = r;
341  node(curr).next = node(r).next;
342  };
343  };
344  // adjust the point info of the curr
345  adjustPointInfo(curr);
346 
347  // we are done with this subtree, return the root
348  return curr;
349 };
