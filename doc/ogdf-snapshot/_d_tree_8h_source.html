227 template<typename IntType, int Dim>
228 void DTree<IntType, Dim>::prepareNodeLayer()
229 {
230  Node* leafLayer = m_nodes;
231  Node* innerLayer = m_nodes + m_numPoints;
232 
233 #if 0
234  int last = 0;
235 #endif
236  for (int i = 0; i<m_numPoints;) {
237  Node& leaf = leafLayer[i];
238  Node& innerNode = innerLayer[i];
239  // i represents the current node on both layers
240  int j = i+1;
241 
242  // find the next morton number that differs or stop when j is equal to m_numPoints
243  while ((j < m_numPoints) &&
244  (m_mortonOrder[i] == m_mortonOrder[j]))
245  j++;
246  // j is the index of the next cell (node)
247 
248  // init the node on the leaf layer
249  leaf.firstPoint = i; //< node sits above the first point of the cell
250  leaf.numPoints = j-i; //< number of points with equal morton numbers (number of points in grid cell)
251  leaf.numChilds = 0; //< it's a leaf
252  leaf.level = 0; //< it's a leaf
253  leaf.next = j; //< this leaf hasnt been created yet but we use indices so its ok
254 
255  if (j < m_numPoints) {
256  // Note: the n-th inner node is not needed because we only need n-1 inner nodes to cover n leaves
257  // if we reach the n-1-th inner node, the variable last is set for the last time
258 #if 0
259  last = i;
260 #endif
261  // init the node on the inner node layer
262  innerNode.child[0] = i; //< node sits above the first leaf
263  innerNode.child[1] = j; //< this leaf hasnt been created yet but we use indices so its ok
264  innerNode.numChilds = 2; //< every inner node covers two leaves in the beginning
265  innerNode.level = lowestCommonAncestorLevel<IntType, Dim>(m_mortonOrder[i].mortonNr, m_mortonOrder[j].mortonNr);
266  innerNode.next = m_numPoints + j; // the inner node layer is shifted by n
267  } else {
268  // no next for the last
269  innerLayer[i].next = 0;
270  // this is important to make the recursion stop
271  innerLayer[i].level = m_maxLevel + 1;
272  }
273 
274  // advance to the next cell
275  i = j;
276  };
277  // here we set the successor of the n-1-th inner node to zero to avoid dealing with the n-th inner node
278 #if 0
279  innerLayer[last].next = 0;
280 #endif
281 };

312 template<typename IntType, int Dim>
313 int DTree<IntType, Dim>::linkNodes(int curr, int maxLevel)
314 {
315  // while the subtree is smaller than maxLevel
316  while (node(curr).next && (node(node(curr).next).level < maxLevel)) {
317  // get next node in the chain
318  int next = node(curr).next;
319  // First case: same level => merge, discard next
320  if (node(curr).level == node(next).level) {
321  mergeWithNext(curr);
322  } else // Second case: next is higher => become first child
323  if (node(curr).level < node(next).level) {
324  // set the first child of next to the current node
325  node(next).child[0] = curr;
326 
327  // adjust the point info of the curr
328  adjustPointInfo(curr);
329 
330  // this is the only case where we advance curr
331  curr = next;
332  } else { // Third case: next is smaller => construct a maximal subtree starting with next
333  int r = linkNodes(next, node(curr).level);
334  node(curr).child[node(curr).numChilds-1] = r;
335  node(curr).next = node(r).next;
336  };
337  };
338  // adjust the point info of the curr
339  adjustPointInfo(curr);
340 
341  // we are done with this subtree, return the root
342  return curr;
343 };
