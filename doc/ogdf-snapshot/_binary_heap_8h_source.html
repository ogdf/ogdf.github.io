315 template <typename T, typename C>
316 void BinaryHeap<T, C>::siftUp(int pos)
317 {
318  OGDF_ASSERT(pos > 0);
319  OGDF_ASSERT(pos <= m_size);
320 
321  if (pos == 1) {
322  *(m_heapArray[1].handle) = 1;
323  } else {
324  HeapEntry tempEntry = m_heapArray[pos];
325  int run = pos;
326 
327  while ((parentIndex(run) >= 1) &&
328  this->comparator()(tempEntry.value, m_heapArray[parentIndex(run)].value)) {
329  m_heapArray[run] = m_heapArray[parentIndex(run)];
330  *(m_heapArray[run].handle) = run;
331 
332  run = parentIndex(run);
333  }
334 
335  m_heapArray[run] = tempEntry;
336  *(m_heapArray[run].handle) = run;

340 template <typename T, typename C>
341 void BinaryHeap<T, C>::siftDown(int pos)
342 {
343  OGDF_ASSERT(pos > 0);
344  OGDF_ASSERT(pos <= m_size);
345 
346  // is leaf?
347  if (pos >= m_size/2 + 1) {
348  *(m_heapArray[pos].handle) = pos;
349  // leafs cant move further down
350  } else {
351  T value = m_heapArray[pos].value;
352  int sIndex = pos;
353  C const &compare = this->comparator();
354 
355  // left child is smallest child?
356  if (hasLeft(pos) && compare(m_heapArray[leftChildIndex(pos)].value, value)
357  && compare(m_heapArray[leftChildIndex(pos)].value, m_heapArray[rightChildIndex(pos)].value)) {
358  sIndex = leftChildIndex(pos);
359  // or is right child smaller?
360  } else if (hasRight(pos) && compare(m_heapArray[rightChildIndex(pos)].value, value)) {
361  sIndex = rightChildIndex(pos);
362  }
363 
364  // need recursive sift?
365  if (sIndex != pos) {
366  HeapEntry tempEntry = m_heapArray[pos];
367  m_heapArray[pos] = m_heapArray[sIndex];
368  m_heapArray[sIndex] = tempEntry;
369 
370  // update both index entries
371  *(m_heapArray[pos].handle) = pos;
372  *(m_heapArray[sIndex].handle) = sIndex;
373 
374  siftDown(sIndex);
375  } else {
376  *(m_heapArray[pos].handle) = pos;
377  }
378  }
