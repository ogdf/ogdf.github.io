311 template <typename T, typename C>
312 void BinaryHeap<T, C>::siftUp(int pos)
313 {
314  OGDF_ASSERT(pos > 0);
315  OGDF_ASSERT(pos <= m_size);
316 
317  if (pos == 1) {
318  *(m_heapArray[1].handle) = 1;
319  } else {
320  HeapEntry tempEntry = m_heapArray[pos];
321  int run = pos;
322 
323  while ((parentIndex(run) >= 1) &&
324  this->m_comp(tempEntry.value, m_heapArray[parentIndex(run)].value)) {
325  m_heapArray[run] = m_heapArray[parentIndex(run)];
326  *(m_heapArray[run].handle) = run;
327 
328  run = parentIndex(run);
329  }
330 
331  m_heapArray[run] = tempEntry;
332  *(m_heapArray[run].handle) = run;
333  }

336 template <typename T, typename C>
337 void BinaryHeap<T, C>::siftDown(int pos)
338 {
339  OGDF_ASSERT(pos > 0);
340  OGDF_ASSERT(pos <= m_size);
341 
342  // is leaf?
343  if (pos >= m_size/2 + 1) {
344  *(m_heapArray[pos].handle) = pos;
345  // leafs cant move further down
346  } else {
347  T value = m_heapArray[pos].value;
348  int sIndex = pos;
349 
350  // left child is smallest child?
351  if (hasLeft(pos) && this->m_comp(m_heapArray[leftChildIndex(pos)].value, value)
352  && this->m_comp(m_heapArray[leftChildIndex(pos)].value, m_heapArray[rightChildIndex(pos)].value)) {
353  sIndex = leftChildIndex(pos);
354  // or is right child smaller?
355  } else if (hasRight(pos) && this->m_comp(m_heapArray[rightChildIndex(pos)].value, value)) {
356  sIndex = rightChildIndex(pos);
357  }
358 
359  // need recursive sift?
360  if (sIndex != pos) {
361  HeapEntry tempEntry = m_heapArray[pos];
362  m_heapArray[pos] = m_heapArray[sIndex];
363  m_heapArray[sIndex] = tempEntry;
364 
365  // update both index entries
366  *(m_heapArray[pos].handle) = pos;
367  *(m_heapArray[sIndex].handle) = sIndex;
368 
369  siftDown(sIndex);
370  } else {
371  *(m_heapArray[pos].handle) = pos;
372  }
373  }
