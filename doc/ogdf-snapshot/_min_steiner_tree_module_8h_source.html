165 T MinSteinerTreeModule<T>::call(const EdgeWeightedGraph<T> &G,
166  const List<node> &terminals,
167  const NodeArray<bool> &isTerminal,
168  EdgeWeightedGraphCopy<T> *&finalSteinerTree
169  )
170 {
171  if (terminals.size() > 2) {
172  return this->computeSteinerTree(G, terminals, isTerminal, finalSteinerTree);
173  }
174 
175  finalSteinerTree = new EdgeWeightedGraphCopy<T>();
176  finalSteinerTree->createEmpty(G);
177  switch (terminals.size()) {
178  case 1:
179  finalSteinerTree->newNode(terminals.front());
180  case 0:
181  return 0;
182  case 2:
183  T cost(0);
184  AStarSearch<T> astar;
185  NodeArray<edge> pred;
186  NodeArray<T> dist;
187  astar.call(G, G.edgeWeights(), terminals.front(), terminals.back(), pred);
188  if(pred[terminals.back()] != nullptr) {
189  finalSteinerTree->newNode(terminals.back());
190  for (node t = terminals.back(); t != terminals.front(); t = pred[t]->opposite(t)) {
191  const edge e = pred[t];
192  finalSteinerTree->newNode(e->opposite(t));
193  finalSteinerTree->newEdge(e, G.weight(e));
194  cost += G.weight(e);
195  }
196  return cost;
197  }
198  }
199  return -1; // unreachable

444 {
445  // initialization
446  for (node u = G.firstNode(); u; u = u->succ()) {
447  distance[u].init(G, numeric_limits<T>::max());
448  pred[u].init(G, nullptr);
449  }

451  const node u = e->source(), v = e->target();
452  distance[u][v] = G.weight(e);

468  pred[u][w] = pred[v][w];
469  pred[w][u] = pred[v][u];

483  pred[u].init(G, nullptr);

488  pred[u][v] = pred[v][u] = e;

493  if (isTerminal[v]) { // v is a terminal
494  for (node u : G.nodes) {

496  if (duv < numeric_limits<T>::max()) {
497  for (node w = u->succ(); w; w = w->succ()) {
498  const T dvw = distance[v][w];
499  const T duvw = duv + dvw;
500  if (dvw < numeric_limits<T>::max()
501  && duvw <= distance[u][w]) { // prefer terminals
502  distance[w][u] = distance[u][w] = duvw;
503  pred[w][u] = pred[u][w] = nullptr;
504  }
505  }
506  }
507  }
508  } else { // v is not a terminal
509  for (node u : G.nodes) {
510  const T duv = distance[u][v];
511  if (duv < numeric_limits<T>::max()) {
512  for (node w = u->succ(); w; w = w->succ()) {
513  const T dvw = distance[v][w];
514  const T duvw = duv + dvw;
515  if (dvw < numeric_limits<T>::max()
516  && duvw < distance[u][w]) { // do not prefer nonterminals
517  distance[w][u] = distance[u][w] = duvw;
518  pred[u][w] = (pred[u][v] ? pred[v][w] : nullptr);
