237  void preprocess(NodeArray<bool> &isNewTerminal)
238  {
239  Graph H; // a graph where each component is a star
240  NodeArray<int> id(H); // ids each center of the star to the component id
241  NodeArray<node> copy(m_G, nullptr); // ids orig in m_G -> copy in H
242 
243  List<node> centers; // all centers
244  for (int i = 0; i < m_fullCompStore.size(); ++i) {
245  const node center = H.newNode();
246  centers.pushBack(center);
247  id[center] = i;
248 
249  for (node vG : m_fullCompStore.terminals(i)) {
250  node vH = copy[vG];
251  if (!vH) {
252  vH = H.newNode();
253  copy[vG] = vH;
254  }
255  H.newEdge(vH, center); // target is always center
256  }
257  }
258 
259  // find components to be inserted into the steinerTree and insert them
260  ArrayBuffer<int> inactive; // ids of components we insert; we have to remove them from the set of active components afterwards
261  bool changed;
262  do {
263  changed = false;
264  ListIterator<node> it2;
265  for (ListIterator<node> it = centers.begin(); it.valid(); it = it2) {
266  it2 = it.succ();
267  node c = *it;
268  int innerNodes = 0; // count inner nodes
269  for (adjEntry adj : c->adjEntries) {
270  innerNodes += (adj->twinNode()->degree() != 1);
271  }
272  if (innerNodes <= 1) { // this center represents a component to add to steinerTree
273  // insert component into steinerTree
274  addComponent(isNewTerminal, id[c]);
275 
276  // remove center from H (adjacent leaves can remain being isolated nodes)
277  inactive.push(id[c]);
278  H.delNode(c);
279  centers.del(it);
280 
281  changed = true;
282  }
283  }
284  } while (changed);
285 
286  removeComponents(inactive);
287  }

388 template<typename T>
389 void MinSteinerTreeGoemans139<T>::Main::findFullComponents()
390 {
391  if (m_restricted >= 4) { // use Dreyfus-Wagner based full component generation
392  SubsetEnumerator<node> terminalSubset(m_terminals);
393  steiner_tree::FullComponentGeneratorDreyfusWagner<T> fcg(m_G, m_terminals, m_distance);
394  fcg.call(m_restricted);
395  for (terminalSubset.begin(2, m_restricted); terminalSubset.valid(); terminalSubset.next()) {
396  EdgeWeightedGraphCopy<T> component;
397  List<node> terminals;
398  terminalSubset.list(terminals);
399  fcg.getSteinerTreeFor(terminals, component);
400  if (steiner_tree::FullComponentGeneratorDreyfusWagner<T>::isValidComponent(component, m_predAPSP, m_isTerminal)) {
401  m_fullCompStore.insert(component);
402  }
403  }
404  } else {
405  findFull2Components();
406  if (m_restricted == 3) {
407  findFull3Components();
408  }
409  }

413 T
414 MinSteinerTreeGoemans139<T>::Main::getApproximation(EdgeWeightedGraphCopy<T> *&finalSteinerTree, const std::minstd_rand &rng, const bool doPreprocessing)
415 {
416  if (m_use2approx == Approx2State::JustUseIt) {
417  // no remaining components
418  finalSteinerTree = m_approx2SteinerTree;
419  return m_approx2Weight;
420  }
421 
422  removeInactiveComponents();
423 
424  NodeArray<bool> isNewTerminal(m_G, false);
425  for (node v : m_terminals) {
426  isNewTerminal[v] = true;
427  }
428 
429  if (doPreprocessing) {
430  preprocess(isNewTerminal);
431  }
432 
433  if (!m_fullCompStore.isEmpty()) {
434  steiner_tree::goemans::Approximation<T> approx(m_G, m_terminals, m_isTerminal, m_fullCompStore, rng, m_eps);
435  approx.solve(isNewTerminal);
436  }
437 
438  T cost = steiner_tree::obtainFinalSteinerTree(m_G, isNewTerminal, m_isTerminal, finalSteinerTree);
439  if (m_use2approx != Approx2State::Off) {
440  if (m_approx2Weight < cost) {
441  delete finalSteinerTree;
442  finalSteinerTree = m_approx2SteinerTree;
443  } else {
444  delete m_approx2SteinerTree;
445  }
446  }
447 
448  return cost;
449 }
