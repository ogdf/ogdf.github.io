206  /* witness set (for some component and specified K (core edge set))
207  *
208  * W(e) = { e } if e is a core edge
209  * W(e) =
210  * take the path P of loss edges from u to e in the component
211  * add every incident core edge of u
212  * that is: if we root the loss connected component at the terminal,
213  * W(e) = all core edges incident to the subtree below e
214  *
215  * Need fast lookups for:
216  * (1) |W(f)|
217  * (2) all loss edges f such that given core edge e is in W(f)
218  * (3) all loss edges f such that W(f) is a subset of a given basis -> for each e \in B: do (2)
219  *
220  * Data Structures:
221  * - EdgeArray<int> witnessCard with witnessCard[e] = |W(e)|

223  * (Note that core edges are given as nodes.)

247  // We virtually contract all terminals in the union-find data structure.
248  NodeArray<int> setID(G, -1);
249  isInTree.init(G, false);
250  DisjointSets<> uf(G.numberOfNodes() - terminals.size() + 1);
251 
252  int contractedID = uf.makeSet();
253  OGDF_ASSERT(contractedID >= 0);
254  for (node t : terminals) {
255  setID[t] = contractedID;
256  }
257  for (node v = G.firstNode(); v; v = v->succ()) {

918  int findComponentAndMaxBasis(List<std::pair<node,int>> *&maxBasis, EdgeWeightedGraphCopy<T> &blowupGraph, const List<node> &bgTerminals, const CoreWitness &cw, EdgeArray<int> &capacity, const BlowupComponents &gamma, const int N, const int y_R, const node source, const node target)
919  {
920  // there should always be saturated flow to the component roots
921  // (contracted matroid)
922  EdgeArray<int> lB(blowupGraph, 0);
923  for (adjEntry adj : source->adjEntries) {
924  const edge e = adj->theEdge();
925  lB[e] = capacity[e];
926  }
927 
928  // compute weights of core edges and add source->core edges
929  List<edge> sourceCoreEdges;
930  EdgeArray<double> cost(blowupGraph, 0);
931  for (ListConstIterator<node> it = cw.core().rbegin(); it.valid(); --it) { // XXX: why backwards?
932  // compute weight of core edge
933  const node v = *it;
934  edge tmp = v->firstAdj()->theEdge();
935  double weight = (double)blowupGraph.weight(tmp);
936  for (edge e : cw.witnessList(v)) {
937  OGDF_ASSERT(cw.numberOfWitnesses(e) > 0);
938  weight += (double)blowupGraph.weight(e) / cw.numberOfWitnesses(e);
939  }
940 
941  // add edges from source to core edges v
942  edge e = blowupGraph.newEdge(source, v, 0);
943  capacity[e] = capacity[tmp];
944  cost[e] = -weight;
945  sourceCoreEdges.pushBack(e);
946  }
947 
948  maxBasis = nullptr;
949  NodeArray<int> supply(blowupGraph, 0);
950  EdgeArray<int> flow(blowupGraph);
951  MinCostFlowReinelt<double> mcf;
952 
953  for (int id = 1; id <= gamma.size(); ++id) {
954  /* We want to find maximum-weight basis B \in B^K_Q
955  * B_Q = minimal edge set to remove after contracting Q to obtain feasible solution (blowup graph)
956  * = bases of gammoid M_Q
957  * B^K_Q = { B \in B_Q | B \subseteq K } [K is the set of core edges]
958  * = bases of gammoid M^K_Q
959  * M^K_Q is gammoid "obtained by restricting M_Q to K"
960  * M_Q = M'_Q / X' (M'_Q contracted by X') is a gammoid
961  * M'_Q = gammoid from X \cup X' to Y in D' with arc-disjointness instead of vertex-disjointness
962  * D' is D like for separation but without source node and with interim node v_e for each edge e
963  * X = inserted nodes v_e in each edge in blowup graph
964  * X' = the (arbitrary) roots of each component
965  * Y = Q \cup {t}
966  * that means:
967  * I (subset of X) is an independent set of M_Q
968  * <=> (if X' is always an independent set in M'_Q ... which we assume here)
969  * I \cup X' is an independent set of M'_Q
970  * Restricting to K means:
971  * only put v_e in X with e \in K (that is, we only need to *generate* v_e if e \in K)
972  * Hence:
973  * - we generate D'^K (this is blowupGraph)
974  * - compute the max flow from X^K \cup X' to Q \cup {t}
975  * - ASSERT that X' is "saturated"!
976  * - check which subset of X is saturated -> these are the nodes representing the edge set we need
977  */
978  // add edges from component's terminals to target
979  List<edge> Q_to_target;
980  for (node t : gamma.terminals(id)) {
981  const edge e = blowupGraph.newEdge(t, target, 0);
982  capacity[e] = N * y_R; // upper bound for capacity
983  Q_to_target.pushBack(e);
984  }
985 
986  List<std::pair<node,int>> *basis = new List<std::pair<node,int>>;
987 
988  int rank = gammoidGetRank(blowupGraph, capacity, source, target);
989  supply[source] = rank;
990  supply[target] = -rank;
991 
992  // find maximum weight basis
993 #ifdef OGDF_DEBUG
994  bool feasible = mcf.call(blowupGraph, lB, capacity, cost, supply, flow);
995  OGDF_ASSERT(feasible);
996  double minCostVal;
997  OGDF_ASSERT(mcf.checkComputedFlow(blowupGraph, lB, capacity, cost, supply, flow, minCostVal));
998 #else
999  mcf.call(blowupGraph, lB, capacity, cost, supply, flow);
1000 #endif
1001  double weight(0);
1002  for (edge e : sourceCoreEdges) {
1003  if (flow[e] > 0) {
1004  basis->pushBack(std::pair<node,int>(e->target(), flow[e]));
1005  weight -= flow[e] * cost[e];
1006  }
1007  }
1008 
1009  // remove temporary edges for multi-target max-flow again
1010  for (edge e : Q_to_target) {
1011  blowupGraph.delEdge(e);
1012  }
1013  // XXX/TODO: we could also keep target edges from all terminals to the target
1014  // and just change the capacity
1015 
1016  // we choose the component with max cost*N <= weight
1017  if (gamma.cost(id) * N <= weight + m_eps) {
1018  maxBasis = basis;
1019  for (edge e : sourceCoreEdges) { // clean up (XXX: check if necessary)
1020  blowupGraph.delEdge(e);
1021  }
1022  return id;
1023  }
1024  delete basis;
1025  }
1026  return 0; // no component chosen, fail
1027  }

1154  // and, while at it, remove cleanup edges from blowup graph (X := X - F)
