218  //delta values, due to placement problems, cut to box_size / 2
219  void set_delta(OrthoDir bside, OrthoDir bneighbour, int dval) {
220  int bsidei = static_cast<int>(bside);
221  int bneighbouri = static_cast<int>(bneighbour);
222  switch (bside)
223  {
224  case OrthoDir::North:
225  case OrthoDir::South:
226  if (dval > box_y_size) {
227  dval = int(floor(((double)box_y_size / 2))) - m_eps[bsidei][bneighbouri];
228  } break;
229  case OrthoDir::East:
230  case OrthoDir::West:
231  if (dval > box_x_size) {
232  dval = int(floor(((double)box_x_size / 2))) - m_eps[bsidei][bneighbouri];
233  } break;
234  default:
235  OGDF_ASSERT(false);
236  }
237  m_delta[bsidei][bneighbouri] = dval;
238  }
