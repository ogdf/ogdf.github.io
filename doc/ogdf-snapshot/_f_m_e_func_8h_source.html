750  const NodeAdjInfo& b_info = nodeInfo[e_info.b];

753  float d_y = y[e_info.a] - y[e_info.b];
754  float d_sq = d_x*d_x + d_y*d_y;
755

760

769  fa += (nodeSize[e_info.b] / d_sq);
770  fb += (nodeSize[e_info.a] / d_sq);
771  }
