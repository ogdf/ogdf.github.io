449 template<typename T>
450 edge MinSteinerTreeShore<T>::determineBranchingEdge(T prevCost) const
451 {
452  edge result = nullptr;
453  T maxPenalty = -1;
454 
455  // calculate penalties for nodes
456  T sumOfMinWeights = 0; // b
457  T sumOfMinTermWeights = 0; // c
458  T absoluteMinTermWeight = MAX_WEIGHT;
459  for(ListConstIterator<node> it = m_terminals.begin();
460  sumOfMinWeights < MAX_WEIGHT && it.valid();
461  ++it) {
462  const node t = *it;
463  T minTermWeight = MAX_WEIGHT,
464  minWeight = MAX_WEIGHT,
465  secondMinWeight = MAX_WEIGHT;
466  edge minEdge = nullptr;
467 
468  // investigate all edges of each terminal
469  // calculate lower boundary and find branching edge
470  for(adjEntry adj = t->firstAdj(); adj; adj = adj->succ()) {
471  edge e = adj->theEdge();
472  if(weightOf(e) < minWeight) {
473  secondMinWeight = minWeight;
474  minWeight = weightOf(e);
475  minEdge = e;
476  }
477  else {
478  if(weightOf(e) < secondMinWeight) {
479  secondMinWeight = weightOf(e);
480  }
481  }
482 
483  if(isTerminal(adj->twinNode()) && weightOf(e) < minTermWeight) {
484  minTermWeight = weightOf(e);
485  if(minTermWeight < absoluteMinTermWeight) {
486  absoluteMinTermWeight = minTermWeight;
487  }
488  }
489  }
490 
491  if(sumOfMinTermWeights < MAX_WEIGHT && minTermWeight < MAX_WEIGHT) {
492  sumOfMinTermWeights += minTermWeight;
493  }
494  else {
495  sumOfMinTermWeights = MAX_WEIGHT;
496  }
497  OGDF_ASSERT(absoluteMinTermWeight <= sumOfMinTermWeights);
498 
499  // is terminal isolated or has only one edge?
500  // if so we can break here
501  if(minWeight == MAX_WEIGHT ||
502  secondMinWeight == MAX_WEIGHT) {
503  result = minEdge;
504  if(minWeight == MAX_WEIGHT) {
505  sumOfMinWeights = MAX_WEIGHT;
506  }
507  } else {
508  sumOfMinWeights += minWeight;
509  // update branching edge if need be
510  T penalty = secondMinWeight - minWeight;
511  if(penalty > maxPenalty) {
512  maxPenalty = penalty;
513  result = minEdge;
514  }
515  }
516  }
517 
518  // compare bounds for this graph
519  if(result != nullptr) {
520  T maxCost = m_upperBound - prevCost;
521  if(sumOfMinTermWeights < MAX_WEIGHT) {
522  sumOfMinTermWeights -= absoluteMinTermWeight;
523  }
524  if(maxCost <= sumOfMinWeights && maxCost <= sumOfMinTermWeights) {
525  result = nullptr;
526  }
527  }
528 
529  return result;
530 }

532 template<typename T>
533 T MinSteinerTreeShore<T>::bnbInternal(T prevCost, List<edge> &currentEdges)
534 {
535  T result = MAX_WEIGHT;
536  m_recursionDepth++;
537 
538 #ifdef ENABLE_SHORE_LOGGING
539  printSVG();
540 #endif
541 
542  if(prevCost <= m_upperBound) {
543  if(m_terminals.size() < 2) {
544  // update currently chosen edges
545  if(prevCost != m_upperBound || m_chosenEdges.empty()) {
546 
547  m_chosenEdges = List<edge>(currentEdges);
548  }
549  // all terminals are connected
550  m_upperBound = prevCost;
551  result = prevCost;
552  }
553  else {
554  edge branchingEdge = determineBranchingEdge(prevCost);
555  T branchingEdgeWeight = weightOf(branchingEdge);
556 
557 #ifdef ENABLE_SHORE_LOGGING
558  for(int i = 0; i < m_recursionDepth; i++) cout << " ";
559  cout << "branching on edge: " << branchingEdge << std::endl;
560 #endif
561  // branching edge has been found or there is no feasible solution
562  if(branchingEdgeWeight < MAX_WEIGHT) {
563  // chose node to remove
564  node nodeToRemove = branchingEdge->source();
565  node targetNode = branchingEdge->target();
566 
567  // This seems to speed up things.
568  if(nodeToRemove->degree() < targetNode->degree()) {
569  nodeToRemove = branchingEdge->target();
570  targetNode = branchingEdge->source();
571  }
572 
573  // remove branching edge
574  edge origBranchingEdge = deleteEdge(branchingEdge);
575 
576  List<node> delEdges, movedEdges;
577  List<edge> origDelEdges;
578 
579  // first branch: Inclusion of the edge
580  // remove source node of edge and calculate new edge weights
581  OGDF_ASSERT(targetNode != nullptr);
582  OGDF_ASSERT(nodeToRemove != nullptr);
583 
584  // remove edges in case of multigraph
585  while(m_graph.searchEdge(targetNode, nodeToRemove) != nullptr) {
586  edge e = m_graph.searchEdge(targetNode, nodeToRemove);
587  delEdges.pushFront(e->target());
588  delEdges.pushFront(e->source());
589  origDelEdges.pushFront(m_mapping[e]);
590  deleteEdge(e);
591  }
592  while(m_graph.searchEdge(nodeToRemove, targetNode) != nullptr) {
593  edge e = m_graph.searchEdge(targetNode, nodeToRemove);
594  delEdges.pushFront(e->target());
595  delEdges.pushFront(e->source());
596  origDelEdges.pushFront(m_mapping[e]);
597  deleteEdge(e);
598  }
599 
600  OGDF_ASSERT(m_graph.searchEdge(targetNode, nodeToRemove) == nullptr);
601  OGDF_ASSERT(m_graph.searchEdge(nodeToRemove, targetNode) == nullptr);
602 
603  adjEntry adjNext;
604  for(adjEntry adj = nodeToRemove->firstAdj(); adj; adj = adjNext) {
605  adjNext = adj->succ();
606  edge e = adj->theEdge();
607 
608  OGDF_ASSERT(e != branchingEdge);
609  OGDF_ASSERT(e->target() == nodeToRemove || e->source() == nodeToRemove);
610  OGDF_ASSERT(adj->twinNode() != targetNode);
611 
612  edge f = lookupEdge(targetNode, adj->twinNode());
613  bool deletedEdgeE = false;
614  if(f != nullptr) {
615  if(weightOf(f) < weightOf(e)) {
616  delEdges.pushFront(e->target());
617  delEdges.pushFront(e->source());
618  origDelEdges.pushFront(m_mapping[e]);
619  deleteEdge(e);
620 
621  deletedEdgeE = true;
622  }
623  else {
624  delEdges.pushFront(f->target());
625  delEdges.pushFront(f->source());
626  origDelEdges.pushFront(m_mapping[f]);
627  deleteEdge(f);
628  }
629  }
630  if(!deletedEdgeE) {
631  if(e->target() == nodeToRemove) {
632  OGDF_ASSERT(e->source() != targetNode);
633  movedEdges.pushFront(e->source());
634  moveTarget(e, targetNode);
635  }
636  else {
637  OGDF_ASSERT(e->source() == nodeToRemove)
638  OGDF_ASSERT(e->target() != targetNode)
639  movedEdges.pushFront(e->target());
640  moveSource(e, targetNode);
641  }
642  }
643  }
644  // nodeToRemove is isolated at this point
645  // thus no need to actually remove it
646  // (easier to keep track of CopyGraph mapping)
647 
648  // remove node from terminals too
649  bool targetNodeIsTerminal = isTerminal(targetNode),
650  nodeToRemoveIsTerminal = isTerminal(nodeToRemove);
651 
652  OGDF_ASSERT(targetNodeIsTerminal || nodeToRemoveIsTerminal);
653  setTerminal(nodeToRemove, false);
654  setTerminal(targetNode, true);
655 
656 #ifdef ENABLE_SHORE_LOGGING
657  for(int i = 0; i < m_recursionDepth; i++) cout << " ";
658  cout << "inclusion branch" << std::endl;
659 #endif
660  // calculate result on modified graph
661  currentEdges.pushFront(origBranchingEdge);
662  result = bnbInternal(branchingEdgeWeight + prevCost, currentEdges);
663  OGDF_ASSERT(currentEdges.front() == origBranchingEdge);
664  currentEdges.popFront();
665 
666  // restore previous graph
667 
668  // restore terminals
669  setTerminal(nodeToRemove, nodeToRemoveIsTerminal);
670  setTerminal(targetNode, targetNodeIsTerminal);
671 
672  // restore moved edges
673  while(!movedEdges.empty()) {
674  node v = movedEdges.popFrontRet();
675 
676  edge e = lookupEdge(v, targetNode);
677  OGDF_ASSERT(e != nullptr);
678  OGDF_ASSERT(e->opposite(targetNode) != nodeToRemove);
679 
680  if(e->source() == v) {
681  moveTarget(e, nodeToRemove);
682  }
683  else {
684  moveSource(e, nodeToRemove);
685  }
686  }
687 
688  // restore deleted edges
689  while(!delEdges.empty()) {
690  OGDF_ASSERT(!origDelEdges.empty());
691 
692  node source = delEdges.popFrontRet();
693  node target = delEdges.popFrontRet();
694 
695  newEdge(source, target, origDelEdges.popFrontRet());
696  }
697  OGDF_ASSERT(origDelEdges.empty());
698 
699 #ifdef ENABLE_SHORE_LOGGING
700  for(int i = 0; i < m_recursionDepth; i++) cout << " ";
701  cout << "exclusion branch" << std::endl;
702 #endif
703  // sencond branch: Exclusion of the edge
704  T exEdgeResult = bnbInternal(prevCost, currentEdges);
705 
706  // decide which branch returned best result
707  if(exEdgeResult < result) {
708  result = exEdgeResult;
709  }
710 
711  // finally: restore the branching edge
712  newEdge(nodeToRemove, targetNode, origBranchingEdge);
713  }
714  }
715  OGDF_ASSERT(validateMapping());
716  }
717  m_recursionDepth--;
718  return result;
719 }
