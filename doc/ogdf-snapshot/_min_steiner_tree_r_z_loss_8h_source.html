250 {
251  // generate complete graph
252  for (node v : m_terminals) {
253  steinerTree.newNode(v);
254  }
255  for (node u : steinerTree.nodes) {
256  const node uO = steinerTree.original(u);
257  const NodeArray<T> &dist = m_distance[uO];
258  const NodeArray<edge> &pred = m_pred[uO];
259  for (node v = u->succ(); v; v = v->succ()) {
260  const node vO = steinerTree.original(v);
261  if (pred[vO] != nullptr) {
262  steinerTree.newEdge(u, v, dist[vO]);

264  }
265  }
266  // compute MST
267  makeMinimumSpanningTree(steinerTree, steinerTree.edgeWeights());

337 void MinSteinerTreeRZLoss<T>::multiPass(SaveStatic<T> &save, EdgeWeightedGraphCopy<T> &steinerTree, NodeArray<bool> &isNewTerminal)
338 {
339  while (!m_fullCompStore->isEmpty()) {
340  int maxCompId;
341  double r = extractMaxComponent(steinerTree, maxCompId, save);
342  if (r > 1e-9) {

351  m_fullCompStore->remove(maxCompId);
352 
353  if (!m_fullCompStore->isEmpty()) {
354  save.rebuild();
355  }

362 template<typename T>
363 double MinSteinerTreeRZLoss<T>::extractMaxComponent(
364  const EdgeWeightedGraphCopy<T> &steinerTree,
365  int &maxCompId,
366  SaveStatic<T> &save)
367 {
368  maxCompId = -1;
369  double max(0);
370  for (int i = 0; i < m_fullCompStore->size();) {
371  ++m_componentsLookUps;
372  const double winAbs = gain(m_fullCompStore->terminals(i), steinerTree, save) - m_fullCompStore->cost(i);
373  if (winAbs > 1e-9) {
374  const double r = winAbs / m_fullCompStore->loss(i);
375  if (r > max) {
376  max = r;
377  maxCompId = i;
378  }
379  ++i;
380  }
381 #ifdef OGDF_STEINERTREE_RZLOSS_REDUCE_ON
382  else {
383  // reduction
384  m_fullCompStore->remove(i);
385  }
386 #endif
387  }
388  return max;
389 }

398 
399  // extract edges and compute their sum (result value)
400  for (auto it1 = terminals.begin(); it1 != terminals.end(); ++it1) {
401  auto next = it1;
402  ++next;
403  for (auto it2 = next; it2 != terminals.end(); ++it2) {
404  const edge e = save.saveEdge(*it1, *it2);
405  saveEdges.insert(e);

411  }

415 template<typename T>
416 void MinSteinerTreeRZLoss<T>::contractLoss(EdgeWeightedGraphCopy<T> &steinerTree, int compId)
417 {
418  // for every non-loss edge {st} in the component,
419  // where s belongs to the loss component of terminal u
420  // and t belongs to the loss component of terminal v,
421  // we insert edges from u to v in the terminal-spanning tree
422  for (edge e : m_fullCompStore->lossBridges(compId)) {
423  const node u = m_fullCompStore->lossTerminal(e->source());
424  OGDF_ASSERT(u);
425  const node v = m_fullCompStore->lossTerminal(e->target());
426  OGDF_ASSERT(v);
427  steinerTree.newEdge(steinerTree.copy(u), steinerTree.copy(v), m_fullCompStore->graph().weight(e));
428  // parallel edges are ok, will be removed by MST
429  }
430  if (steinerTree.numberOfNodes() != steinerTree.numberOfEdges() + 1) {
431  makeMinimumSpanningTree(steinerTree, steinerTree.edgeWeights());
432  }
433 }
