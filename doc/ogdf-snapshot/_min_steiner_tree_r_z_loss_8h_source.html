226 T MinSteinerTreeRZLoss<T>::computeSteinerTree(const EdgeWeightedGraph<T> &G, const List<node> &terminals, const NodeArray<bool> &isTerminal, EdgeWeightedGraphCopy<T> *&finalSteinerTree)
227 {
228  m_originalGraph = &G;
229  m_restricted = min(m_restricted, terminals.size());
230  m_terminals = terminals; // copy
231  steinertree::sortTerminals(m_terminals); // sort
232  m_isTerminal = &isTerminal;
233  m_fullCompStore = new steinertree::FullComponentWithLossStore<T>(G, m_terminals, isTerminal);
234 
235  for (node u : G.nodes) {
236  if (!isTerminal[u]) {
237  m_nonterminals.pushBack(u);
238  }
239  }
240 
241  m_distance.init(G);
242  m_pred.init(G);
243  if (m_ssspDistances
244  && m_restricted <= 3) {
245  // for 2- and 3-restricted computations, it is ok to use SSSP from all terminals
246  for (node u : terminals) {
247  MinSteinerTreeModule<T>::singleSourceShortestPathsStrict(*m_originalGraph, u, *m_isTerminal, m_distance[u], m_pred[u]);
248  }
249  } else {
250  m_ssspDistances = false;
251  MinSteinerTreeModule<T>::allPairShortestPathsStrict(*m_originalGraph, *m_isTerminal, m_distance, m_pred);
252  }
253 
254  NodeArray<bool> isNewTerminal(G, false);
255  for (node v : terminals) {
256  isNewTerminal[v] = true;
257  }
258 
259  // init terminal-spanning tree and its save-edge data structure
260  EdgeWeightedGraphCopy<T> steinerTree; // the terminal-spanning tree to be modified
261  steinerTree.createEmpty(*m_originalGraph);
262  generateInitialTerminalSpanningTree(steinerTree);
263 
264  SaveStatic<T> save(steinerTree);
265 
266  // component enumeration phase
267  if (m_restricted >= 3) {
268  findFullComponents(steinerTree, save);
269  }
270  m_fullCompStore->computeAllLosses();
271  m_componentsGenerated = m_fullCompStore->size();
272 
273  // contraction phase
274  multiPass(save, steinerTree, isNewTerminal);
275 
276  // cleanup
277  delete m_fullCompStore;
278  m_nonterminals.clear();
279 
280  // obtain final Steiner Tree using (MST-based) Steiner tree approximation algorithm
281  return steinertree::obtainFinalSteinerTree(*m_originalGraph, isNewTerminal, *m_isTerminal, finalSteinerTree);
282 }

285 void MinSteinerTreeRZLoss<T>::generateInitialTerminalSpanningTree(EdgeWeightedGraphCopy<T> &steinerTree)
286 {
287  // generate complete graph
288  for (node v : m_terminals) {
289  steinerTree.newNode(v);
290  }
291  for (node u : steinerTree.nodes) {
292  const node uO = steinerTree.original(u);
293  const NodeArray<T> &dist = m_distance[uO];
294  const NodeArray<edge> &pred = m_pred[uO];
295  for (node v = u->succ(); v; v = v->succ()) {
296  const node vO = steinerTree.original(v);
297  if (pred[vO] != nullptr) {
298  steinerTree.newEdge(u, v, dist[vO]);
299  }
300  }
301  }
302  // compute MST
303  makeMinimumSpanningTree(steinerTree, steinerTree.edgeWeights());
304  OGDF_ASSERT(steinerTree.numberOfNodes() == steinerTree.numberOfEdges() + 1);

307 template<typename T>
308 void MinSteinerTreeRZLoss<T>::findFull3Components(SubsetEnumerator<node> &terminalSubset,
309  const EdgeWeightedGraphCopy<T> &tree,
310  const SaveStatic<T> &save)
311 {
312  // compute special case of 3-components (3-stars)
313  for (terminalSubset.begin(3); terminalSubset.valid(); terminalSubset.next()) {
314  const node t0 = terminalSubset[0];
315  const node t1 = terminalSubset[1];
316  const node t2 = terminalSubset[2];
317  OGDF_ASSERT(m_distance[t0][t1] < numeric_limits<T>::max());
318  OGDF_ASSERT(m_distance[t0][t2] < numeric_limits<T>::max());
319  OGDF_ASSERT(m_distance[t1][t2] < numeric_limits<T>::max());
320  T minCost = // bounded by the sum of the 2-components
321  min(m_distance[t0][t1] + m_distance[t0][t2],
322  min(m_distance[t1][t0] + m_distance[t1][t2],
323  m_distance[t2][t0] + m_distance[t2][t1]));
324  node minCenter = nullptr;
325  for (node center : m_nonterminals) {
326  OGDF_ASSERT(m_distance[t0][center] < numeric_limits<T>::max());
327  OGDF_ASSERT(m_distance[t1][center] < numeric_limits<T>::max());
328  OGDF_ASSERT(m_distance[t2][center] < numeric_limits<T>::max());
329  const T cost = m_distance[t0][center] + m_distance[t1][center] + m_distance[t2][center];
330  if (cost < minCost) {
331  minCenter = center;
332  minCost = cost;
333  }
334  }
335  if (minCenter
336  && m_pred[t0][minCenter]
337  && m_pred[t1][minCenter]
338  && m_pred[t2][minCenter]) {
339  // create a full 3-component
340  EdgeWeightedGraphCopy<T> minComp;
341  minComp.createEmpty(*m_originalGraph);
342  node minCenterC = minComp.newNode(minCenter);
343  minComp.newEdge(minComp.newNode(t0), minCenterC, m_distance[t0][minCenter]);
344  minComp.newEdge(minComp.newNode(t1), minCenterC, m_distance[t1][minCenter]);
345  minComp.newEdge(minComp.newNode(t2), minCenterC, m_distance[t2][minCenter]);
346  OGDF_ASSERT(isTree(minComp));
347 #ifdef OGDF_STEINERTREE_RZLOSS_REDUCE_ON
348  Array<node> terminals;
349  terminalSubset.array(terminals);
350  if (gain(terminals, tree, save) > minCost) { // reduction
351  m_fullCompStore->insert(minComp);
352  }
353 #else
354  m_fullCompStore->insert(minComp);
355 #endif
356  }
357  }

360 template<typename T>
361 void MinSteinerTreeRZLoss<T>::findFullComponents(
362  const EdgeWeightedGraphCopy<T> &tree,
363  const SaveStatic<T> &save)
364 {
365  SubsetEnumerator<node> terminalSubset(m_terminals);
366  if (m_restricted >= 4) { // use Dreyfus-Wagner based full component generation
367  steinertree::DreyfusWagnerFullComponentGenerator<T> fcg(*m_originalGraph, m_terminals, m_distance);
368  fcg.call(m_restricted);
369  for (terminalSubset.begin(3, m_restricted); terminalSubset.valid(); terminalSubset.next()) {
370  EdgeWeightedGraphCopy<T> component;
371  List<node> terminals;
372  terminalSubset.list(terminals);
373 #ifdef OGDF_STEINERTREE_RZLOSS_REDUCE_ON
374  T cost = fcg.getSteinerTreeFor(terminals, component);
375  if (gain(terminals, tree, save) > cost) {
376  if (isValidComponent(component)) {
377  m_fullCompStore->insert(component);
378  }
379  }
380 #else
381  fcg.getSteinerTreeFor(terminals, component);
382  if (isValidComponent(component)) {
383  m_fullCompStore->insert(component);
384  }
385 #endif
386  }
387  } else {
388  findFull3Components(terminalSubset, tree, save);
389  }

393 void MinSteinerTreeRZLoss<T>::multiPass(SaveStatic<T> &save, EdgeWeightedGraphCopy<T> &steinerTree, NodeArray<bool> &isNewTerminal)
394 {
395  while (!m_fullCompStore->isEmpty()) {
396  int maxCompId;
397  double r = extractMaxComponent(steinerTree, maxCompId, save);
398  if (r > 1e-9) {
399  ++m_componentsContracted;
400 
401  // convert nodes of component to terminals
402  m_fullCompStore->foreachNode(maxCompId, m_pred, [&](node v) {
403  isNewTerminal[v] = true;
404  });
405 
406  contractLoss(steinerTree, maxCompId);
407  m_fullCompStore->remove(maxCompId);
408 
409  if (!m_fullCompStore->isEmpty()) {
410  save.rebuild();
411  }
412  } else {
413  return;
414  }

418 template<typename T>
419 double MinSteinerTreeRZLoss<T>::extractMaxComponent(
420  const EdgeWeightedGraphCopy<T> &steinerTree,
421  int &maxCompId,
422  SaveStatic<T> &save)
423 {
424  maxCompId = -1;
425  double max(0);
426  for (int i = 0; i < m_fullCompStore->size();) {
427  ++m_componentsLookUps;
428  const double winAbs = gain(m_fullCompStore->terminals(i), steinerTree, save) - m_fullCompStore->cost(i);
429  if (winAbs > 1e-9) {
430  const double r = winAbs / m_fullCompStore->loss(i);
431  if (r > max) {
432  max = r;
433  maxCompId = i;
434  }
435  ++i;
436  }
437 #ifdef OGDF_STEINERTREE_RZLOSS_REDUCE_ON
438  else {
439  // reduction
440  m_fullCompStore->remove(i);
441  }
442 #endif
443  }
444  return max;
445 }

449 template<typename TERMINAL_CONTAINER>
450 T MinSteinerTreeRZLoss<T>::gain(const TERMINAL_CONTAINER &terminals, const EdgeWeightedGraphCopy<T> &steinerTree, const SaveStatic<T> &save)
451 {
452  std::set<edge> saveEdges;
453  T gain(0);
454 
455  // extract edges and compute their sum (gain value)
456  for (auto it1 = terminals.begin(); it1 != terminals.end(); ++it1) {
457  auto next = it1;
458  ++next;
459  for (auto it2 = next; it2 != terminals.end(); ++it2) {
460  const edge e = save.saveEdge(*it1, *it2);
461  saveEdges.insert(e);
462  }
463  }
464 
465  for (edge e : saveEdges) {
466  gain += steinerTree.weight(e);
467  }
468  return gain;
469 }

471 template<typename T>
472 void MinSteinerTreeRZLoss<T>::contractLoss(EdgeWeightedGraphCopy<T> &steinerTree, int compId)
473 {
474  // for every non-loss edge {st} in the component,
475  // where s belongs to the loss component of terminal u
476  // and t belongs to the loss component of terminal v,
477  // we insert edges from u to v in the terminal-spanning tree
478  for (edge e : m_fullCompStore->lossBridges(compId)) {
479  const node u = m_fullCompStore->lossTerminal(e->source());
480  OGDF_ASSERT(u);
481  const node v = m_fullCompStore->lossTerminal(e->target());
482  OGDF_ASSERT(v);
483  steinerTree.newEdge(steinerTree.copy(u), steinerTree.copy(v), m_fullCompStore->graph().weight(e));
484  // parallel edges are ok, will be removed by MST
485  }
486  if (steinerTree.numberOfNodes() != steinerTree.numberOfEdges() + 1) {
487  makeMinimumSpanningTree(steinerTree, steinerTree.edgeWeights());
488  }
489 }
