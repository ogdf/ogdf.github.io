91  static void insertSorted(List<node> &list, node v)
92  {
93  for (ListIterator<node> it = list.begin(); it.valid(); ++it) {
94  OGDF_ASSERT((*it)->index() != v->index());
95  if (v->index() < (*it)->index()) {

115  void call(int restricted)
116  {
117  SubsetEnumerator<node> terminalSubset(m_terminals);
118  for (terminalSubset.begin(2, restricted-1); terminalSubset.valid(); terminalSubset.next()) {
119  List<node> terminals;
120  terminalSubset.list(terminals);
121  SubsetEnumerator<node> subset(terminals);
122  // populate "split"
123  NodeArray<DWMData> split(m_G);
124  auto computeSplit = [&](node v) {
125  OGDF_ASSERT(!terminalSubset.hasMember(v));
126  DWMData best;
127  for (subset.begin(1, terminals.size()-1); subset.valid(); subset.next()) {
128  List<node> list1, list2;
129  subset.list(list1, list2);
130  insertSorted(list1, v);
131  insertSorted(list2, v);
132  T tmp = costOf(list1) + costOf(list2);
133  if (tmp < best.cost) {
134  best = dataOf(list1);
135  DWMData data = dataOf(list2);
136  best.cost += data.cost;
137  int bsize = best.nodepairs.size();
138  int dsize = data.nodepairs.size();
139  best.nodepairs.grow(dsize);
140  for (int i = 0; i < dsize; ++i) { // copy
141  best.nodepairs[bsize + i] = data.nodepairs[i];
142  }
143  }
144  }
145  split[v] = best;
146  };
147  // compute partial solutions
148  auto computePartialSolutions = [&](node v) {
149  List<node> newTerminals(terminals);
150  insertSorted(newTerminals, v);
151  if (m_map.member(newTerminals)) { // already defined
152  return; // abort
153  }
154  DWMData best;
155  for (node w : m_G.nodes) {
156  T dist = m_distance[v][w];
157  if (v == w) {
158  dist = 0;
159  }
160  if (terminalSubset.hasMember(w)) {
161  // we attach edge vw to tree containing terminal w
162  T tmp;
163  tmp = costOf(terminals) + dist;
164  if (tmp < best.cost) {
165  best = dataOf(terminals);
166  best.cost += dist;
167  best.nodepairs.grow(1, NodePair(v,w));
168  }
169  } else {
170  // we attach edge vw to tree split[w]
171  T tmp;
172  if (split[w].nodepairs.size() == 0) {
173  computeSplit(w);
174  }
175  tmp = split[w].cost + dist;
176  if (tmp < best.cost) {
177  best = split[w];
178  if (v != w) {
179  best.cost += dist;
180  best.nodepairs.grow(1, NodePair(v,w));
181  }
182  }
183  }
184  }
185  m_map.fastInsert(newTerminals, best);
186  };
187  if (terminalSubset.size() != restricted - 1) {
188  for (node v : m_G.nodes) {
189  if (!terminalSubset.hasMember(v)) {
190  computePartialSolutions(v);
191  }
192  }
193  } else { // maximal terminal subset
194  for (node v : m_terminals) { // save time by only adding terminals instead of all nodes
195  if (!terminalSubset.hasMember(v)) {
196  computePartialSolutions(v);
197  }
198  }
199  }
200  }
201  }

204  T getSteinerTreeFor(const List<node> &terminals, EdgeWeightedGraphCopy<T> &tree) const
205  {
206  T cost(0);
207  DWMData data = dataOf(terminals);
208  tree.createEmpty(m_G);
209  for (auto nodepair : data.nodepairs) {
210  node uO = nodepair.source;
211  node vO = nodepair.target;
212  node uC = tree.copy(uO);
213  node vC = tree.copy(vO);
214  if (uC == nullptr) {
215  uC = tree.newNode(uO);
216  }
217  if (vC == nullptr) {
218  vC = tree.newNode(vO);
219  }
220  const T dist = m_distance[uO][vO];
221  tree.newEdge(uC, vC, dist);
222  cost += dist;
223  }
224  OGDF_ASSERT(isTree(tree));
225  return cost;
226  }

233  static bool isValidComponent(const EdgeWeightedGraphCopy<T> &graph, const NodeArray<NodeArray<edge>> &pred, const NodeArray<bool> &isTerminal)
234  {
235  for (edge e : graph.edges) {
236  const node u = graph.original(e->source());
237  const node v = graph.original(e->target());
238  if (pred[u][v] == nullptr) {
239  return false;
240  }
241  }
242  for (node v : graph.nodes) {
243  if (isTerminal[graph.original(v)] // is a terminal
244  && v->degree() > 1) { // but not a leaf
245  return false;
246  }
247  }
248  return true;
249  }
