255  void undoAssociationClass(AssociationClass* ac)
256  {
257  node v = m_associationClassModel[ac->m_edge];
258  OGDF_ASSERT(v);
259  OGDF_ASSERT(v->degree() == 1);
260  if (v->degree() != 1) throw AlgorithmFailureException(AlgorithmFailureCode::Label);
261  //save layout information
262  ac->m_x = x(v);
263  ac->m_y = y(v);
264 
265  //remove node and unsplit edge
266 
267  //run around the dummy node connected to v
268  adjEntry outAdj = v->firstAdj();
269  adjEntry dummyAdj = outAdj->twin();
270 
271  node dummy = dummyAdj->theNode();
272  OGDF_ASSERT(dummy->degree() == 3);
273
