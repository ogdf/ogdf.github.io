513 /*
514 The function [[changeEndmost]] replaces the old endmost child [[oldEnd]] of
515 the node by a new child [[newEnd]].
516 If the node is a $Q$-node, then it must have two valid
517 pointers to its endmost children. If one of the endmost children is [[oldEnd]],
518 it is replaced by [[newEnd]].
519 The function [[changeEndmost]] returns [[1]] if it succeeded in
520 replacing [[oldEnd]] by [[newEnd]]. Otherwise the function returns
521 [[0]], leaving with an error message.
522 */
523 template<class T,class X,class Y>
524 bool PQNode<T,X,Y>::changeEndmost(PQNode<T,X,Y>* oldEnd, PQNode<T,X,Y>* newEnd)
525 {
526  if (m_leftEndmost == oldEnd)
527  {
528  m_leftEndmost = newEnd;
529  return true;
530  }
531  else if (m_rightEndmost == oldEnd)
532  {
533  m_rightEndmost = newEnd;
534  return true;
535  }
536  return false;
537 }

539 /*

541 node by a new sibling [[newSib]].
542 
543 If the node has [[oldSib]] as sibling, then it changes the
544 sibling pointer that references to [[oldSib]] and places [[newSib]]
545 at its position.
546 
547 The function [[changeSiblings]] returns [[1]] if it succeeded in replacing
548 [[oldSib]] by [[newSib]]. Otherwise the function returns
549 [[0]], leaving with an error message.
550 */

563  }
564  return false;
