204  inline node successorOnExternalFace(node w, int& direction) {
205  OGDF_ASSERT(w!=nullptr);
206  OGDF_ASSERT(w->degree()>0);
207  OGDF_ASSERT(m_link[BoyerMyrvoldPlanar::DirectionCW][w]!=nullptr);
208  OGDF_ASSERT(m_link[BoyerMyrvoldPlanar::DirectionCCW][w]!=nullptr);
209  adjEntry adj = m_link[direction][w];
210  OGDF_ASSERT(adj->theNode()!=nullptr);
211 
212  if (w->degree() > 1) direction =
213  adj==beforeShortCircuitEdge(adj->theNode(),BoyerMyrvoldPlanar::DirectionCCW)->twin();
214  OGDF_ASSERT(direction || adj==beforeShortCircuitEdge(adj->theNode(),BoyerMyrvoldPlanar::DirectionCW)->twin());
215  return adj->theNode();
216  }

275  inline bool wNodesExist(node root, node stopx, node stopy) {
276  OGDF_ASSERT(root != stopx);
277  OGDF_ASSERT(root != stopy);
278  OGDF_ASSERT(stopx != stopy);
279  int dir = BoyerMyrvoldPlanar::DirectionCCW;
280  bool between = false;
281  while (root != nullptr) {
282  root = successorOnExternalFace(root,dir);
283  if (between && pertinent(root)) {
284  return true;
285  }
286  if (root == stopx || root == stopy) {
287  if (between) {
288  return false;
289  }
290  between = true;
291  }
292  }
293  return false;
294  }
