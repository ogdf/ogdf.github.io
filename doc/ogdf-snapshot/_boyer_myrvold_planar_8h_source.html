203  inline node successorOnExternalFace(node w, int& direction) {
204  OGDF_ASSERT(w!=nullptr);
205  OGDF_ASSERT(w->degree()>0);
206  OGDF_ASSERT(m_link[CW][w]!=nullptr);
207  OGDF_ASSERT(m_link[CCW][w]!=nullptr);
208  adjEntry adj = m_link[direction][w];
209  OGDF_ASSERT(adj->theNode()!=nullptr);
210 
211  if (w->degree() > 1) direction =
212  adj==beforeShortCircuitEdge(adj->theNode(),CCW)->twin();
213  OGDF_ASSERT(direction || adj==beforeShortCircuitEdge(adj->theNode(),CW)->twin());
214  return adj->theNode();
215  }

274  inline bool wNodesExist(node root, node stopx, node stopy) {
275  OGDF_ASSERT(root != stopx);
276  OGDF_ASSERT(root != stopy);
277  OGDF_ASSERT(stopx != stopy);
278  int dir = CCW;
279  bool between = false;
280  while (root != nullptr) {
281  root = successorOnExternalFace(root,dir);
282  if (between && pertinent(root)) {
283  return true;
284  }
285  if (root == stopx || root == stopy) {
286  if (between) {
287  return false;
288  }
289  between = true;
290  }
291  }
292  return false;
293  }
