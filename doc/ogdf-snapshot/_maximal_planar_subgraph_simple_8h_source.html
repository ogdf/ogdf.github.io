226  const EdgeArray<TCost> *pCost,
227  bool preferredImplyPlanar) override
228  {
229  Module::ReturnType result;
230  delEdges.clear();
231 
232  // scale the costs and do multiple runs (if needed)
233  List<edge> delEdgesCurrentBest;
234 
235  // normalize costs to [0,1] and apply randomness
236  EdgeArray<TCost> normalizedCost(graph);
237 
238  for (auto i=0u; i < m_runs; i++) {
239 
240  List<edge> heuDelEdges;
241 
242  if(pCost == nullptr) {
243  result = m_heuristic.call(graph, preferredEdges, heuDelEdges, preferredImplyPlanar);
244  } else {
245  std::uniform_real_distribution<TCost> distribution (0.0,1.0);
246  TCost maxCost = (*pCost)[graph.firstEdge()];
247  TCost minCost = (*pCost)[graph.firstEdge()];
248  for (edge e: graph.edges) {
249  maxCost = max(maxCost, (*pCost)[e]);
250  minCost = min(minCost, (*pCost)[e]);
251  }
252  for (edge e: graph.edges) {
253  // do not merge with first FOR !
254  // normalized = pCost transformed to [0,1]
255  TCost normalized = ((*pCost)[e] - minCost) / (maxCost - minCost);
256  // now use randomness
257  normalizedCost[e] = (1.0 - m_randomness)*normalized + m_randomness*distribution(m_randomGenerator);
258  }
259  result = m_heuristic.call(graph, normalizedCost, preferredEdges, heuDelEdges, preferredImplyPlanar);
260  }
261 
262  if(Module::isSolution(result)) {
263  GraphCopy copy(graph);
264 
265  if (pCost != nullptr) {
266  MaximalPlanarSubgraphSimpleInternal::EdgeComparer<TCost> cmp(normalizedCost);
267  heuDelEdges.quicksort(cmp);
268  }
269 
270  for (edge e : heuDelEdges) {
271  copy.delEdge(copy.copy(e));
272  }
273 
274  delEdgesCurrentBest.clear();
275  for (edge e : heuDelEdges) {
276  edge f = copy.newEdge(e);
277 
278  if (!isPlanar(copy)) {
279  delEdgesCurrentBest.pushBack(e);
280  copy.delEdge(f);
281  }
282  }
283 
284  if (pCost == nullptr) {
285  if (i == 0 || delEdgesCurrentBest.size() < delEdges.size()) {
286  // better solution: copy to delEdges
287  delEdges.clear();
288  for (auto e: delEdgesCurrentBest) { delEdges.pushBack(e); };
289  }
290  } else {
291  if (i == 0 || weightOfList(delEdgesCurrentBest,normalizedCost) < weightOfList(delEdges,normalizedCost)) {
292  // better solution: copy to delEdges
293  delEdges.clear();
294  for (auto e: delEdgesCurrentBest) { delEdges.pushBack(e); };
295  }
296  }
297  }
298  }
