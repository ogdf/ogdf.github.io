222 template<typename T, typename C>
223 inline PairingHeapNode<T> *PairingHeap<T, C>::pair(
224  PairingHeapNode<T> *node)
225 {
226  if(node == nullptr) {
227  return nullptr;
228  }
229 
230  /*
231  * We move towards the end of list counting elements along the way. We do
232  * this for two reasons: to know whether the list has even or odd number of
233  * elements and for possible speed up of going-back through the list (loop
234  * unrolling is not applicable for iterators but works for integers).
235  */
236  size_t children = 1;
237  PairingHeapNode<T> *it = node;
238  while (it->next != nullptr) {
239  it = it->next;
240  children++;
241  }
242 
243  PairingHeapNode<T> *result;
244 
245  if (children % 2 == 1) {
246  PairingHeapNode<T> *a = it;
247  it = it->prev;
248  a->prev = a->next = nullptr;
249  result = a;
250  } else {
251  PairingHeapNode<T> *a = it, *b = it->prev;
252  it = it->prev->prev;
253  a->prev = a->next = b->prev = b->next = nullptr;
254  result = merge(a, b);
255  }
256 
257  for (size_t i = 0; i < (children - 1) / 2; i++) {
258  PairingHeapNode<T> *a = it, *b = it->prev;
259  it = it->prev->prev;
260  a->prev = a->next = b->prev = b->next = nullptr;
261  result = merge(merge(a, b), result);
262  }
263 
264  return result;
265 }

312 template<typename T, typename C>
313 inline void PairingHeap<T, C>::release(PairingHeapNode<T> *node)
314 {
315  /*
316  * Recursive version of this function is infinitely prettier than that
317  * abomination. Please, make it prettier if you can.
318  */
319  PairingHeapNode<T> *it = node;
320 
321  if(it == nullptr) {
322  return;
323  }
324 
325  for(;;) {
326  // Slide down as long as possible.
327  if(it->child != nullptr) {
328  it = it->child;
329  continue;
330  }
331  if(it->next != nullptr) {
332  it = it->next;
333  continue;
334  }
335 
336  // Climb up until you find first non-visited node.
337  for(;;) {
338  PairingHeapNode<T> *curr = it, *prev = it->prev;
339  delete it;
340 
341  if(prev == nullptr) {
342  return;
343  }
344  if(curr == prev->child && prev->next != nullptr) {
345  it = prev->next;
346  break;
347  } else {
348  it = prev;
349  }
350  }
351  }
