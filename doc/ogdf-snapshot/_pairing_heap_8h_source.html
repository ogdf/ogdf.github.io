218 template<typename T, typename C>
219 inline PairingHeapNode<T> *PairingHeap<T, C>::pair(
220  PairingHeapNode<T> *heapNode)
221 {
222  if(heapNode == nullptr) {
223  return nullptr;
224  }
225 
226  /*
227  * We move towards the end of list counting elements along the way. We do
228  * this for two reasons: to know whether the list has even or odd number of
229  * elements and for possible speed up of going-back through the list (loop
230  * unrolling is not applicable for iterators but works for integers).
231  */
232  size_t children = 1;
233  PairingHeapNode<T> *it = heapNode;
234  while (it->next != nullptr) {
235  it = it->next;
236  children++;
237  }
238 
239  PairingHeapNode<T> *result;
240 
241  if (children % 2 == 1) {
242  PairingHeapNode<T> *a = it;
243  it = it->prev;
244  a->prev = a->next = nullptr;
245  result = a;
246  } else {
247  PairingHeapNode<T> *a = it, *b = it->prev;
248  it = it->prev->prev;
249  a->prev = a->next = b->prev = b->next = nullptr;
250  result = merge(a, b);
251  }
252 
253  for (size_t i = 0; i < (children - 1) / 2; i++) {
254  PairingHeapNode<T> *a = it, *b = it->prev;
255  it = it->prev->prev;
256  a->prev = a->next = b->prev = b->next = nullptr;
257  result = merge(merge(a, b), result);
258  }
259 
260  return result;
261 }

308 template<typename T, typename C>
309 inline void PairingHeap<T, C>::release(PairingHeapNode<T> *heapNode)
310 {
311  /*
312  * Recursive version of this function is infinitely prettier than that
313  * abomination. Please, make it prettier if you can.
314  */
315  PairingHeapNode<T> *it = heapNode;
316 
317  if(it == nullptr) {
318  return;
319  }
320 
321  for(;;) {
322  // Slide down as long as possible.
323  if(it->child != nullptr) {
324  it = it->child;
325  continue;
326  }
327  if(it->next != nullptr) {
328  it = it->next;
329  continue;
330  }
331 
332  // Climb up until you find first non-visited node.
333  for(;;) {
334  PairingHeapNode<T> *curr = it, *prev = it->prev;
335  delete it;
336 
337  if(prev == nullptr) {
338  return;
339  }
340  if(curr == prev->child && prev->next != nullptr) {
341  it = prev->next;
342  break;
343  } else {
344  it = prev;
345  }
346  }
