83  void readFrom(const Graph& G, NodeArray<CoordinateType>& xPos, NodeArray<CoordinateType>& yPos, const EdgeArray<LengthType>& edgeLength, const NodeArray<SizeType>& nodeSize)
84  {
85  m_numNodes = 0;
86  m_numEdges = 0;
87  NodeArray<uint32_t> nodeIndex(G);
88  m_numNodes = 0;
89  m_numEdges = 0;
90  m_desiredAvgEdgeLength = 0;
91  m_avgNodeSize = 0;
92  for(node v : G.nodes)
93  {
94  m_nodeXPos[m_numNodes] = (float)xPos[v];
95  m_nodeYPos[m_numNodes] = (float)yPos[v];
96  m_nodeSize[m_numNodes] = (float)nodeSize[v];
97  m_avgNodeSize += nodeSize[v];
98  nodeIndex[v] = m_numNodes;
99  m_numNodes++;
100  }
101  m_avgNodeSize = m_avgNodeSize / (double)m_numNodes;
102 
103  for(edge e : G.edges)
104  {
105  pushBackEdge(nodeIndex[e->source()], nodeIndex[e->target()], (float)edgeLength[e]);
106  }
107  m_desiredAvgEdgeLength = m_desiredAvgEdgeLength / (double)m_numEdges;
