86  List <edge> &edgeList, edge e_st) {

88  delete m_gc;
89  m_gc = new GraphCopy(graph);
90  CombinatorialEmbedding CE;
91  GraphCopy m_weightedGc;
92  EdgeArray<edge> mapE(m_weightedGc, nullptr);
93 
94  std::function<edge(edge)> orig = [&](edge e) -> edge { return m_gc->original(e);};
95 
96  if(weighted) {
97  m_weightedGc.init(graph);
98  if(e_st != nullptr) {
99  e_st = m_weightedGc.copy(e_st);
100  }
101  s = m_weightedGc.copy(s);
102  t = m_weightedGc.copy(t);
103  List<edge> edges;
104  graph.allEdges(edges);
105  for(edge e : edges) {
106  mapE[m_weightedGc.copy(e)] = e;
107  if(m_weightedGc.copy(e) == e_st) {
108  continue;
109  }
110  OGDF_ASSERT((*weight)[e] >= 1);
111  TCost i = 1;
112  for(; i < (*weight)[e]; i++) {
113  edge copyEdge = m_weightedGc.copy(e);
114  edge newEdge = m_weightedGc.newEdge(copyEdge->source(), copyEdge->target());
115  m_weightedGc.move(newEdge, copyEdge->adjSource(), ogdf::Direction::before, copyEdge->adjTarget(), ogdf::Direction::after);
116  mapE[newEdge] = e;
117  }
118  OGDF_ASSERT((*weight)[e] == i);
119  }
120  // we can't reinit m_gc, because it's possible that the original graph of m_gc
121  // doesn't exist anymore.
122  delete m_gc;
123  m_gc = new GraphCopy();
124  m_gc->init(m_weightedGc);
125  orig = [&](edge e) -> edge { return mapE[m_gc->original(e)];};
126  MinSTCutModule<TCost>::preprocessingDual(m_weightedGc, *m_gc, CE, s, t, e_st);
127  } else {
128  MinSTCutModule<TCost>::preprocessingDual(graph, *m_gc, CE, s, t, e_st);
129  }
130 
131  MinSTCutModule<TCost>::m_direction.init(graph);
132 
133  DualGraph dual(CE);
134  if (e_st != nullptr) {
135  e_st = m_gc->copy(e_st);
136  } else {
137  e_st = m_gc->searchEdge(m_gc->copy(s), m_gc->copy(t));
138  }
139 
140  edgeList.clear();
141  node source = dual.dualNode(CE.rightFace(e_st->adjSource()));
142  node target = dual.dualNode(CE.leftFace(e_st->adjSource()));
143 
144  NodeArray<edge> spPred(dual, nullptr);
145  EdgeArray<node> prev(dual, nullptr);
146  EdgeArray<bool> direction(dual, true);
147  QueuePure<edge> queue;
148  for (adjEntry adj : source->adjEntries) {
149  if (dual.primalEdge(adj->theEdge()) != e_st) {
150  queue.append(adj->theEdge());
151  prev[adj->theEdge()] = source;
152  }
153  }
154  // actual search (using bfs on directed dual)
155  for (;;) {
156  // next candidate edge
157  edge eCand = queue.pop();
158  bool dir = (eCand->source() == prev[eCand]);
159  node v = (dir ? eCand->target() : eCand->source());
160 
161  // leads to an unvisited node?
162  if (spPred[v] == nullptr) {
163  // yes, then we set v's predecessor in search tree
164  spPred[v] = eCand;
165  direction[eCand] = dir;
166 
167  // have we reached t ...
168  if (v == target) {
169  // ... then search is done.
170  // constructed list of used edges (translated to crossed
171  // edges entries in G) from t back to s (including first
172  // and last!)
173 
174  do {
175  edge eDual = spPred[v];
176  OGDF_ASSERT(eDual != nullptr);
177  // this should be the right way round
178  edge eG = dual.primalEdge(eDual);
179  OGDF_ASSERT(eG != nullptr);
180  edgeList.pushBack(orig(eG));
181  MinSTCutModule<TCost>::m_direction[orig(eG)] = !direction[eDual];
182  v = prev[eDual];
183  } while (v != source);
184 
185  break;
186  }
187 
188  // append next candidate edges to queue
189  // (all edges leaving v)
190  for (adjEntry adj : v->adjEntries) {
191  if (prev[adj->theEdge()] == nullptr) {
192  queue.append(adj->theEdge());
193  prev[adj->theEdge()] = v;
194  }
195  }
196  }
197  }
198  if(weighted) {
199  auto prevIt = edgeList.begin();
200  for(auto it = prevIt.succ(); it != edgeList.end(); prevIt = it++) {
201  if(*prevIt == *it) {
202  edgeList.del(prevIt);
203  }
204  }
205  }
206  return true;
207 }
