69 #define OGDF_FME_KERNEL_COMPUTE_FORCE_PROTECTION_FACTOR 0.25f
70 // makro for force computation via SSE s / max(s*0.5, (dx*dx + dy*dy))
71 #define OGDF_FME_KERNEL_MM_COMPUTE_FORCE(dx,dy,s) \
72  _mm_div_ps((s), \
73  _mm_max_ps(_mm_mul_ps((s), \
74  _mm_set1_ps(OGDF_FME_KERNEL_COMPUTE_FORCE_PROTECTION_FACTOR)), \
75  _mm_add_ps(_mm_mul_ps((dx),(dx)), \
76  _mm_mul_ps((dy),(dy)))))
77 #define OGDF_FME_KERNEL_COMPUTE_FORCE(dx,dy,s) \

94 inline void eval_edges(const ArrayGraph& graph, const uint32_t begin, const uint32_t end, float* fx, float* fy)
95 {
96  const float* x = graph.nodeXPos();
97  const float* y = graph.nodeYPos();
98  const float* e = graph.desiredEdgeLength();
99  for (uint32_t i=begin; i <= end; i++)
100  {
101  const auto& e_info = graph.edgeInfo(i);
102  const uint32_t a = e_info.a;
103  const uint32_t b = e_info.b;
104  const auto& a_info = graph.nodeInfo(a);
105  const auto& b_info = graph.nodeInfo(b);
106 
107  float dx = x[a] - x[b];
108  float dy = y[a] - y[b];
109  float dsq = dx*dx + dy*dy;
110  float f = (logf(dsq)*0.5f-logf(e[i])) * 0.25f;
111  float fa = (float)(f/((float)a_info.degree));
112  float fb = (float)(f/((float)b_info.degree));
113  fx[a] -= dx*fa;
114  fy[a] -= dy*fa;
115  fx[b] += dx*fb;
116  fy[b] += dy*fb;
117  }
118 }

122 inline void eval_direct(float* x, float* y, float* s, float* fx, float* fy, size_t n)
123 {
124  for (uint32_t i=0; i < n; i++)
125  {
126  for (uint32_t j=i+1; j < n; j++)
127  {
128  float dx = x[i] - x[j];
129  float dy = y[i] - y[j];
130 #ifdef OGDF_FME_KERNEL_USE_OLD
131  float s_sum = s[i]+s[j];
132 #else
133  float s_sum = s[i]*s[j];
134 #endif
135  float f = OGDF_FME_KERNEL_COMPUTE_FORCE(dx, dy, s_sum);
136  fx[i] += dx*f;
137  fy[i] += dy*f;
138  fx[j] -= dx*f;
139  fy[j] -= dy*f;
140  }
141  }
142 }

234  inline void simpleForceDirected(ArrayGraph& graph, float timeStep, uint32_t minIt, uint32_t maxIt, uint32_t preProcIt, double threshold)
235  {
236  bool earlyExit = false;
237  float* fx = (float*)OGDF_MALLOC_16(sizeof(float)*graph.numNodes());
238  float* fy = (float*)OGDF_MALLOC_16(sizeof(float)*graph.numNodes());
239 
240  for (uint32_t i = 0; i<preProcIt; i++)
241  {
242  for (uint32_t j = 0; j<graph.numNodes(); j++)
243  {
244  fx[j] = 0.0f;
245  fy[j] = 0.0f;
246  }
247  simpleEdgeIteration(graph, fx, fy, timeStep);
248  }
249  for(uint32_t i = 0; (i<maxIt) && (!earlyExit); i++)
250  {
251  for (uint32_t j = 0; j<graph.numNodes(); j++)
252  {
253  fx[j] = 0.0f;
254  fy[j] = 0.0f;
255  }
256  double dsq = simpleIteration(graph, fx, fy, timeStep);
257  if (dsq < threshold && i>(minIt))
258  earlyExit = true;
259  }
