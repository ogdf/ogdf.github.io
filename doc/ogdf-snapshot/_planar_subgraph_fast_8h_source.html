369  static void planarize(const Graph &G, NodeArray<int> &numbering, List<edge> &delEdges) {
370  using PlanarLeafKey = booth_lueker::PlanarLeafKey<whaInfo*>;
371 
372  NodeArray<SListPure<PlanarLeafKey*>> inLeaves(G);
373  NodeArray<SListPure<PlanarLeafKey*>> outLeaves(G);
374  Array<node> table(G.numberOfNodes()+1);
375 
376  for(node v : G.nodes) {
377  for(adjEntry adj : v->adjEntries) {
378  edge e = adj->theEdge();
379  if (numbering[e->opposite(v)] > numbering[v]) { // sideeffect: ignores selfloops
380  PlanarLeafKey* L = new PlanarLeafKey(e);
381  inLeaves[v].pushFront(L);
382  }
383  }
384  table[numbering[v]] = v;
385  }
386 
387  for(node v : G.nodes) {
388  for (PlanarLeafKey *L : inLeaves[v]) {
389  outLeaves[L->userStructKey()->opposite(v)].pushFront(L);
390  }
391  }
392 
393  SList<PQLeafKey<edge,whaInfo*,bool>*> totalEliminatedKeys;
394 
395  PlanarSubgraphPQTree T;
396  T.Initialize(inLeaves[table[1]]);
397  for (int i = 2; i < G.numberOfNodes(); i++) {
398  SList<PQLeafKey<edge,whaInfo*,bool>*> eliminatedKeys;
399  T.Reduction(outLeaves[table[i]], eliminatedKeys);
400  totalEliminatedKeys.conc(eliminatedKeys);
401  T.ReplaceRoot(inLeaves[table[i]]);
402  T.emptyAllPertinentNodes();
403  }
404 
405  for (PQLeafKey<edge, whaInfo*, bool> *key : totalEliminatedKeys) {
406  edge e = key->userStructKey();
407  delEdges.pushBack(e);
408  }
409 
410  //cleanup
411  for(node v : G.nodes) {
412  while (!inLeaves[v].empty()) {
413  PlanarLeafKey *L = inLeaves[v].popFrontRet();
414  delete L;
415  }
416  }
417 
418  T.Cleanup(); // Explicit call for destructor necessary. This allows to call virtual
419  // function CleanNode for freeing node information class.
420  }

422  static void doWorkHelper(ThreadMaster &master) {
423  const int nBlocks = master.numBlocks();
424 
425  do {
426  for(int i = 0; i < nBlocks; ++i) {
427  if(master.considerBlock(i)) {
428  const Graph &B = master.block(i);

431  computeSTNumbering(B, numbering, nullptr, nullptr, true);
432

434  planarize(B, numbering, *pCurrentDelEdges);
