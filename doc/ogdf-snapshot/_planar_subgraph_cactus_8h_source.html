89  newOrder.quicksort(adjCmp);

91  }
92  }

95  for (edge e : edges) {

97  node t = e->target();
98  int setS = components.find(id[s]);
99  int setT = components.find(id[t]);
100 
101  if(setS != setT) {
102  for(adjEntry adj : s->adjEntries) {
103  edge f = adj->theEdge();
104  node v = f->opposite(s);
105  int setV = components.find(id[v]);
106 
107  if(setV != setS && setV != setT) {
108  edge g = copy.searchEdge(v, t);
109 
110  if(g != nullptr) {
111  includeEdge[e] = includeEdge[f] = includeEdge[g] = true;
112  components.link(components.link(setS, setT), setV);
113  break;
114  }
115  }
116  }
117  }
118  }
119 
120  // connect the subgraph and store the result
121  for(edge e : edges) {
122  int setS = components.find(id[e->source()]);
123  int setT = components.find(id[e->target()]);
124 
125  if(setS != setT) {
126  includeEdge[e] = true;
127  components.link(setS, setT);
128  }
129

132  }
