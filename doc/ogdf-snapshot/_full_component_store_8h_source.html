105  void insert(const EdgeWeightedGraphCopy<T> &comp)
106  {
107  OGDF_ASSERT(!comp.empty());
108  OGDF_ASSERT(isTree(comp));
109 
110  // we temporarily use m_nodeCopy for nonterminals also
111  ArrayBuffer<node> tempUse(comp.numberOfNodes() / 2);
112 
113  // add all nonterminals of comp to m_graph
114  // and find terminals
115  Metadata<ExtraDataType> data;
116  for (node v : comp.nodes) {
117  node vO = comp.original(v);
118  if (m_nodeCopy[vO] == nullptr) {
119  node vC = m_graph.newNode();
120  m_nodeCopy[vO] = vC;
121  m_nodeOrig[vC] = vO;
122  tempUse.push(vO);
123  } else {
124  data.terminals.grow(1, vO);
125  }
126  }
127  IndexComparer<node> idx;
128  data.terminals.quicksort(idx);
129 
130  // add all edges of comp to m_graph
131  // and find start adjEntry
132  for (edge e : comp.edges) {
133  node uO = comp.original(e->source());
134  node vO = comp.original(e->target());
135  T weight = comp.weight(e);
136  edge eC = m_graph.newEdge(m_nodeCopy[uO], m_nodeCopy[vO], weight);
137  data.cost += weight;
138  if (m_isTerminal[uO]) {
139  data.start = eC->adjSource();
140  } else
141  if (m_isTerminal[vO]) {
142  data.start = eC->adjTarget();
143  }
144  }
145 
146  // cleanup m_nodeCopy
147  for (node vO : tempUse) {
148  m_nodeCopy[vO] = nullptr;
149  }
150 
151  m_components.push(data);
152  }

238  template<typename Fun>
239  void foreachAdjEntry(int i, Fun f) const
240  {
241  adjEntry start = m_components[i].start;
242  int size = m_components[i].terminals.size();
243  if (size == 2) {
244  f(start->twin());
245  return;
246  }
247  // size >= 3: do DFS over nonterminals (terminals are only leaves)
248  BoundedStack<adjEntry> stack(2*size - 2);
249  stack.push(start);
250  while (!stack.empty()) {
251  const adjEntry back = stack.pop()->twin();
252  f(back);
253  if (!this->isTerminal(back->theNode())) {
254  for (adjEntry adj = back->cyclicSucc(); adj != back; adj = adj->cyclicSucc()) {
255  stack.push(adj);
256  }
257  }
258  }
259  }

376  void computeAllLosses()
377  {
378  m_lossTerminal.init(this->m_graph, nullptr);
379 
380  // add zero-cost edges between all terminals (to be removed later),
381  // and set m_lossTerminal mapping for terminals
382  List<edge> zeroEdges;
383  const node s = this->m_terminals.front();
384  const node sC = this->m_nodeCopy[s];
385  m_lossTerminal[sC] = s;
386  for (ListConstIterator<node> it = this->m_terminals.begin().succ(); it.valid(); ++it) {
387  const node v = *it;
388  const node vC = this->m_nodeCopy[v];
389  m_lossTerminal[vC] = v;
390  zeroEdges.pushBack(this->m_graph.newEdge(sC, vC, 0));
391  }
392 
393  // compute minimum spanning tree
394  NodeArray<edge> pred(this->m_graph);
395  EdgeArray<bool> isLossEdge(this->m_graph, false);
396  computeMinST(sC, this->m_graph, this->m_graph.edgeWeights(), pred, isLossEdge);
397 
398  // remove zero-cost edges again
399  for (edge e : zeroEdges) {
400  this->m_graph.delEdge(e);
401  }
402 
403  // find loss bridges and compute loss value
404  for (int id = 0; id < this->size(); ++id) {
405  this->foreachAdjEntry(id, [&](adjEntry adj) {
406  edge e = adj->theEdge();

410  findLossTerminal(e->target(), pred);
