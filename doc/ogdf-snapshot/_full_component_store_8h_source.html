101  void insert(const EdgeWeightedGraphCopy<T> &comp)
102  {
103  OGDF_ASSERT(!comp.empty());
104  OGDF_ASSERT(isTree(comp));
105 
106  // we temporarily use m_nodeCopy for nonterminals also
107  ArrayBuffer<node> tempUse(comp.numberOfNodes() / 2);
108 
109  // add all nonterminals of comp to m_graph
110  // and find terminals
111  Metadata<ExtraDataType> data;
112  for (node v : comp.nodes) {
113  node vO = comp.original(v);
114  if (m_nodeCopy[vO] == nullptr) {
115  node vC = m_graph.newNode();
116  m_nodeCopy[vO] = vC;
117  m_nodeOrig[vC] = vO;
118  tempUse.push(vO);
119  } else {
120  data.terminals.grow(1, vO);
121  }
122  }
123  data.terminals.quicksort(GenericComparer<node, int>([](node v) { return v->index(); }));
124 
125  // add all edges of comp to m_graph
126  // and find start adjEntry
127  for (edge e : comp.edges) {
128  node uO = comp.original(e->source());
129  node vO = comp.original(e->target());
130  T weight = comp.weight(e);
131  edge eC = m_graph.newEdge(m_nodeCopy[uO], m_nodeCopy[vO], weight);
132  data.cost += weight;
133  if (m_isTerminal[uO]) {
134  data.start = eC->adjSource();
135  } else
136  if (m_isTerminal[vO]) {
137  data.start = eC->adjTarget();
138  }
139  }
140 
141  // cleanup m_nodeCopy
142  for (node vO : tempUse) {
143  m_nodeCopy[vO] = nullptr;
144  }
145 
146  m_components.push(data);
147  }

238  template<typename Fun>
239  void foreachAdjEntry(int i, Fun f) const
240  {
241  adjEntry start = m_components[i].start;
242  int size = m_components[i].terminals.size();
243  if (size == 2) {
244  f(start->twin());
245  return;
246  }
247  // size >= 3: do DFS over nonterminals (terminals are only leaves)
248  ArrayBuffer<adjEntry> stack(2*size - 2);
249  stack.push(start);
250  while (!stack.empty()) {
251  const adjEntry back = stack.popRet()->twin();
252  f(back);
253  if (!this->isTerminal(back->theNode())) {
254  for (adjEntry adj = back->cyclicSucc(); adj != back; adj = adj->cyclicSucc()) {
255  stack.push(adj);
256  }
257  }
258  }
259  }

378  void computeAllLosses()
379  {
380  m_lossTerminal.init(this->m_graph, nullptr);
381 
382  // add zero-cost edges between all terminals (to be removed later),
383  // and set m_lossTerminal mapping for terminals
384  List<edge> zeroEdges;
385  const node s = this->m_terminals.front();
386  const node sC = this->m_nodeCopy[s];
387  m_lossTerminal[sC] = s;
388  for (ListConstIterator<node> it = this->m_terminals.begin().succ(); it.valid(); ++it) {
389  const node v = *it;
390  const node vC = this->m_nodeCopy[v];
391  m_lossTerminal[vC] = v;
392  zeroEdges.pushBack(this->m_graph.newEdge(sC, vC, 0));
393  }
394 
395  // compute minimum spanning tree
396  NodeArray<edge> pred(this->m_graph);
397  EdgeArray<bool> isLossEdge(this->m_graph, false);
398  computeMinST(sC, this->m_graph, this->m_graph.edgeWeights(), pred, isLossEdge);
399 
400  // remove zero-cost edges again
401  for (edge e : zeroEdges) {
402  this->m_graph.delEdge(e);
403  }
404 
405  // find loss bridges and compute loss value
406  for (int id = 0; id < this->size(); ++id) {
407  this->foreachAdjEntry(id, [&](adjEntry adj) {
408  edge e = adj->theEdge();
409  if (!isLossEdge[e]) {
410  this->extra(id).bridges.pushBack(e);
411  findLossTerminal(e->source(), pred);
412  findLossTerminal(e->target(), pred);
413  } else {
414  this->extra(id).loss += this->m_graph.weight(e);
415  }
416  });
417  }
418  }
