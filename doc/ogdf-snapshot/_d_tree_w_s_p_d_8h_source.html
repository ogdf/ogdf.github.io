232 template<int Dim>
233 void DTreeWSPD<Dim>::wspdRecursive(int curr)
234 {
235  // iterate over all ordered pairs of children
236  for (int i = 0; i < m_pTree->numChilds(curr); i++)
237  {
238  // the first child index
239  const int first_child = m_pTree->child(curr, i);
240 
241  // the second loop for the pair
242  for (int j = i+1; j < m_pTree->numChilds(curr); j++)
243  {
244  // the second child index
245  const int second_child = m_pTree->child(curr, j);
246 
247  // call for each ordered pair the binary function
248  wspdRecursive(first_child, second_child);
249  }
250 
251  // now do all this for every child
252  wspdRecursive(first_child);
253  }
254 }

256 template<int Dim>
257 void DTreeWSPD<Dim>::wspdRecursive(int a, int b)
258 {
259  if (areWellSeparated(a, b))
260  {
261  // far enough away => approx
262  if (m_pIWSPD)
263  m_pIWSPD->onWellSeparatedPair(a, b);
264  } else
265  {
266  // two cells are too close
267  int small_node = a;
268  int large_node = b;
269 
270  // make sure the small one is not the bigger one
271  if (node(small_node).radius_sq > node(large_node).radius_sq)
272  {
273  std::swap(small_node, large_node);
274  }
275 
276  // split the bigger one
277  for (int i = 0; i < tree().numChilds(large_node); ++i)
278  {
279  // recurse on the child
280  wspdRecursive(small_node, tree().child(large_node, i));
281  }
282  }
283 }

285 template<int Dim>
286 bool DTreeWSPD<Dim>::areWellSeparated(int a, int b) const
287 {
288  // take the bigger radius
289  double r_max_sq = std::max(node(a).radius_sq, node(b).radius_sq);
290 
291  // compute the square distance
292  double dist_sq = 0.0;
293 
294  // the d^2 loop
295  for (int d = 0; d < Dim; d++)
296  {
297  double dx = node(a).x[d] - node(b).x[d];
298  dist_sq += dx * dx;
299  }
300 
301  // two circles are well separated iff d - 2r > sr
302  // where d is the distance between the two centers
303  // (not the distance of circles!)
304  // this ws <=> d > (s + 2)r
305  // more efficient: d^2 > (s + 2)^2 r^2
306  // d_sq > (s^2 + 4s + 4) * r_max
307  // const double s = (m_wspdSeparationFactor + 2.0);
308  //return dist_sq > (m_wspdSeparationFactor * m_wspdSeparationFactor + 4.0 * m_wspdSeparationFactor + 4) * r_max * r_max;
309 // return dist_sq > s * s * r_max_sq;
310  return dist_sq > m_wspdSeparationFactorPlus2Squared_cached * r_max_sq;
311 }

423 template<int Dim>
424 void DTreeWSPD<Dim>::updateTreeNodeGeometry(int curr)
425 {
426  // if this is an inner node
427  if (m_pTree->numChilds(curr))
428  {
429  // init with the bbox of the first child
430 
431  // iterate over the rest of the children
432  for (int i = 0; i < m_pTree->numChilds(curr); i++)
433  {
434  // the child index
435  const int child = m_pTree->child(curr, i);
436 
437  // compute the size of the subtree
438  updateTreeNodeGeometry(child);
439 
440  // lazy set the first
441  if (!i)
442  {
443  for (int d = 0; d < Dim; d++)
444  {
445  node(curr).min_x[d] = node(child).min_x[d];
446  node(curr).max_x[d] = node(child).max_x[d];
447  }
448  } else {
449  for (int d = 0; d < Dim; d++)
450  {
451  node(curr).min_x[d] = std::min(node(curr).min_x[d], node(child).min_x[d]);
452  node(curr).max_x[d] = std::max(node(curr).max_x[d], node(child).max_x[d]);
453  }
454  }
455  };
456  } // else it is a leaf
457  else {
458  // init from first point
459  for (int d = 0; d < Dim; d++)
460  {
461  node(curr).min_x[d] = node(curr).max_x[d] = point(tree().point(curr,0)).x[d];
462  }
463 
464  // and the remaining points
465  for (int i = 1; i < tree().numPoints(curr); ++i)
466  {
467  for (int d = 0; d < Dim; d++)
468  {
469  node(curr).min_x[d] = std::min(node(curr).min_x[d], point(tree().point(curr,i)).x[d]);
470  node(curr).max_x[d] = std::max(node(curr).max_x[d], point(tree().point(curr,i)).x[d]);
471  }
472  }
473  }
474 
475  // init the radius
476  node(curr).radius_sq = 0.0;
477 
478  // compute the center and radius of the rect
479  for (int d = 0; d < Dim; d++)
480  {
481  node(curr).x[d] = (node(curr).min_x[d] + node(curr).max_x[d]) * 0.5;
482  double s_x = (node(curr).max_x[d] - node(curr).min_x[d]);
483  node(curr).radius_sq += s_x * s_x;
484  }
485 
486  // and the smallest enclosing circle radius squared
487  node(curr).radius_sq *= 0.25; // sqrt(node(curr).radius) * 0.5 squared;
488 }
