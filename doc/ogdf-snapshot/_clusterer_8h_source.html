102  virtual double computeCIndex(const Graph &G, node v) override
103  {
104  OGDF_ASSERT(v->graphOf() == &G);
105  if (v->degree() < 2) return 1.0;
106  int conns = 0; //connections, without v
107  NodeArray<bool> neighbor(G, false);
108  for (adjEntry adjE : v->adjEntries)
109  {
110  neighbor[adjE->twinNode()] = true;
111  }
112  for (adjEntry adjE : v->adjEntries)
113  {
114  for (adjEntry adjEE : adjE->twinNode()->adjEntries)
115  {
116  if (neighbor[adjEE->twinNode()])
117  conns++;
118  }
119  }
120  //connections were counted twice
121  double index = conns / 2.0;
122  return index / (v->degree()*(v->degree() - 1));
123  }
