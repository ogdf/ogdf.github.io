382 inline bool AdjElement::isBetween(adjEntry adjBefore, adjEntry adjAfter) const {
383 #ifdef OGDF_DEBUG
384  node v = this->theNode();
385  OGDF_ASSERT(adjBefore->theNode() == v);
386  OGDF_ASSERT(adjAfter->theNode() == v);
387 #endif
388  bool result = this != adjBefore && this != adjAfter && adjBefore != adjAfter;
389 
390  if (result) {
391  adjEntry adj = adjBefore;
392  for (; adj != this && adj != adjAfter; adj = adj->cyclicSucc());
393  result = adj == this;
394  }
395 
396  return result;

987  void collapse(NODELIST &nodesToCollapse){
988  node v = nodesToCollapse.popFrontRet();
989  while (!nodesToCollapse.empty())
990  {
991  node w = nodesToCollapse.popFrontRet();
992  adjEntry adj = w->firstAdj();

996  edge e = adj->theEdge();
997  if (e->source() == v || e->target() == v)
