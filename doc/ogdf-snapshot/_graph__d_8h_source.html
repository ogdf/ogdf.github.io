385 inline bool AdjElement::isBetween(adjEntry adjBefore, adjEntry adjAfter) const {
386 #ifdef OGDF_DEBUG
387  node v = this->theNode();
388  OGDF_ASSERT(adjBefore->theNode() == v);
389  OGDF_ASSERT(adjAfter->theNode() == v);
390 #endif
391  bool result = this != adjBefore && this != adjAfter && adjBefore != adjAfter;
392 
393  if (result) {
394  adjEntry adj = adjBefore;
395  for (; adj != this && adj != adjAfter; adj = adj->cyclicSucc());
396  result = adj == this;
397  }
398 
399  return result;

1002  void collapse(NODELIST &nodesToCollapse){

1004  while (!nodesToCollapse.empty())
1005  {
1006  node w = nodesToCollapse.popFrontRet();
1007  adjEntry adj = w->firstAdj();
