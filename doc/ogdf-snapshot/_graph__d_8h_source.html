947  template<class NODELIST>
948  void collapse(NODELIST &nodes){
949  node v = nodes.popFrontRet();
950  while (!nodes.empty())
951  {
952  node w = nodes.popFrontRet();
953  adjEntry adj = w->firstAdj();
954  while (adj != nullptr)
955  {
956  adjEntry succ = adj->succ();
957  edge e = adj->theEdge();
958  if (e->source() == v || e->target() == v)
959  delEdge(e);
960  else if (e->source() == w)
961  moveSource(e,v);
962  else
963  moveTarget(e,v);
964  adj = succ;
965  }
966  delNode(w);
967  }
968  }
