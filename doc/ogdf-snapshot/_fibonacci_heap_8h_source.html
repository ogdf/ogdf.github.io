316 template<typename T, typename C>
317 inline void FibonacciHeap<T, C>::compress()
318 {
319  size_t maxr = 0;
320 
321  for(auto it = m_knot->next; it != m_knot;) {
322  FibonacciHeapNode<T> *next = it->next;
323 
324  size_t r = it->rank;
325  maxr = std::max(r, maxr);
326  while(m_ranked[r]) {
327  if(this->comparator()(m_ranked[r]->value, it->value)) {
328  link(m_ranked[r], it);
329  it = m_ranked[r];
330  } else {
331  link(it, m_ranked[r]);
332  }
333  m_ranked[r] = nullptr;
334  r++;
335  maxr = std::max(maxr, r);
336  }
337  m_ranked[r] = it;
338 
339  it = next;
340  }
341 
342  for(size_t i = 0; i <= maxr; i++) {
343  m_ranked[i] = nullptr;
344  }

397 template<typename T, typename C>
398 inline void FibonacciHeap<T, C>::restore(FibonacciHeapNode<T> *heapNode)
399 {
400  for(;;) {
401  FibonacciHeapNode<T> *parent = heapNode->parent;
402  if(parent == nullptr) {
403  return;
404  }
405 
406  // We need to make sure parent has valid children after the splice.
407  parent->rank--;
408  if(parent->rank == 0) {
409  parent->child = nullptr;
410  } else if(parent->child == heapNode) {
411  parent->child = heapNode->next;
412  }
413 
414  heapNode->parent = nullptr;
415  splice(m_knot, heapNode);
416 
417  // If parent is unmarked we can stop cut cascade.
418  if(!parent->marked) {
419  parent->marked = true;
420  return;
421  }
422 
423  heapNode = parent;
424  }
