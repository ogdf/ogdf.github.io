210  doCall(
211  const Graph &graph,
212  const List<edge> &preferredEdges,
213  List<edge> &delEdges,
214  const EdgeArray<TCost> *pCost,
215  bool preferredImplyPlanar) override
216  {
217  Module::ReturnType result = Module::ReturnType::Error;
218  delEdges.clear();
219 
220  // scale the costs and do multiple runs (if needed)
221  List<edge> delEdgesCurrentBest;
222 
223  // normalize costs to [0,1] and apply randomness
224  EdgeArray<TCost> normalizedCost(graph);
225 
226  for (auto i=0u; i < m_runs; i++) {
227 
228  List<edge> heuDelEdges;
229 
230  if(pCost == nullptr) {
231  result = m_heuristic.call(graph, preferredEdges, heuDelEdges, preferredImplyPlanar);
232  } else {
233  std::uniform_real_distribution<TCost> distribution (0.0,1.0);
234  edge firstEdge = graph.firstEdge();
235  TCost maxCost = firstEdge == nullptr ? 0 : (*pCost)[firstEdge];
236  TCost minCost = firstEdge == nullptr ? 0 : (*pCost)[firstEdge];
237  for (edge e: graph.edges) {
238  Math::updateMax(maxCost, (*pCost)[e]);
239  Math::updateMin(minCost, (*pCost)[e]);
240  }
241  for (edge e: graph.edges) {
242  // do not merge with first FOR !
243  // normalized = pCost transformed to [0,1]
244  TCost normalized = 1;
245  if (maxCost > minCost) {
246  normalized = ((*pCost)[e] - minCost) / (maxCost - minCost);
247  }
248  // now use randomness
249  normalizedCost[e] = (1.0 - m_randomness)*normalized + m_randomness*distribution(m_randomGenerator);
250  }
251  result = m_heuristic.call(graph, normalizedCost, preferredEdges, heuDelEdges, preferredImplyPlanar);
252  }
253 
254  if(Module::isSolution(result)) {
255  GraphCopy copy(graph);
256 
257  if (pCost != nullptr) {
258  GenericComparer<edge, TCost> cmp(normalizedCost);
259  heuDelEdges.quicksort(cmp);
260  }
261 
262  for (edge e : heuDelEdges) {
263  copy.delEdge(copy.copy(e));
264  }
265 
266  delEdgesCurrentBest.clear();
267  for (edge e : heuDelEdges) {
268  edge f = copy.newEdge(e);
269 
270  if (!isPlanar(copy)) {
271  delEdgesCurrentBest.pushBack(e);
272  copy.delEdge(f);
273  }
274  }
275 
276  if (pCost == nullptr) {
277  if (i == 0 || delEdgesCurrentBest.size() < delEdges.size()) {
278  // better solution: copy to delEdges
279  delEdges.clear();
280  for (auto e: delEdgesCurrentBest) { delEdges.pushBack(e); };
281  }
282  } else {
283  if (i == 0 || weightOfList(delEdgesCurrentBest,normalizedCost) < weightOfList(delEdges,normalizedCost)) {
284  // better solution: copy to delEdges
285  delEdges.clear();
286  for (auto e: delEdgesCurrentBest) { delEdges.pushBack(e); };
287  }
288  }
289  }
290  }

292  return result;
293  }
