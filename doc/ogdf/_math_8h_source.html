204 inline void getFraction(double d, int &num, int &denom, const double epsilon = 5e-10, const int count = 10) {
205  ArrayBuffer<int> continuedFrac;
206 
207  // build continued fraction
208  int z((int)d);
209  continuedFrac.push(z);
210  d = d - z;
211  int i = 0;
212  while (d > epsilon && i++ < count) {
213  d = 1 / d;
214  z = (int)d;
215  continuedFrac.push(z);
216  d = d - z;
217  }
218 
219  // simplify continued fraction to simple fraction
220  num = 1;
221  denom = 0;
222  while (!continuedFrac.empty()) {
223  int last = continuedFrac.popRet();
224  std::swap(num, denom);
225  num += last * denom;
226  }
227 }
