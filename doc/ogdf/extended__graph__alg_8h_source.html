109  {

112  {
113  subGraph.newEdge(nodeTableOrig2New[e->source()],nodeTableOrig2New[e->target()]);
114  mark[e] = true;
115  }
116  }

306 template<typename T>
307 T computeMinST(node s, const Graph &G, const EdgeArray<T> &weight, NodeArray<edge> &pred, EdgeArray<bool> &isInTree)
308 {
309  BinaryHeap2<T, node> pq(G.numberOfNodes()); // priority queue of front vertices
310  NodeArray<int> pqpos(G, -1); // position of each node in pq
311 
312  // insert start node
313  T tmp(0);
314  pq.insert(s, tmp, &pqpos[s]);
315 
316  // extract the nodes again along a minimum ST
317  NodeArray<bool> processed(G, false);
318  pred.init(G, NULL);
319  while (!pq.empty()) {
320  const node v = pq.extractMin();
321  processed[v] = true;
322  for (adjEntry adj = v->firstAdj(); adj; adj = adj->succ()) {
323  const node w = adj->twinNode();
324  const edge e = adj->theEdge();
325  const int wPos = pqpos[w];
326  if (wPos == -1) {
327  tmp = weight[e];
328  pq.insert(w, tmp, &pqpos[w]);
329  pred[w] = e;
330  } else
331  if (!processed[w]
332  && weight[e] < pq.getPriority(wPos)) {
333  pq.decreaseKey(wPos, weight[e]);
334  pred[w] = e;
335  }
336  }
337  }
338 
339  int rootcount = 0;
340  T treeWeight = 0;
341  isInTree.init(G, false);
342  for (node v = G.firstNode(); v; v = v->succ()) {
343  if (!pred[v]) {
344  ++rootcount;
345  } else {
346  isInTree[pred[v]] = true;
347  treeWeight += weight[pred[v]];
348  }
349  }
350  OGDF_ASSERT(rootcount == 1); // is connected
351 
352  return treeWeight;
353 }//computeMinST

362 template<typename T>
363 T makeMinimumSpanningTree(Graph &G, const EdgeArray<T> &weight)
364 {
365  T total(0);
366  List< Prioritized<edge, T> > sortEdges;
367  for (edge e = G.firstEdge(); e; e = e->succ()) {
368  sortEdges.pushBack(Prioritized<edge,T>(e, weight[e]));
369  }
370  sortEdges.quicksort();
371 
372  // now let's do Kruskal's algorithm
373  NodeArray<int> setID(G);
374  DisjointSets<> uf(G.numberOfNodes());
375  for (node v = G.firstNode(); v; v = v->succ()) {
376  setID[v] = uf.makeSet();
377  }
378 
379  for (ListConstIterator< Prioritized<edge,T> > it = sortEdges.begin(); it.valid(); ++it) {
380  const edge e = (*it).item();
381  const int v = setID[e->source()];
382  const int w = setID[e->target()];
383  if (uf.find(v) != uf.find(w)) {
384  uf.link(uf.find(v), uf.find(w));
385  total += weight[e];
386  } else {
387  G.delEdge(e);
388  }
389  }
390  return total;
391 }
