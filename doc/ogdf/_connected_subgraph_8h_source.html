308 void ConnectedSubgraph<T>::recursion(
309  Graph& SG,
310  NodeArray<bool> &nodeVisited,
311  EdgeArray<bool> &edgeVisited,
312  const node& nG,
313  const NodeArray<T>& nodeLengthG,
314  NodeArray<T>& nodeLengthSG,
315  const EdgeArray<T>& edgeLengthG,
316  EdgeArray<T>& edgeLengthSG,
317  NodeArray<node>& nSG_to_nG,
318  EdgeArray<edge>& eSG_to_eG,
319  NodeArray<node>& nG_to_nSG,
320  EdgeArray<edge>& eG_to_eSG)
321 {
322  node nSG = SG.newNode();
323  nodeLengthSG[nSG] = nodeLengthG[nG];
324  nG_to_nSG[nG] = nSG;
325  nSG_to_nG[nSG] = nG;
326  nodeVisited[nG] = true;
327 
328  edge eG;
329  forall_adj_edges(eG, nG)
330  {
331  if (!nodeVisited[eG->source()])
332  recursion(SG, nodeVisited, edgeVisited, eG->source(),
333  nodeLengthG, nodeLengthSG, edgeLengthG, edgeLengthSG,
334  nSG_to_nG, eSG_to_eG, nG_to_nSG, eG_to_eSG);
335  else if (!nodeVisited[eG->target()])
336  recursion(SG, nodeVisited, edgeVisited, eG->target(),
337  nodeLengthG, nodeLengthSG, edgeLengthG, edgeLengthSG,
338  nSG_to_nG, eSG_to_eG, nG_to_nSG, eG_to_eSG);
339  if (!edgeVisited[eG])
340  {
341  edge eSG = SG.newEdge(nG_to_nSG[eG->source()], nG_to_nSG[eG->target()]);
342  edgeLengthSG[eSG] = edgeLengthG[eG];
343  eG_to_eSG[eG] = eSG;
344  eSG_to_eG[eSG] = eG;
345  edgeVisited[eG] = true;
346  }
347  }
