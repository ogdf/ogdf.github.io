215  int link(int set1, int set2)
216  {
217  if (set1==set2) return -1;
218  this->numberOfSets--;
219  int superset = link(LinkOption<linkOption>(), set1, set2);
220  //Collapse subset tree.
221  if (compressionOption == CO)
222  {

224  int id = subset;

226  {

228  this->parents[id]=superset;
229  }
230  this->siblings[id] = this->siblings[superset];
231  this->siblings[superset] = subset;
232  }
233  return superset;
234  }
