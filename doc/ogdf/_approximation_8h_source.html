89  int findComponentAndMaxBasis(std::unique_ptr<ArrayBuffer<std::pair<node,int>>> &maxBasis,
90  BlowupGraph<T> &blowupGraph,
91  const BlowupComponents<T> &gamma)
92  {
93  // there should always be saturated flow to the component roots
94  // (contracted matroid)
95  EdgeArray<int> lB(blowupGraph.getGraph(), 0);
96  for (adjEntry adj : blowupGraph.getSource()->adjEntries) {
97  const edge e = adj->theEdge();
98  lB[e] = blowupGraph.getCapacity(e);
99  }
100 
101  // compute weights of core edges and add source->core edges
102  TemporaryEdges sourceCoreEdges(blowupGraph);
103  EdgeArray<double> cost(blowupGraph.getGraph(), 0);
104  for (node v : blowupGraph.core()) {
105  // compute weight of core edge
106  double weight = blowupGraph.computeCoreWeight(v);
107 
108  // add edges from source to core edges v
109  edge e = sourceCoreEdges.add(blowupGraph.getSource(), v, 0, blowupGraph.getCoreCapacity(v));
110  cost[e] = -weight;
111  }
112 
113  NodeArray<int> supply(blowupGraph.getGraph(), 0);
114  EdgeArray<int> flow(blowupGraph.getGraph());
115  MinCostFlowReinelt<double> mcf;
116 
117  for (int id = 1; id <= gamma.size(); ++id) {
118  /* We want to find maximum-weight basis B \in B^K_Q
119  * B_Q = minimal edge set to remove after contracting Q to obtain feasible solution (blowup graph)
120  * = bases of gammoid M_Q
121  * B^K_Q = { B \in B_Q | B \subseteq K } [K is the set of core edges]
122  * = bases of gammoid M^K_Q
123  * M^K_Q is gammoid "obtained by restricting M_Q to K"
124  * M_Q = M'_Q / X' (M'_Q contracted by X') is a gammoid
125  * M'_Q = gammoid from X \cup X' to Y in D' with arc-disjointness instead of vertex-disjointness
126  * D' is D like for separation but without source node and with interim node v_e for each edge e
127  * X = inserted nodes v_e in each edge in blowup graph
128  * X' = the (arbitrary) roots of each component
129  * Y = Q \cup {t}
130  * that means:
131  * I (subset of X) is an independent set of M_Q
132  * <=> (if X' is always an independent set in M'_Q ... which we assume here)
133  * I \cup X' is an independent set of M'_Q
134  * Restricting to K means:
135  * only put v_e in X with e \in K (that is, we only need to *generate* v_e if e \in K)
136  * Hence:
137  * - we generate D'^K (this is blowupGraph)
138  * - compute the max flow from X^K \cup X' to Q \cup {t}
139  * - ASSERT that X' is "saturated"!
140  * - check which subset of X is saturated -> these are the nodes representing the edge set we need
141  */
142  // add edges from component's terminals to target
143  TemporaryEdges Q_to_target(blowupGraph);
144  for (node t : gamma.terminals(id)) {
145  Q_to_target.add(t, blowupGraph.getTarget(), 0, blowupGraph.getLCM() * blowupGraph.getY());
146  // the last value is an upper bound for the capacity
147  }
148  // TODO: we could also use static edges from all terminals to the target
149  // and just change their capacities each time; needs extra data structures
150 
151  std::unique_ptr<ArrayBuffer<std::pair<node,int>>> basis(new ArrayBuffer<std::pair<node,int>>);
152 
153  int rank = gammoidGetRank(blowupGraph);
154  OGDF_ASSERT(rank >= blowupGraph.getY() + blowupGraph.getLCM());
155  supply[blowupGraph.getSource()] = rank;
156  supply[blowupGraph.getTarget()] = -rank;
157 
158 #ifdef OGDF_STEINER_TREE_GOEMANS_APPROXIMATION_LOGGING
159  std::cout << "Computing min-cost flow for blowup component " << id << " of " << gamma.size() << std::endl;
160  std::cout << " * terminals of component are " << gamma.terminals(id) << std::endl;
161  for (node v : blowupGraph.getGraph().nodes) {
162  if (supply[v] > 0) {
163  std::cout << " * supply node " << v << " with supply " << supply[v] << std::endl;
164  }
165  if (supply[v] < 0) {
166  std::cout << " * demand node " << v << " with demand " << -supply[v] << std::endl;
167  }
168  }
169  for (edge e : blowupGraph.getGraph().edges) {
170  std::cout
171  << " * edge " << e
172  << " with cost " << blowupGraph.getCost(e)
173  << " and flow bounds [" << lB[e]
174  << ", " << blowupGraph.getCapacity(e)
175  << "]" << std::endl;
176  }
177 #endif
178 
179  // find maximum weight basis
180 #ifdef OGDF_DEBUG
181  bool feasible =
182 #endif
183  mcf.call(blowupGraph.getGraph(), lB, blowupGraph.capacities(), cost, supply, flow);
184  OGDF_ASSERT(feasible);
185  OGDF_ASSERT(mcf.checkComputedFlow(blowupGraph.getGraph(), lB, blowupGraph.capacities(), cost, supply, flow));
186 
187  double weight(0);
188  for (edge e : sourceCoreEdges) {
189  if (flow[e] > 0) {
190  basis->push(std::make_pair(e->target(), flow[e]));
191  weight -= flow[e] * cost[e];
192  }
193  }
194 
195 #ifdef OGDF_STEINER_TREE_GOEMANS_APPROXIMATION_LOGGING
196  std::cout
197  << "Basis weight is " << weight << std::endl
198  << "Checking if "
199  << gamma.cost(id) << "(component cost) * " << blowupGraph.getLCM()
200  << "(lcm) <= " << weight
201  << "(basis weight)" << std::endl;
202 #endif
203 
204  // we choose the component with max cost*N <= weight
205  if (gamma.cost(id) * blowupGraph.getLCM() <= weight + m_eps) {
206 #ifdef OGDF_STEINER_TREE_GOEMANS_APPROXIMATION_LOGGING
207  std::cout << "Using basis because it is feasible." << std::endl;
208 #endif
209  maxBasis.swap(basis);
210  return id;
211  }
212  }
213  return 0; // no component chosen, fail
214  }

218  int findCheapestComponentAndRemainingBasis(std::unique_ptr<ArrayBuffer<std::pair<node,int>>> &maxBasis,
219  const BlowupGraph<T> &blowupGraph,
220  const BlowupComponents<T> &gamma)
221  {
222  // find cheapest component
223  int compId = 0;
224  double cost = 0;
225  for (int id = 1; id <= gamma.size(); ++id) {
226  if (gamma.cost(id) > cost) {
227  cost = gamma.cost(id);
228  compId = id;
229  }
230  }
231  // use all core edges as basis
232  maxBasis.reset(new ArrayBuffer<std::pair<node,int>>());
233  for (node v : blowupGraph.core()) {
234  maxBasis->push(std::make_pair(v, blowupGraph.getCoreCapacity(v)));
235  }
236  return compId;
237  }

240  void addComponent(NodeArray<bool> &isNewTerminal, const BlowupGraph<T> &blowupGraph, const edge rootEdge)
241  {
242  OGDF_ASSERT(blowupGraph.isTerminal(rootEdge->source()));
243  ArrayBuffer<node> stack;
244  stack.push(rootEdge->target());
245  while (!stack.empty()) {
246  const node v = stack.popRet();
247  if (blowupGraph.isTerminal(v)) {
248  continue;
249  }
250  const node vO = blowupGraph.getOriginal(v);
251  if (vO) {
252  isNewTerminal[vO] = true;
253  }
254  for (adjEntry adj : v->adjEntries) {
255  const node w = adj->theEdge()->target();
256  if (v != w) { // outgoing edge
257  stack.push(w);
258  }
259  }
260  }
261  }

264  void removeFractionalBasisAndCleanup(ArrayBuffer<std::pair<node,int>> &basis,
265  BlowupGraph<T> &blowupGraph,
266  BlowupComponents<T> &gamma)
267  {
268 #ifdef OGDF_STEINER_TREE_GOEMANS_APPROXIMATION_LOGGING
269  std::cout << "Remove basis from blowup graph" << std::endl;
270 #endif
271  // remove B from K (K := K \ B) and from blowup graph (X := X - B)
272  // and, while at it, remove cleanup edges from blowup graph (X := X - F)
273  // and fix components that have no incoming edges
274  ArrayBuffer<Prioritized<node, int>> fractionalCoreEdges; // we defer fractional basis elements
275  for (auto p : basis) {
276  const node v = p.first;
277  const int count = p.second;
278  int origCap = blowupGraph.getCoreCapacity(v);
279  OGDF_ASSERT(count <= origCap);
280 #ifdef OGDF_STEINER_TREE_GOEMANS_APPROXIMATION_LOGGING
281  std::cout << " * node " << v << " with count " << count << " (of " << origCap << ")" << std::endl;
282 #endif
283  if (count < origCap) { // only remove a fraction?
284  fractionalCoreEdges.push(Prioritized<node,int>(v, -count));
285 #ifdef OGDF_STEINER_TREE_GOEMANS_APPROXIMATION_LOGGING
286  std::cout << " -> deferred because fractional" << std::endl;
287 #endif
288  } else {
289  // we are deleting the core edge from the whole component
290  blowupGraph.delCore(v);
291  blowupGraph.removeBasis(v);
292 #ifdef OGDF_STEINER_TREE_GOEMANS_APPROXIMATION_LOGGING
293  std::cout << " -> done" << std::endl;
294 #endif
295  }
296  }
297  fractionalCoreEdges.quicksort(); // sort decreasing by flow value
298 #ifdef OGDF_STEINER_TREE_GOEMANS_APPROXIMATION_LOGGING
299  if (!fractionalCoreEdges.empty()) {
300  std::cout << "Deferred core edges:" << std::endl;
301  }
302 #endif
303  for (auto p : fractionalCoreEdges) {
304  const node v = p.item();
305  const int count = -p.priority();
306  int origCap = blowupGraph.getCoreCapacity(v);
307 #ifdef OGDF_STEINER_TREE_GOEMANS_APPROXIMATION_LOGGING
308  std::cout << " * node " << v << " with count " << count << " of " << origCap << std::endl;
309 #endif
310  OGDF_ASSERT(count <= origCap);
311  // copy (split) the component
312  blowupGraph.copyComponent(blowupGraph.findRootEdge(v), count, origCap - count);
313  // we are deleting the core edge from the whole component
314  blowupGraph.delCore(v);
315  blowupGraph.removeBasis(v);
316  }
317  }

338 template<typename T>
339 void
340 Approximation<T>::solve(NodeArray<bool> &isNewTerminal)
341 {
342 #ifdef OGDF_STEINER_TREE_GOEMANS_APPROXIMATION_LOGGING
343  std::cout << "Start solving based on LP solution" << std::endl;
344  int iteration = 0;
345 #endif
346  CoreEdgeRandomSpanningTree<T> cer(m_rng);
347  BlowupGraph<T> blowupGraph(m_G, m_terminals, m_fullCompStore, cer, m_eps);
348 
349  while (blowupGraph.terminals().size() > 1) { // T is not a Steiner tree
350 #ifdef OGDF_STEINER_TREE_GOEMANS_APPROXIMATION_LOGGING
351  std::cout << "Iteration " << ++iteration << " with " << blowupGraph.terminals().size() << " terminals" << std::endl;
352 #endif
353  BlowupComponents<T> gamma(blowupGraph); // Gamma(X)
354 
355  OGDF_ASSERT(isLoopFree(blowupGraph.getGraph()));
356 
357  // take a component Q in Gamma(X)
358  std::unique_ptr<ArrayBuffer<std::pair<node,int>>> maxBasis;
359  int compId = blowupGraph.getY() > 0
360  ? findComponentAndMaxBasis(maxBasis, blowupGraph, gamma)
361  : findCheapestComponentAndRemainingBasis(maxBasis, blowupGraph, gamma);
362  OGDF_ASSERT(compId != 0);
363 
364  // add component Q to T
365  addComponent(isNewTerminal, blowupGraph, gamma.rootEdge(compId));
366 
367  // remove (maybe fractional) basis and do all the small things necessary for update
368  removeFractionalBasisAndCleanup(*maxBasis, blowupGraph, gamma);
369 
370  // contract (X := X / Q)
371  blowupGraph.contract(gamma.terminals(compId));
372 
373  if (blowupGraph.terminals().size() > 1) {
374  blowupGraph.updateSpecialCapacities();
375  }
376  }
377 }
