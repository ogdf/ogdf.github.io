70  {
71  if (left == right) { // leaf
72  segmentTree[nodeIndex] = baseArray[left];

173  void dfsHeavyPathDecomposition(node v, node closestSteinerUpNode)
174  {
175  weightOfSubtree[v] = 1;
176  node heaviestSon = nullptr;
177  closestSteinerAncestor[v] = closestSteinerUpNode;
178 
179  for(adjEntry adj : v->adjEntries) {
180  edge e = adj->theEdge();
181  node son = adj->twinNode();
182 
183  if (weightOfSubtree[son] != 0) { // the parent
184  continue;
185  }
186 
187  nodeLevel[son] = nodeLevel[v] + 1;
188  distanceToRoot[son] = distanceToRoot[v] + tree.weight(e);
189 
190  dfsHeavyPathDecomposition(son, v);
191 
192  fatherOfChain[chainOfNode[son]] = v;
193  weightOfSubtree[v] += weightOfSubtree[son];
194  if (heaviestSon == nullptr
195  || weightOfSubtree[heaviestSon] < weightOfSubtree[son]) {
196  heaviestSon = son;
197  }
198  }
199 
200  if (heaviestSon == nullptr) { // it's leaf => new chain
201  OGDF_ASSERT((v->degree() <= 1));
202 
203  std::vector<node>new_chain;
204  chains.push_back(new_chain);
205  chainsOfTerminals.push_back(new_chain);
206 
207  fatherOfChain.push_back(nullptr);
208  chainOfNode[v] = (int)chains.size()-1;
209  } else {
210  chainOfNode[v] = chainOfNode[heaviestSon];
211  }
212 
213  chains[chainOfNode[v]].push_back(v);
214  chainsOfTerminals[chainOfNode[v]].push_back(v);
215  positionOnChain[v] = (int)chains[chainOfNode[v]].size()-1;
216  }

223  node binarySearchUpmostTerminal(node v, const std::vector<node> &chainOfTerminals) const
224  {
225  int left = 0, middle, right = (int)chainOfTerminals.size()-1;
226  while (left <= right) {
227  middle = (left+right)>>1;
228 
229  if (nodeLevel[chainOfTerminals[middle]] >= nodeLevel[v]) {
230  right = middle-1;

232  left = middle+1;
233  }
234  }

251  void computeBottleneckOnBranch(node x, node ancestor, T &longestPathDistance, T &fromLowestToAncestor) const
252  {
253  node upmostTerminal = x;
254  while (closestSteinerAncestor[chains[chainOfNode[x]][0]] != nullptr
255  && nodeLevel[closestSteinerAncestor[chains[chainOfNode[x]][0]]] >= nodeLevel[ancestor]) {
256  Math::updateMax(longestPathDistance, longestDistToSteinerAncestorOnChain[chainOfNode[x]][positionOnChain[x]]);
257 
258  if (!chainsOfTerminals[chainOfNode[x]].empty()
259  && nodeLevel[chainsOfTerminals[chainOfNode[x]][0]] <= nodeLevel[x]) {
260  upmostTerminal = chainsOfTerminals[chainOfNode[x]][0];
261  }
262  x = fatherOfChain[chainOfNode[x]];
263  }
264 
265  // search the upmost terminal on the current chain that has level >= level[ancestor]
266  node upmostTerminalLastChain = binarySearchUpmostTerminal(ancestor, chainsOfTerminals[chainOfNode[x]]);
267  if (nodeLevel[upmostTerminalLastChain] > nodeLevel[x]) {
268  upmostTerminalLastChain = nullptr;
269  }
270  if (upmostTerminalLastChain != nullptr) {
271  upmostTerminal = upmostTerminalLastChain;
272  }
273 
274  if (upmostTerminalLastChain != nullptr) {
275  Math::updateMax(longestPathDistance,
276  getMaxSegmentTree(longestDistToSteinerAncestorSegTree[chainOfNode[x]], 0, 0,
277  static_cast<int>(chains[chainOfNode[x]].size()) - 1,
278  positionOnChain[upmostTerminalLastChain] + 1,
279  positionOnChain[x]));
280  }
281 
282  fromLowestToAncestor = distanceToAncestor(upmostTerminal, ancestor);
283  }
