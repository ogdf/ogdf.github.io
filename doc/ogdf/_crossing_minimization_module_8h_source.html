140  static int computeCrossingNumber(GraphCopy &graphCopy,
141  const EdgeArray<int> *pCost,
142  const EdgeArray<uint32_t> *pEdgeSubGraphs)
143  {
144  if (pCost == nullptr) {
145  return graphCopy.numberOfNodes() - graphCopy.original().numberOfNodes();
146  } else {
147  int crossingNumber = 0;
148  for (node v : graphCopy.nodes) {
149  if (graphCopy.isDummy(v)) {
150  // Dummy node (crossing) found, calculate its cost.
151  edge e1 = graphCopy.original(v->firstAdj()->theEdge());
152  edge e2 = graphCopy.original(v->lastAdj()->theEdge());
153  if (pEdgeSubGraphs != nullptr) {
154  int subgraphCounter = 0;
155  for (int i = 0; i < 32; i++) {
156  if (((*pEdgeSubGraphs)[e1] & (1<<i)) != 0
157  && ((*pEdgeSubGraphs)[e2] & (1<<i)) != 0) {
158  subgraphCounter++;
159  }
160  }
161  crossingNumber += (subgraphCounter *
162  (*pCost)[e1] * (*pCost)[e2]);
163  } else {
164  crossingNumber += (*pCost)[e1] * (*pCost)[e2];
165  }
166  }
167  }
168  return crossingNumber;
