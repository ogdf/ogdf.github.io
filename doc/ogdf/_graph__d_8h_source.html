896  template<class NODELIST>
897  void collaps(NODELIST &nodes){
898  node v = nodes.popFrontRet();
899  while (!nodes.empty())
900  {
901  node w = nodes.popFrontRet();
902  adjEntry adj = w->firstAdj();
903  while (adj !=0)
904  {
905  adjEntry succ = adj->succ();
906  edge e = adj->theEdge();
907  if (e->source() == v || e->target() == v)
908  delEdge(e);
909  else if (e->source() == w)
910  moveSource(e,v);
911  else
912  moveTarget(e,v);
913  adj = succ;
914  }
915  delNode(w);
916  }
917  }
