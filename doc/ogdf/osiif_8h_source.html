474  void convertSenseToBound(
475  double inf,
476  const char sense,
477  const double right,
478  const double range,
479  double& lower,
480  double& upper) const
481  {
482  switch (sense) {
483  case 'E':
484  lower = upper = right;
485  break;
486  case 'L':
487  lower = -inf;
488  upper = right;
489  break;
490  case 'G':
491  lower = right;
492  upper = inf;
493  break;
494  case 'R':
495  lower = right - range;
496  upper = right;
497  break;
498  case 'N':
499  lower = -inf;
500  upper = inf;
501  break;
502  }
503  }
