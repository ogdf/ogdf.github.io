265 typename HotQueue<V, P, H>::Handle HotQueue<V, P, H>::push(
266  const V &value, const P &priority)
267 {
268  m_size++;
269 
270  std::size_t ind = bucketIndex(priority);

272  if(m_heapedBucket == NONE) {
273  m_heapedBucket = ind;
274  }
275 
276  if(ind == m_heapedBucket) {
277  m_heapSize++;
278  HeapHandle handle = m_heap.push(std::make_pair(value, priority));
279  return Handle(handle);
280  } else {

282 
283  if(bucketAt(ind) != nullptr) {
284  bucketAt(ind)->prev = queueNode;
285  }
286  queueNode->next = bucketAt(ind);
287  bucketAt(ind) = queueNode;
288 
289  m_lastBucket = std::max(m_lastBucket, ind);

291  }

295 template<typename V, typename P, template<typename T, typename C> class H>
296 void HotQueue<V, P, H>::pop()
297 {
298  m_size--;
299 
300  m_heap.pop();
301  m_heapSize--;
302 
303  /*
304  * If the heap became empty and there is non-empty bucket afterwards
305  * we need to heapify first non-empty bucket. If the heap became empty
306  * but there is no non-empty bucket afterwards it may be the case that
307  * element will be inserted into the current heap (therefore, do nothing).
308  */
309  if(!(m_heapSize == 0 && m_heapedBucket != m_lastBucket)) {
310  return;
311  }
312 
313  // Find first non-empty bucket.
314  do {
315  m_heapedBucket++;
316  } while(bucketAt(m_heapedBucket) == nullptr);
317 
318  // Move bucket contents into a heap.
319  for(HotQueueNode<V, P> *it = bucketAt(m_heapedBucket); it != nullptr;) {
320  m_heapSize++;
321  m_heap.push(std::make_pair(it->value, it->priority));
322 
323  HotQueueNode<V, P> *next = it->next;
324  delete it;
325  it = next;
326  }
327  bucketAt(m_heapedBucket) = nullptr;
328 }

332 void HotQueue<V, P, H>::decrease(

338  // Simple case, just use native heap decrease key.
339  const HeapHandle &elem = handle.heapHandle;
340  m_heap.decrease(elem, std::make_pair(

343  break;
344  }
345  case Handle::Type::bucket:
346  // Remove element from bucket (as with ordinary linked-list)...
347  HotQueueNode<V, P> *queueNode = std::get<1>(handle.bucketHandle);
348  if(queueNode->next != nullptr) {
349  queueNode->next->prev = queueNode->prev;
350  }
351  if(queueNode->prev != nullptr) {
352  queueNode->prev->next = queueNode->next;
353  } else {
354  m_buckets[std::get<0>(handle.bucketHandle)] = queueNode->next;
355  }
356 
357  // ... and push it again with new priority, releasing the memory.
358  m_size--;
359  handle = push(queueNode->value, priority);
360  delete queueNode;
361  break;
362  }
363 }
