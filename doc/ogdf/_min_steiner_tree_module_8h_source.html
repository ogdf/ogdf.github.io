389 T MinSteinerTreeModule<T>::call(const EdgeWeightedGraph<T> &G,
390  const List<node> &terminals,
391  const NodeArray<bool> &isTerminal,
392  EdgeWeightedGraphCopy<T> *&finalSteinerTree
393  )
394 {
395  OGDF_ASSERT(isConnected(G));
396 
397  if (terminals.size() > 2) {
398  return this->computeSteinerTree(G, terminals, isTerminal, finalSteinerTree);
399  }
400 
401  finalSteinerTree = new EdgeWeightedGraphCopy<T>();
402  finalSteinerTree->createEmpty(G);
403  if (!terminals.empty()) {
404  finalSteinerTree->newNode(terminals.back());
405  }
406  if (terminals.size() <= 1) {
407  return 0;
408  }
409 
410  OGDF_ASSERT(terminals.size() == 2);
411  T cost(0);
412  AStarSearch<T> astar;
413  NodeArray<edge> pred;
414  NodeArray<T> dist;
415  astar.call(G, G.edgeWeights(), terminals.front(), terminals.back(), pred);
416  OGDF_ASSERT(pred[terminals.back()] != nullptr); // connected
417  for (node t = terminals.back(); t != terminals.front(); t = pred[t]->opposite(t)) {
418  const edge e = pred[t];
419  finalSteinerTree->newNode(e->opposite(t));
420  finalSteinerTree->newEdge(e, G.weight(e));
421  cost += G.weight(e);
422  }
423  return cost;

513 template<typename T>
514 T MinSteinerTreeModule<T>::removeCyclesFrom(EdgeWeightedGraphCopy<T> &steinerTree, const NodeArray<bool> &isTerminal)
515 {
516  if (steinerTree.numberOfEdges() > steinerTree.numberOfNodes() - 1) {
517  EdgeArray<bool> isInTree(steinerTree);
518  T oldCost(0);
519  T newCost(computeMinST(steinerTree, steinerTree.edgeWeights(), isInTree));
520 
521  List<node> pendant; // collect resulting pendant edges
522  for (edge nextEdge, e = steinerTree.firstEdge(); e; e = nextEdge) {
523  oldCost += steinerTree.weight(e);
524  nextEdge = e->succ();
525  if (!isInTree[e]) {
526  if (e->source()->degree() == 2) {
527  pendant.pushBack(e->source());
528  }
529  if (e->target()->degree() == 2) {
530  pendant.pushBack(e->target());
531  }
532  steinerTree.delEdge(e);
533  }
534  }
535  newCost -= MinSteinerTreeModule<T>::pruneDanglingSteinerPathsFrom(steinerTree, isTerminal, pendant);
536  return oldCost - newCost;
537  }
538  return 0;

554 template<typename T>
555 void MinSteinerTreeModule<T>::singleSourceShortestPathsPreferringTerminals(const EdgeWeightedGraph<T> &G, node source,
556  const NodeArray<bool> &isTerminal, NodeArray<T> &distance, NodeArray<edge> &pred)
557 {
558  PrioritizedMapQueue<node, T> queue(G);
559  ssspInit(G, source, queue, distance, pred);
560 
561  // we must handle the source explicitly because it is a terminal
562  node v = queue.topElement();
563  queue.pop();
564  OGDF_ASSERT(v == source);
565  for (adjEntry adj : v->adjEntries) {
566  edge e = adj->theEdge();
567  node w = adj->twinNode();
568  if (distance[w] > G.weight(e)) { // this check is only necessary for multigraphs, otherwise this is always true
569  queue.decrease(w, (distance[w] = G.weight(e)));
570  pred[w] = e;
571  }
572  }
573 
574  auto setPredecessor = [&](node w, edge e) {
575  bool wOnPathWithTerminal = isTerminal[v] || pred[v] == nullptr;
576  pred[w] = wOnPathWithTerminal ? nullptr : e;
577  };
578  while (!queue.empty()) {
579  v = queue.topElement();
580  queue.pop();
581 
582  if (distance[v] == std::numeric_limits<T>::max()) { // min is unreachable, finished
583  break;
584  }
585  for (adjEntry adj : v->adjEntries) {
586  edge e = adj->theEdge();
587  node w = adj->twinNode();
588  T dist = distance[v] + G.weight(e);
589  if (distance[w] > dist) {
590  queue.decrease(w, (distance[w] = dist));
591  setPredecessor(w, e);
592  } else
593  if (distance[w] == dist
594  && pred[w] != nullptr) { // tie
595  setPredecessor(w, e);
596  }
597  }
598  }
599 }

601 template<typename T>
602 void MinSteinerTreeModule<T>::allPairShortestPathsPreferringTerminals(const EdgeWeightedGraph<T>& G,
603  const NodeArray<bool>& isTerminal, NodeArray<NodeArray<T>>& distance, NodeArray<NodeArray<edge>>& pred) {
604  apspInit(G, distance, pred);
605 
606  for (node v : G.nodes) {
607  if (isTerminal[v]) { // v is a terminal
608  apspInnerLoop(v, G, distance, [&distance, &pred](node u, node w, T duvw) {
609  if (duvw <= distance[u][w]) { // prefer terminals
610  distance[w][u] = distance[u][w] = duvw;
611  pred[w][u] = pred[u][w] = nullptr;
612  }
613  });
614  } else { // v is not a terminal
615  apspInnerLoop(v, G, distance, [&v, &distance, &pred](node u, node w, T duvw) {
616  if (duvw < distance[u][w]) { // do not prefer nonterminals
617  distance[w][u] = distance[u][w] = duvw;
618  pred[u][w] = (pred[u][v] ? pred[v][w] : nullptr);
619  pred[w][u] = (pred[w][v] ? pred[v][u] : nullptr);
620  }
621  });
622  }
623  }
624  for (node u : G.nodes) {
625  distance[u][u] = 0;
626  }
627 }
