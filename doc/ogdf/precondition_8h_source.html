123 bool dfsGenTree(
124  UMLGraph& UG,
125  List<edge>& fakedGens,
126  bool fakeTree)
127 {
128  edge e;
129  EdgeArray<bool> used(UG.constGraph(), false);
130  //NodeArray<bool> visited(UG,false);
131  NodeArray<int> hierNumber(UG.constGraph(), 0);
132 
133  int hierNum = 0; //number of hierarchy tree
134 
135  const Graph& G = UG.constGraph();
136  forall_edges(e, G)
137  {
138  //descent in the hierarchy containing e
139  if ((!used[e]) && (UG.type(e) == Graph::generalization))
140  {
141  hierNum++; //current hierarchy tree
142  //first we search for the sink
143  node sink = e->target();
144  edge sinkPath = firstOutGen(UG, e->target(), used);
145  int cycleCounter = 0;
146  while (sinkPath)
147  {
148  sink = sinkPath->target();
149  sinkPath = firstOutGen(UG, sinkPath->target(), used);
150  cycleCounter++;
151  //if there is no sink, convert Generalizations to Associations and draw
152  if (cycleCounter > G.numberOfEdges())
153  {
154  UG.type(sinkPath) = Graph::association;
155  fakedGens.pushBack(sinkPath);
156  sink = sinkPath->source();
157  sinkPath = 0;
158  }
159  }
160 
161  //now sink is the hierarchy sink
162 
163  //used is set in dfsGenTreeRec
164  bool isTree = dfsGenTreeRec(UG, used, hierNumber, hierNum, sink, fakedGens, fakeTree);
165  if (!isTree) return false;
166  }
167 
168  }//forall_edges
169 
170  return true;
