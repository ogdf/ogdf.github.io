318 inline void FibonacciHeap<T, C>::compress()
319 {
320  size_t maxr = 0;
321 
322  for(auto it = m_knot->next; it != m_knot;) {

324

326  maxr = std::max(r, maxr);
327  while(m_ranked[r]) {

398 template<typename T, typename C>
399 inline void FibonacciHeap<T, C>::restore(FibonacciHeapNode<T> *heapNode)
400 {
401  for(;;) {
402  FibonacciHeapNode<T> *parent = heapNode->parent;
403  if(parent == nullptr) {
404  return;
405  }
406 
407  // We need to make sure parent has valid children after the splice.
408  parent->rank--;
409  if(parent->rank == 0) {

411  } else if(parent->child == heapNode) {

413  }
414

416  splice(m_knot, heapNode);
417 
418  // If parent is unmarked we can stop cut cascade.
