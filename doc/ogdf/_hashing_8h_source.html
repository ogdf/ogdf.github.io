318  HashElement<K,I> *lookup(const K &key) const {
319  HashElement<K,I> *pElement =
320  (HashElement<K,I> *)firstListElement(m_hashFunc.hash(key));
321  for (; pElement; pElement = pElement->next())
322  if (pElement->key() == key) return pElement;
323 
324  return 0;
