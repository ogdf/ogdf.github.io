209 if (a == nullptr) {
210 return b;
211 }
212 if (b == nullptr) {
213 return a;
214 }
215
216 if (this->comparator()(a->value, b->value)) {
217 if (m_rand() % 2 == 0) {
218 a->left = merge(a->left, b);
219 if (a->left != nullptr) {
220 a->left->parent = a;
221 }
222 } else {
223 a->right = merge(a->right, b);
224 if (a->right != nullptr) {
225 a->right->parent = a;

228 return a;
229 } else {
