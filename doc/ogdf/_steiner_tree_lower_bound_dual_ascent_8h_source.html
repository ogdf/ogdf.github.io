85  bool addNode(ListIterator<TerminalData>& it, node t) {
86  if (t == m_root) {
87  return false;
88  }
89  TerminalData& td = *it;
90  td.inCut[t] = true;
91  td.references.push({t, m_inTerminalCut[t].pushBack(it)});
92  for (adjEntry adj : t->adjEntries) {
93  node w = adj->twinNode();
94  if (!td.inCut[w]) {
95  // not in cut, so check (recursively) if we should add nodes
96  if (m_eps.equal(m_reducedCost[adj], T(0))) {
97  if (!addNode(it, w)) { // recurse
98  return false;
99  }
100  } else {
101  td.cutIterators[adj] = td.cut.pushBack(adj);
102  }
103  }
104  }
105 
106  // delete arcs that are inside the cut
107  for (adjEntry adj : t->adjEntries) {
108  node w = adj->twinNode();
109  if (td.inCut[w]) {
110  auto& cutAdjIt = td.cutIterators[adj->twin()];
111  if (cutAdjIt != td.cut.end()) {
112  td.cut.del(cutAdjIt);
113  cutAdjIt = nullptr;
114  }
115  }
116  }
117 
118  return true;
119  }

161  T findCheapestCutArcCost(const TerminalData& td) const {
162  OGDF_ASSERT(!td.cut.empty());
163  T cost = std::numeric_limits<T>::max();

172  void update(TerminalData& td, T delta) {
173  // reduce costs
174  ArrayBuffer<adjEntry> zeroed;
175  for (adjEntry adj : td.cut) {
176  m_reducedCost[adj] -= delta;
177  OGDF_ASSERT(m_eps.geq(m_reducedCost[adj], T(0)));
178  if (m_eps.leq(m_reducedCost[adj], T(0))) {
179  zeroed.push(adj);
180  }
181  }
182  m_lower += delta;
183 
184  // extend
185  for (adjEntry adj : zeroed) {
186  extendCut(adj);
187  }
188  }

263  void compute(const EdgeWeightedGraph<T>& graph, const List<node>& terminals, node root, NodeArray<T>& lbNodes, EdgeArray<T>& lbEdges) {
264  OGDF_ASSERT(isConnected(graph));
265 
266  // compute first
267  steiner_tree::LowerBoundDualAscent<T> alg(graph, terminals, root);
268  alg.compute();
269 
270  // generate the auxiliary network
271  Graph network;
272  NodeArray<node> copy(graph);
273  EdgeArray<T> weights(network);
274  EdgeArray<edge> orig(network);
275 
276  for (node v : graph.nodes) {
277  copy[v] = network.newNode();
278  }
279  for (edge e : graph.edges) {
280  edge uv = network.newEdge(copy[e->source()], copy[e->target()]);
281  edge vu = network.newEdge(copy[e->target()], copy[e->source()]);
282  weights[uv] = alg.reducedCost(e->adjTarget());
283  weights[vu] = alg.reducedCost(e->adjSource());
284  orig[uv] = orig[vu] = e;
285  }
286 
287  // compute shortest path tree on network starting from root
288  Dijkstra<T> sssp;
289  NodeArray<edge> pred;
290  NodeArray<T> distance;
291  sssp.call(network, weights, copy[root], pred, distance, true);
292 
293  // set all lower bounds to global lower bound
294  lbNodes.init(graph, alg.get());
295  lbEdges.init(graph, alg.get());
296  EdgeArray<T> lbArcs(network, alg.get());
297 
298  // add cost of path root -> v
299  for (node v : graph.nodes) {
300  lbNodes[v] += distance[copy[v]];
301  }
302  // add cost of path root -> e
303  for (edge a : network.edges) {
304  lbArcs[a] += distance[a->source()] + weights[a];
305  }
306 
307  // to find the (reduced) distance from v / e to any (non-root) terminal,
308  // hence compute (directed) Voronoi regions
309  network.reverseAllEdges();
310 
311  List<node> nonRootTerminals;
312  for (node t : terminals) {
313  if (t != root) {
314  nonRootTerminals.pushBack(copy[t]);
315  }
316  }
317  sssp.call(network, weights, nonRootTerminals, pred, distance, true);
318 
319  // add cost of path v -> any terminal
320  for (node v : graph.nodes) {
321  lbNodes[v] += distance[copy[v]];
322  }
323  // add cost of path e -> any terminal
324  for (edge a : network.edges) {
325  lbArcs[a] += distance[a->source()]; // the former target is now the source
326 
327  // both lower bounds must be larger than the upper bound, so take the minimum
328  Math::updateMin(lbEdges[orig[a]], lbArcs[a]);
329  }
330  }

334  void setRepetitions(int num) {
335  m_repetitions = num;

339  T call(const EdgeWeightedGraph<T>& graph, const List<node>& terminals) {
340  T lb(0);
341  int i = 0;
342  for (node root : terminals) {
343  Math::updateMax(lb, compute(graph, terminals, root));
344  if (++i >= m_repetitions) {
345  break;
346  }
347  }
348  return lb;
349  }

355  void call(const EdgeWeightedGraph<T>& graph, const List<node>& terminals, NodeArray<T>& lbNodes, EdgeArray<T>& lbEdges) {
356  if (m_repetitions <= 1) {
357  // catch this special case to avoid copying
358  compute(graph, terminals, terminals.front(), lbNodes, lbEdges);
359  } else {
360  lbNodes.init(graph, 0);
361  lbEdges.init(graph, 0);
362  int i = 0;
363  for (node root : terminals) {
364  NodeArray<T> nodes;
365  EdgeArray<T> edges;
366  compute(graph, terminals, root, nodes, edges);
367  for (node v : graph.nodes) {
368  Math::updateMax(lbNodes[v], nodes[v]);
369  }
370  for (edge e : graph.edges) {
371  Math::updateMax(lbEdges[e], edges[e]);
372  }
373  if (++i >= m_repetitions) {
374  break;
375  }
