108  virtual double computeCIndex(const Graph &G, node v)
109  {
110  OGDF_ASSERT(v->graphOf() == &G);
111  if (v->degree()<2) return 1.0;
112  int conns = 0; //connections, without v
113  NodeArray<bool> neighbor(G, false);
114  adjEntry adjE;
115  forall_adj(adjE, v)
116  {
117  neighbor[adjE->twinNode()] = true;
118  }
119  forall_adj(adjE, v)
120  {
121  adjEntry adjEE;
122  forall_adj(adjEE, adjE->twinNode())
123  {
124  if (neighbor[adjEE->twinNode()])
125  conns++;
126  }
127  }
128  //connections were counted twice
129  double index = conns / 2.0;
130  return index / (v->degree()*(v->degree()-1));
131  }
