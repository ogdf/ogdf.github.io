60 //---------------------------------------------------------
61 // ClusterSetSimple
62 // maintains a subset S of the clusters contained in an associated
63 // cluster graph G (only insertion of elements and clear operation)
64 //---------------------------------------------------------
65 class OGDF_EXPORT ClusterSetSimple {
66 public:
67  // creates a new empty cluster set associated with cluster graph CG
68  ClusterSetSimple(const ClusterGraph &CG) : m_isContained(CG,false) { }
69 
70  // destructor
71  ~ClusterSetSimple() { }
72 
73  // inserts cluster c into set S
74  // running time: O(1)
75  // Precond.: c is a cluster in the associated graph
76  void insert(cluster c) {
77  OGDF_ASSERT(c->graphOf() == m_isContained.graphOf());
78  bool &isContained = m_isContained[c];
79  if (isContained == false) {
80  isContained = true;
81  m_clusters.pushFront(c);
82  }
83  }
84 
85 
86  // removes all clusters from set S
87  // running time: O(|S|)
88  void clear() {
89  SListIterator<cluster> it;
90  for(it = m_clusters.begin(); it.valid(); ++it) {
91  m_isContained[*it] = false;
92  }
93  m_clusters.clear();
94  }
95 
96 
97  // returns true iff cluster c is contained in S
98  // running time: O(1)
99  // Precond.: c is a cluster in the asociated graph
100  bool isMember(cluster c) const {
101  OGDF_ASSERT(c->graphOf() == m_isContained.graphOf());
102  return m_isContained[c];
103  }
104 
105  // returns the list of clusters contained in S
106  const SListPure<cluster> &clusters() const {
107  return m_clusters;
108  }
109 
110 private:
111  // m_isContained[c] is true <=> c is contained in S
112  ClusterArray<bool> m_isContained;
113  // list of clusters contained in S
114  SListPure<cluster> m_clusters;
115 };
