285 template<class T,class Y>
286 bool MaxSequencePQTree<T,Y>::Bubble(SListPure<PQLeafKey<T,whaInfo*,Y>*> &leafKeys)
287 {
304  // queue for storing all pertinent nodes that still have
305  // to be processed.
306  Queue<PQNode<T,whaInfo*,Y>*> processNodes;
307 
308  /*
309  Enter the [[Full]] leaves into the queue [[processNodes]].
310  In a first step the pertinent leaves have to be identified in the tree
311  and entered on to the queue [[processNodes]]. The identification of
312  the leaves can be done with the help of a pointer stored in every
313  [[PQLeafKey]] (see \ref{PQLeafKey}) in constant time for every element.
314  */
315  SListIterator<PQLeafKey<T,whaInfo*,Y>*> it;
316  for (it = leafKeys.begin(); it.valid(); ++it)
317  {
318  PQNode<T,whaInfo*,Y>* checkLeaf = (*it)->nodePointer();
319  processNodes.append(checkLeaf);
320  cleanUp.pushBack(checkLeaf);
321  if (!checkLeaf->getNodeInfo()) // if leaf does not have an information
322  // class for storing the [wha]-number
323  // allocate one.
324  {
325  whaInfo *newInfo = OGDF_NEW whaInfo;
326  PQNodeKey<T,whaInfo*,Y> *infoPtr = OGDF_NEW PQNodeKey<T,whaInfo*,Y>(newInfo);
327  checkLeaf->setNodeInfo(infoPtr);
328  infoPtr->setNodePointer(checkLeaf);
329  }
330  checkLeaf->getNodeInfo()->userStructInfo()->m_notVisitedCount = 1;
331  checkLeaf->mark(PQNodeRoot::QUEUED);
332  }
333 
334 
335 
336  /*
337  For every node in [[processNodes]], its father is detected using the
338  function [[GetParent]] \ref{GetParent}. The father is placed onto the
339  queue as well, if [[_nodePtr]] is its first child, that is popped from
340  the queue. In that case, the father is marked as [[QUEUED]] to
341  prevent the node from queue more than once. In any case, the number
342  of pertinent children of the father is updated. This is an important
343  number for computing the maximal pertinent sequence.
344  */
345  while (!processNodes.empty())
346  {
347  PQNode<T,whaInfo*,Y>* nodePtr = processNodes.pop();
348  nodePtr->parent(GetParent(nodePtr));
349  if (nodePtr->parent() &&
350  !nodePtr->parent()->getNodeInfo())
351  // if the parent does not have an information
352  // class for storing the [wha]-number
353  // allocate one.
354  {
355  whaInfo *newInfo = OGDF_NEW whaInfo;
356  PQNodeKey<T,whaInfo*,Y> *infoPtr = OGDF_NEW PQNodeKey<T,whaInfo*,Y>(newInfo);
357  nodePtr->parent()->setNodeInfo(infoPtr);
358  infoPtr->setNodePointer(nodePtr->parent());
359  }
360  if (nodePtr != this->m_root)
361  {
362  if (nodePtr->parent()->mark() == PQNodeRoot::UNMARKED)
363  {
364  processNodes.append(nodePtr->parent());
365  cleanUp.pushBack(nodePtr->parent());
366  nodePtr->parent()->mark(PQNodeRoot::QUEUED);
367  }
368  nodePtr->parent()->getNodeInfo()->userStructInfo()->m_notVisitedCount++;
369  int childCount = nodePtr->parent()->pertChildCount();
370  nodePtr->parent()->pertChildCount(++childCount);
371  }
372  }
373 
374 
375  /*
376  This chunk belongs to the function [[Bubble]]. After doing some
377  cleaning up and resetting the flags that have been left at the
378  pertinent nodes during the first bubble up, this chunk computes the
379  maximal pertinent sequence by calling the function [[determineMinRemoveSequence]].
380  It then removes
381  all leaves from the tree that have been marked as [[ELIMINATED]] and
382  possibly some internal nodes of the $PQ$-tree and stores pointers of
383  type [[leafKey]] for the pertinent leaves in the maximal sequence in the
384  array [[survivedElements]].
385  */
386 
387  SListIterator<PQNode<T,whaInfo*,Y>*> itn;
388  for (itn = cleanUp.begin(); itn.valid(); itn++)
389  (*itn)->mark(PQNodeRoot::UNMARKED);
390 
391  return true;
392 }

545 template<class T,class Y>
546 int MaxSequencePQTree<T,Y>::determineMinRemoveSequence(
547  SListPure<PQLeafKey<T,whaInfo*,Y>*> &leafKeys,
548  SList<PQLeafKey<T,whaInfo*,Y>*> &eliminatedKeys)
549 {
576  PQNode<T,whaInfo*,Y> *nodePtr = 0; // dummy
577  PQNode<T,whaInfo*,Y> *checkLeaf = 0; // dummy
578 
579  //Number of leaves that have to be deleted
580  int countDeletedLeaves = 0;
581  //Number of pertinent leaves
582  int maxPertLeafCount = 0;
583 
584 
585  // A queue storing the nodes whose $[w,h,a]$-number has to be computed next.
586  // A node is stored in [[processNodes]], if for all of its children the
587  // [w,h,a]-number has been computed.
588  Queue<PQNode<T,whaInfo*,Y>*> processNodes;
589 
590  // A stack storing all nodes where a $[w,h,a]$-number
591  // has been computed. This is necessary for a valid cleanup.
592  StackPure<PQNode<T,whaInfo*,Y>*> archiv;
593 
594 
595 
596 
597 
598  // Compute a valid parent pointer for every pertinent node.
599  Bubble(leafKeys);
600 
601  // Get all pertinent leaves and stores them on [[processNodes]]
602  // and [[_archiv]].
603  SListIterator<PQLeafKey<T,whaInfo*,Y>*> it;
604  for (it = leafKeys.begin(); it.valid(); ++it)
605  {
606  checkLeaf = (*it)->nodePointer();
607  checkLeaf->getNodeInfo()->userStructInfo()->m_pertLeafCount = 1;
608  checkLeaf->getNodeInfo()->userStructInfo()->m_notVisitedCount--;
609  processNodes.append(checkLeaf);
610  archiv.push(checkLeaf);
611 
612  maxPertLeafCount++;
613  }
614 
615  while (!processNodes.empty())
616  {
617  nodePtr = processNodes.pop();
618  // Compute the $[w,h,a]$ number of [[nodePtr]]. Computing this number is
619  // trivial for leaves and full nodes. When considering a partial node, the
620  // computation has to distinguish between $P$- and $Q$-nodes.
621  if (nodePtr->getNodeInfo()->userStructInfo()->m_pertLeafCount < maxPertLeafCount)
622  {
623  // In case that the actual node, depicted by the pointer
624  // [[nodePtr]] is not the root, the counts of the pertinent
625  // children of [[nodePtr]]'s parent are
626  // updated. In case that all pertinent children of the parent of
627  // [[nodePtr]] have a valid $[w,h,a]$-number, it is possible to compute
628  // the $[w,h,a]$-number of parent. Hence the parent is placed onto
629  // [[processNodes]]and [[_archiv]].
630  nodePtr->parent()->getNodeInfo()->userStructInfo()->m_pertLeafCount =
631  nodePtr->parent()->getNodeInfo()->userStructInfo()->m_pertLeafCount
632  + nodePtr->getNodeInfo()->userStructInfo()->m_pertLeafCount;
633  nodePtr->parent()->getNodeInfo()->userStructInfo()->m_notVisitedCount--;
634  if (!nodePtr->parent()->getNodeInfo()->userStructInfo()->m_notVisitedCount)
635  {
636  processNodes.append(nodePtr->parent());
637  archiv.push(nodePtr->parent());
638  }
639  }
640  if (nodePtr->type() == PQNodeRoot::leaf)
641  {
642  // Compute the $[w,h,a]$-number of a leaf. The computation is trivial.
643  nodePtr->status(PQNodeRoot::FULL);
644  nodePtr->getNodeInfo()->userStructInfo()->m_w = 1;
645  nodePtr->getNodeInfo()->userStructInfo()->m_h = 0;
646  nodePtr->getNodeInfo()->userStructInfo()->m_a = 0;
647  if (nodePtr->getNodeInfo()->userStructInfo()->m_pertLeafCount < maxPertLeafCount)
648  fullChildren(nodePtr->parent())->pushFront(nodePtr);
649  }
650  else
651  {
652  // [[nodePtr]] is a $P$- or $Q$-node
653  // The $[w,h,a]$ number of $P$- or $Q$-nodes is computed identically.
654  // This is done calling the function [[sumPertChild]].
655  nodePtr->getNodeInfo()->userStructInfo()->m_w = sumPertChild(nodePtr);
656 
657  if (fullChildren(nodePtr)->size() == nodePtr->childCount())
658  {
659  // computes the $h$- and $a$-numbers of a full node. The computation
660  // is trivial. It also updates the list of full nodes of the parent
661  // of [[nodePtr]].
662  nodePtr->status(PQNodeRoot::FULL);
663  if (nodePtr->getNodeInfo()->userStructInfo()->m_pertLeafCount
664  < maxPertLeafCount)
665  fullChildren(nodePtr->parent())->pushFront(nodePtr);
666  nodePtr->getNodeInfo()->userStructInfo()->m_h = 0;
667  nodePtr->getNodeInfo()->userStructInfo()->m_a = 0;
668  }
669  else
670  {
671  // computes the $[w,h,a]$-number of a partial node. The computation is
672  // nontrivial for both $P$- and $Q$-nodes and is performed in the
673  // function [[haNumPnode]] for $P$-nodes and in the
674  // function [[haNumQnode]] for $Q$-nodes.
675  // This chunk also updates the partial children stack of the parent.
676  nodePtr->status(PQNodeRoot::PARTIAL);
677  if (nodePtr->getNodeInfo()->userStructInfo()->m_pertLeafCount <
678  maxPertLeafCount)
679  partialChildren(nodePtr->parent())->pushFront(nodePtr);
680 
681  if (nodePtr->type() == PQNodeRoot::PNode)
682  haNumPnode(nodePtr);
683  else
684  haNumQnode(nodePtr);
685  }
686  }
687 
688  }
689 
690  // Determine the root of the pertinent subtree, which the last processed node
691  //[[nodePtr]] and finds the minimum of the $h$- and $a$-number of
692  //[[m_pertinentRoot]]. In case that the minimum is equal to $0$, the
693  //[[m_pertinentRoot]] stays of type $B$. Otherwise the type is selected
694  //according to the minimum.
695  this->m_pertinentRoot = nodePtr;
696  if (this->m_pertinentRoot->getNodeInfo()->userStructInfo()->m_h <
697  this->m_pertinentRoot->getNodeInfo()->userStructInfo()->m_a)
698  countDeletedLeaves = this->m_pertinentRoot->getNodeInfo()->userStructInfo()->m_h;
699  else
700  countDeletedLeaves = this->m_pertinentRoot->getNodeInfo()->userStructInfo()->m_a;
701 
702  if (countDeletedLeaves > 0)
703  {
704  if (this->m_pertinentRoot->getNodeInfo()->userStructInfo()->m_h <
705  this->m_pertinentRoot->getNodeInfo()->userStructInfo()->m_a)
706  this->m_pertinentRoot->getNodeInfo()->userStructInfo()->m_deleteType = H_TYPE;
707  else
708  this->m_pertinentRoot->getNodeInfo()->userStructInfo()->m_deleteType = A_TYPE;
709  }
710 
711  findMinWHASequence(archiv,eliminatedKeys);
712 
713  return countDeletedLeaves;
714 }

722 template<class T,class Y>
723 void MaxSequencePQTree<T,Y>::
724 findMinWHASequence(StackPure<PQNode<T,whaInfo*,Y>*> &archiv,
725  SList<PQLeafKey<T,whaInfo*,Y>*> &eliminatedKeys)
726 {
748  //a pointer to a pertinent node of the $PQ$-tree that is currently examined.
749  PQNode<T,whaInfo*,Y> *nodePtr = 0;
750 
751  //a pointer to the first child of type $h$ of [[nodePtr]].
752  PQNode<T,whaInfo*,Y> *hChild1 = 0;
753 
754  //a pointer to the second child of type $h$ of [[nodePtr]].
755  PQNode<T,whaInfo*,Y> *hChild2 = 0;
756 
757  //a pointer to the child of type $a$ of [[nodePtr]].
758  PQNode<T,whaInfo*,Y> *aChild = 0;
759 
760  // pertinent sibling of hChild1
761  PQNode<T,whaInfo*,Y> *hChild2Sib = 0;
762 
763  //counts the number of children of [[nodePtr]].
764  int childCount = 0;
765 
766  while (!archiv.empty())
767  {
768  childCount = 0;
769  nodePtr = archiv.pop();
770  /*
771  Check if [[nodePtr]] is a full node whose delete type is either of
772  type $h$ or type $a$. Since there are no empty leaves in its
773  frontier, the node must therefore keep all its pertinent leaves
774  in its frontier and is depicted to be of type $b$.
775  */
776  if (nodePtr->status() == PQNodeRoot::FULL &&
777  (nodePtr->getNodeInfo()->userStructInfo()->m_deleteType == H_TYPE ||
778  nodePtr->getNodeInfo()->userStructInfo()->m_deleteType == A_TYPE))
779  {
780  nodePtr->getNodeInfo()->userStructInfo()->m_deleteType = B_TYPE;
781  this->m_pertinentNodes->pushFront(nodePtr);
782  }
783 
784  /*
785  Check if [[nodePtr]] is a leaf whose delete type is either of type $w$ or
786  $b$. In case it is of type $w$, the leaf has to be removed from the
787  tree to gain reducibility of the set $S$.
788  */
789  else if (nodePtr->type() == PQNodeRoot::leaf)
790  {
791  if (nodePtr->getNodeInfo()->userStructInfo()->m_deleteType == W_TYPE)
792  eliminatedKeys.pushBack(nodePtr->getKey());
793  else
794  this->m_pertinentNodes->pushFront(nodePtr);
795  }
796 
797  /*
798  Manage the case of [[nodePtr]] being either a partial $P$-node, a partial
799  $Q$-node, or a full $P$- or $Q$-node, where in the latter case the
800  delete type of [[nodePtr]] is of type $b$.
801  */
802  else
803  switch (nodePtr->getNodeInfo()->userStructInfo()->m_deleteType)
804  {
805  case B_TYPE:
806  this->m_pertinentNodes->pushFront(nodePtr);
807  break;
808 
809  case W_TYPE:
810  markPertinentChildren(nodePtr,PQNodeRoot::PERTINENT,W_TYPE);
811  nodePtr->pertChildCount(0);
812  this->m_pertinentNodes->pushFront(nodePtr);
813  break;
814 
815  case H_TYPE:
816  if (nodePtr->type() == PQNodeRoot::PNode)
817  {
818  /*
819  [[nodePtr]] is a $P$-node of type $h$. Mark all full
820  children of [[nodePtr]] to be of type $b$ (by doing nothing, since
821  the default is type $b$). It furthermore marks all partial children to
822  be of type $w$ except for the child stored in [[hChild1]] of the
823  information class of type [[whaInfo*]] of [[nodePtr]]. This child is
824  marked to be of type $h$.
825  */
826  markPertinentChildren(nodePtr,PQNodeRoot::PARTIAL,W_TYPE);
827  markPertinentChildren(nodePtr,PQNodeRoot::FULL,B_TYPE);
828  if (nodePtr->getNodeInfo()->userStructInfo()->m_hChild1)
829  {
830  hChild1 = (PQNode<T,whaInfo*,Y>*)
831  nodePtr->getNodeInfo()->userStructInfo()->m_hChild1;
832  hChild1->getNodeInfo()->userStructInfo()->m_deleteType = H_TYPE;
833  if (hChild1->getNodeInfo()->userStructInfo()->m_h <
834  hChild1->getNodeInfo()->userStructInfo()->m_w )
835  childCount = 1;
836  }
837  nodePtr->pertChildCount(nodePtr->pertChildCount() +
838  childCount - partialChildren(nodePtr)->size());
839  }
840  else
841  {
842  /*
843  [[nodePtr]] is a $Q$-node. Mark all pertinent children
844  to be of type $w$, except for the full children between the
845  [[hChild1]] and the endmost child of [[nodePtr]]. These full
846  children are marked $b$ while [[hChild1]] is marked to be of type
847  $h$. Setting the type of children to $b$ or $h$ is hidden in the
848  function call [[setHchild]] \label{setHChild}.
849  */
850  markPertinentChildren(nodePtr,PQNodeRoot::PERTINENT,W_TYPE);
851  hChild1 = (PQNode<T,whaInfo*,Y>*)
852  nodePtr->getNodeInfo()->userStructInfo()->m_hChild1;
853  nodePtr->pertChildCount(setHchild(hChild1));
854  }
855  this->m_pertinentNodes->pushFront(nodePtr);
856  break;
857 
858 
859  case A_TYPE:
860  if (nodePtr->type() == PQNodeRoot::PNode)
861  {
862  /*
863  [[nodePtr]] is a $P$-node of type $a$.
864  Distinguish two main cases, based on the existence of a
865  child of [[nodePtr]] stored in [[aChild]] of the information
866  class of type [[_whaInfo]] of [[nodePtr]].
867  \begin{enumerate}
868  \item
869  If [[aChild]] is not empty, the chunk marks all full
870  and partial children of [[nodePtr]] to be of type $w$ and
871  marks the child [[aChild]] to be of type $a$.
872  \item
873  If [[aChild]] is empty, the chunk
874  marks all full children of [[nodePtr]] to be of type $b$
875  (by doing nothing, since the default is type $b$).
876  It furthermore marks all partial children to be of type $w$
877  except for the children stored in [[hChild1]] and
878  [[hChild2]] of the information class of type [[whaInfo*]] of
879  [[nodePtr]] which are marked to be of type $h$. Observe that
880  we have to distinguish the cases where both, [[_hChild]] and
881  [[hChild2]] are available, just [[_h_Child1]] is available or
882  none of the nodes exist.
883  \end{enumerate}
884  */
885  if (nodePtr->getNodeInfo()->userStructInfo()->m_aChild)
886  {
887  markPertinentChildren(nodePtr,PQNodeRoot::PERTINENT,W_TYPE);
888  aChild = (PQNode<T,whaInfo*,Y>*)
889  nodePtr->getNodeInfo()->userStructInfo()->m_aChild;
890  aChild->getNodeInfo()->userStructInfo()->m_deleteType = A_TYPE;
891  nodePtr->pertChildCount(1);
892  }
893  else
894  {
895  markPertinentChildren(nodePtr,PQNodeRoot::FULL,B_TYPE);
896  markPertinentChildren(nodePtr,PQNodeRoot::PARTIAL,W_TYPE);
897  if (nodePtr->getNodeInfo()->userStructInfo()->m_hChild1)
898  {
899  hChild1 = (PQNode<T,whaInfo*,Y>*)
900  nodePtr->getNodeInfo()->userStructInfo()->m_hChild1;
901  hChild1->getNodeInfo()->userStructInfo()->m_deleteType = H_TYPE;
902  if (hChild1->getNodeInfo()->userStructInfo()->m_h <
903  hChild1->getNodeInfo()->userStructInfo()->m_w)
