190  bool finished(double maxdelta)
191  {
192  if (m_prevEnergy == startVal) //first step
193  {
194  m_prevEnergy = maxdelta;
195  return false;
196  }
197 
198  double diff = m_prevEnergy - maxdelta; //energy difference
199  if (diff < 0.0) diff = -diff;
200  //#ifdef OGDF_DEBUG
201  // cout << "Finished(): maxdelta: "<< maxdelta<<" diff/prev: "<<diff / m_prevEnergy<<"\n";
202  //#endif
203  //check if we want to stop
204  bool done = (maxdelta < m_tolerance);// || (diff / m_prevEnergy) < m_tolerance);
205 
206  m_prevEnergy = maxdelta; //save previous energy level
207  m_prevLEnergy = startVal; //reset energy level for local node decision
208 
209  return done;
210  }//finished
212  bool finishedNode(double deltav)
213  {
214  if (m_prevLEnergy == startVal)
215  {
216  m_prevLEnergy = deltav;
217  return deltav == 0.0;//<m_ltolerance; //locally stable
218  }
219  //#ifdef OGDF_DEBUG
220  // cout << "Local delta: "<<deltav<<"\n";
221  //#endif
222  double diff = m_prevLEnergy - deltav;
223  //check if we want to stop
224  bool done = (deltav == 0.0 || (diff / m_prevLEnergy) < m_ltolerance);
225 
226  m_prevLEnergy = deltav; //save previous energy level
227 
228  return done;
229  }//finishedNode
