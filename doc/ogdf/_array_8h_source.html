463  template<class COMPARER>
464  static void quicksortInt(E *pL, E *pR, const COMPARER &comp) {
465  size_t s = pR-pL;
466 
467  // use insertion sort for small instances
468  if (s < maxSizeInsertionSort) {
469  for (E *pI = pL+1; pI <= pR; pI++) {
470  E v = *pI;
471  E *pJ = pI;
472  while (--pJ >= pL && comp.less(v,*pJ)) {
473  *(pJ+1) = *pJ;
474  }
475  *(pJ+1) = v;
476  }
477  return;
478  }
479 
480  E *pI = pL, *pJ = pR;
481  E x = *(pL+(s>>1));
482 
483  do {
484  while (comp.less(*pI,x)) pI++;
485  while (comp.less(x,*pJ)) pJ--;
486  if (pI <= pJ) std::swap(*pI++,*pJ--);
487  } while (pI <= pJ);
488 
489  if (pL < pJ) quicksortInt(pL,pJ,comp);
490  if (pI < pR) quicksortInt(pI,pR,comp);
491  }
