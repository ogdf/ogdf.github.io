66 //---------------------------------------------------------
67 // GmlObject
68 // represents node in GML parse tree
69 //---------------------------------------------------------
70 struct OGDF_EXPORT GmlObject {
71  GmlObject *m_pBrother; // brother of node in tree
72  GmlKey m_key; // tag of node
73  GmlObjectType m_valueType; // type of node
74 
75  // the entry in the union is selected according to m_valueType:
76  // gmlIntValue -> m_intValue
77  // gmlDoubleValue -> m_doubleValue
78  // gmlStringValue -> m_stringValue
79  // gmlListBegin -> m_pFirstSon (in case of a list, m_pFirstSon is pointer
80  // to first son and the sons are chained by m_pBrother)
81  union {
82  int m_intValue;
83  double m_doubleValue;
84  const char *m_stringValue;
85  GmlObject *m_pFirstSon;
86  };
87 
88  // construction
89  GmlObject(GmlKey key, int intValue) : m_pBrother(0), m_key(key),
90  m_valueType(gmlIntValue), m_intValue(intValue) { }
91 
92  GmlObject(GmlKey key, double doubleValue) : m_pBrother(0), m_key(key),
93  m_valueType(gmlDoubleValue), m_doubleValue(doubleValue) { }
94 
95  GmlObject(GmlKey key, const char *stringValue) : m_pBrother(0), m_key(key),
96  m_valueType(gmlStringValue), m_stringValue(stringValue) { }
97 
98  GmlObject(GmlKey key) : m_pBrother(0), m_key(key),
99  m_valueType(gmlListBegin), m_pFirstSon(0) { }
100 
101  OGDF_NEW_DELETE
102 };
