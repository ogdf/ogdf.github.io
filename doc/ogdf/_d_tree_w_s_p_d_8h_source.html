235 void DTreeWSPD<Dim>::wspdRecursive(int curr)
236 {
237  // iterate over all ordered pairs of children
238  for (int i = 0; i < m_pTree->numChilds(curr); i++) {
239  // the first child index
240  const int first_child = m_pTree->child(curr, i);
241 
242  // the second loop for the pair
243  for (int j = i+1; j < m_pTree->numChilds(curr); j++) {
244  // the second child index
245  const int second_child = m_pTree->child(curr, j);
246 
247  // call for each ordered pair the binary function
248  wspdRecursive(first_child, second_child);
249  }
250 
251  // now do all this for every child
252  wspdRecursive(first_child);
253  }

256 template<int Dim>
257 void DTreeWSPD<Dim>::wspdRecursive(int a, int b)
258 {
259  if (areWellSeparated(a, b)) {
260  // far enough away => approx
261  if (m_pIWSPD)
262  m_pIWSPD->onWellSeparatedPair(a, b);
263  } else {
264  // two cells are too close
265  int small_node = a;
266  int large_node = b;
267 
268  // make sure the small one is not the bigger one
269  if (node(small_node).radius_sq > node(large_node).radius_sq) {
270  std::swap(small_node, large_node);
271  }
272 
273  // split the bigger one
274  for (int i = 0; i < tree().numChilds(large_node); ++i) {
275  // recurse on the child
276  wspdRecursive(small_node, tree().child(large_node, i));
277  }
278  }

281 template<int Dim>
282 bool DTreeWSPD<Dim>::areWellSeparated(int a, int b) const
283 {
284  // take the bigger radius
285  double r_max_sq = std::max(node(a).radius_sq, node(b).radius_sq);
286 
287  // compute the square distance
288  double dist_sq = 0.0;
289 
290  // the d^2 loop
291  for (int d = 0; d < Dim; d++) {
292  double dx = node(a).x[d] - node(b).x[d];
293  dist_sq += dx * dx;
294  }
295 
296  // two circles are well separated iff d - 2r > sr
297  // where d is the distance between the two centers
298  // (not the distance of circles!)
299  // this ws <=> d > (s + 2)r
300  // more efficient: d^2 > (s + 2)^2 r^2
301  // d_sq > (s^2 + 4s + 4) * r_max
302 #if 0
303 # if 0
304  const double s = (m_wspdSeparationFactor + 2.0);
305  return dist_sq > (m_wspdSeparationFactor * m_wspdSeparationFactor + 4.0 * m_wspdSeparationFactor + 4) * r_max * r_max;
306 # else
307  return dist_sq > s * s * r_max_sq;
308 # endif
309 #else
310  return dist_sq > m_wspdSeparationFactorPlus2Squared_cached * r_max_sq;
311 #endif

416 template<int Dim>
417 void DTreeWSPD<Dim>::updateTreeNodeGeometry(int curr)
418 {
419  // if this is an inner node
420  if (m_pTree->numChilds(curr)) {
421  // init with the bbox of the first child
422 
423  // iterate over the rest of the children
424  for (int i = 0; i < m_pTree->numChilds(curr); i++) {
425  // the child index
426  const int child = m_pTree->child(curr, i);
427 
428  // compute the size of the subtree
429  updateTreeNodeGeometry(child);
430 
431  // lazy set the first
432  if (!i) {
433  for (int d = 0; d < Dim; d++) {
434  node(curr).min_x[d] = node(child).min_x[d];
435  node(curr).max_x[d] = node(child).max_x[d];
436  }
437  } else {
438  for (int d = 0; d < Dim; d++) {
439  node(curr).min_x[d] = std::min(node(curr).min_x[d], node(child).min_x[d]);
440  node(curr).max_x[d] = std::max(node(curr).max_x[d], node(child).max_x[d]);
441  }
442  }
443  };
444  } else { // else it is a leaf
445  // init from first point
446  for (int d = 0; d < Dim; d++) {
447  node(curr).min_x[d] = node(curr).max_x[d] = point(tree().point(curr,0)).x[d];
448  }
449 
450  // and the remaining points
451  for (int i = 1; i < tree().numPoints(curr); ++i) {
452  for (int d = 0; d < Dim; d++) {
453  node(curr).min_x[d] = std::min(node(curr).min_x[d], point(tree().point(curr,i)).x[d]);
454  node(curr).max_x[d] = std::max(node(curr).max_x[d], point(tree().point(curr,i)).x[d]);
455  }
456  }
457  }
458 
459  // init the radius
460  node(curr).radius_sq = 0.0;
461 
462  // compute the center and radius of the rect
463  for (int d = 0; d < Dim; d++) {
464  node(curr).x[d] = (node(curr).min_x[d] + node(curr).max_x[d]) * 0.5;
465  double s_x = (node(curr).max_x[d] - node(curr).min_x[d]);
466  node(curr).radius_sq += s_x * s_x;
467  }
468 
469  // and the smallest enclosing circle radius squared
470  node(curr).radius_sq *= 0.25; // sqrt(node(curr).radius) * 0.5 squared;
471 }
