63  // Implements the Planar Subgraph interface.
64  // For the given graph \p G, a clustered graph with only
65  // a single root cluster is generated.
66  // Computes set of edges delEdges, which have to be deleted
67  // in order to get a planar subgraph; edges in preferredEdges
68  // should be contained in planar subgraph.
69  // Status: pCost and preferredEdges are ignored in current implementation.
70  virtual Module::ReturnType doCall(
71  const Graph &G,
72  const List<edge> &preferredEdges,
73  List<edge> &delEdges,
74  const EdgeArray<TCost> *pCost,
75  bool preferredImplyPlanar) override
76  {
77  if (G.numberOfEdges() < 9)
78  return Module::ReturnType::Optimal;
79 
80  //if the graph is planar, we don't have to do anything
81  if (isPlanar(G))
82  return Module::ReturnType::Optimal;
83 
84  //Exact ILP
85  MaximumCPlanarSubgraph mc;
86 
87  delEdges.clear();
88 
89  NodeArray<node> tableNodes(G,nullptr);
90  EdgeArray<edge> tableEdges(G,nullptr);
91  NodeArray<bool> mark(G,0);
92 
93  EdgeArray<int> componentID(G);
94 
95  // Determine biconnected components
96  int bcCount = biconnectedComponents(G,componentID);
97  OGDF_ASSERT(bcCount >= 1);
98 
99  // Determine edges per biconnected component
100  Array<SList<edge> > blockEdges(0,bcCount-1);
101  for(edge e : G.edges)
102  {
103  if (!e->isSelfLoop())
104  blockEdges[componentID[e]].pushFront(e);
105  }
106 
107  // Determine nodes per biconnected component.
108  Array<SList<node> > blockNodes(0,bcCount-1);
109  int i;
110  for (i = 0; i < bcCount; i++)
111  {
112  for (edge e : blockEdges[i])
113  {
114  if (!mark[e->source()])
115  {
116  blockNodes[i].pushBack(e->source());
117  mark[e->source()] = true;
118  }
119  if (!mark[e->target()])
120  {
121  blockNodes[i].pushBack(e->target());
122  mark[e->target()] = true;
123  }
124  }
125  for (node v : blockNodes[i])
126  mark[v] = false;
127  }
128 
129 
130  // Perform computation for every biconnected component
131  Module::ReturnType mr;
132  if (bcCount == 1) {
133  mr = callWithDouble(mc, G, pCost, delEdges);
134  }
135  else
136  {
137  for (i = 0; i < bcCount; i++)
138  {
139  Graph C;
140 
141  for (node v : blockNodes[i])
142  {
143  node w = C.newNode();
144  tableNodes[v] = w;
145  }
146 
147  EdgeArray<double> cost(C);
148 
149  for (edge e : blockEdges[i])
150  {
151  edge f = C.newEdge(tableNodes[e->source()],tableNodes[e->target()]);
152  tableEdges[e] = f;
153  if(pCost != nullptr) {
154  cost[tableEdges[e]] = (*pCost)[e];
155  }
156  }
157 
158  // Construct a translation table for the edges.
159  // Necessary, since edges are deleted in a new graph
160  // that represents the current biconnected component
161  // of the original graph.
162  EdgeArray<edge> backTableEdges(C,nullptr);
163  for (edge e : blockEdges[i])
164  backTableEdges[tableEdges[e]] = e;
165 
166  // The deleted edges of the biconnected component
167  List<edge> delEdgesOfBC;
168 
169  ClusterGraph CG(C);
170  mr = mc.call(CG, pCost == nullptr ? nullptr : &cost, delEdgesOfBC);
171 
172  // Abort if no optimal solution found, i.e., feasible is also not allowed
173  if (mr != Module::ReturnType::Optimal)
174  break;
175 
176  // Get the original edges that are deleted and
177  // put them on the list delEdges.
178  while (!delEdgesOfBC.empty())
179  delEdges.pushBack(backTableEdges[delEdgesOfBC.popFrontRet()]);
180 
181  }
182  }
183  return mr;
184  }

188  // Call algorithm with costs as double. All underlying algorithms cast the costs to double on use eventually.
189  // However, only convert the costs if they are not given as double already.
190  template<typename U = TCost>
191  Module::ReturnType callWithDouble(MaximumCPlanarSubgraph &mc, const Graph &G, const EdgeArray<U> *pCost, List<edge> &delEdges) {
192  if (pCost == nullptr) {
193  return callWithDouble(mc, G, nullptr, delEdges);
194  }
195  else {
196  EdgeArray<double> dCost(G);
197  for (auto it = pCost->begin(); it != pCost->end(); ++it) {
198  dCost[it.key()] = it.value();
199  }
200  return callWithDouble(mc, G, &dCost, delEdges);
201  }
202  }
