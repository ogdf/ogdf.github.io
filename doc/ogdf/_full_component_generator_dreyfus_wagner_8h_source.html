217  void computeSplit(NodeArray<DWMSplit> &split, node v, SubsetEnumerator<node> &subset) const {
218  if (split[v].subgraph1 != nullptr) { // already computed
219  return;
220  }
221 
222  DWMSplit& best = split[v];
223  for (subset.begin(1, subset.numberOfMembersAndNonmembers() / 2); subset.valid(); subset.next()) {
224  List<node> newSubset, newComplement;
225  makeKey(newSubset, newComplement, subset, v);
226 
227  if (safeIfSumSmaller(costOf(newSubset), costOf(newComplement), best.cost)) {
228  best.set(dataOf(newSubset), dataOf(newComplement));
229  }
230  }

234  void computePartialSolution(NodeArray<DWMSplit> &split,
235  node v,
236  SubsetEnumerator<node> &subset,
237  const List<node> &terminals)
238  {
239  List<node> newSubset;
240  makeKey(newSubset, v);
241 
242  if (!m_map.member(newSubset)) { // not already defined
243  T oldCost = costOf(terminals);
244  DWMData best;
245  auto addPair = [&](node v1, node v2, T dist) {
246  best.add(NodePair(v1, v2), dist);
247  if (m_pred[v1][v2] == nullptr) {
248  best.invalidate();
249  }
250  };
251  for (node w : m_G.nodes) {
252  T dist = m_distance[v][w];
253  if (m_terminalSubset.hasMember(w)) {
254  // we attach edge vw to tree containing terminal w
255  if (safeIfSumSmaller(oldCost, dist, best.cost)) {
256  best.clear();
257  best.add(dataOf(terminals));
258  addPair(v, w, dist);
259  }
260  } else {
261  // we attach edge vw to tree split[w]
262  OGDF_ASSERT(!m_terminalSubset.hasMember(v));
263  computeSplit(split, w, subset);
264  if (safeIfSumSmaller(split[w].cost, dist, best.cost)) {
265  best.clear();
266  best.add(split[w].subgraph1);
267  best.add(split[w].subgraph2);
268  if (v != w) {
269  addPair(v, w, dist);
270  }
271  }
272  }
273  }
274  m_map.fastInsert(newSubset, best);
275  }

293  void initializeMap() {
294  for (node v : m_G.nodes) {
295  for (node t : m_terminals) {
296  if (t != v) {
297  List<node> key;
298  key.pushBack(t);
299  if (v->index() < t->index()) {
300  key.pushFront(v);
301  } else {
302  key.pushBack(v);
303  }
304 
305  if (!m_map.member(key)) { // not already defined
306  if (m_pred[t][v] == nullptr) {
307  m_map.fastInsert(key, DWMData(m_distance[t][v]));
308  OGDF_ASSERT(!dataOf(key)->valid() || m_distance[t][v] == 0);
309  } else {
310  NodePairs nodepairs;
311  nodepairs.push(NodePair(key.front(), key.back()));
312  m_map.fastInsert(key, DWMData(m_distance[t][v], nodepairs));
313  OGDF_ASSERT(dataOf(key)->valid());

322  T getSteinerTreeFor(const DWMData& data, EdgeWeightedGraphCopy<T>& tree) const {
323  T cost(0);
324  if (data.valid()) {
325  // add edges
326  for (auto nodepair : data.nodepairs) {
327  node uO = nodepair.source;
328  node vO = nodepair.target;
329  node uC = tree.copy(uO);
330  node vC = tree.copy(vO);
331  if (uC == nullptr) {
332  uC = tree.newNode(uO);
333  }
334  if (vC == nullptr) {
335  vC = tree.newNode(vO);
336  }
337 #ifdef OGDF_FULL_COMPONENT_GENERATION_TERMINAL_SSSP_AWARE
338  const T dist = m_isTerminal[uO] ? m_distance[uO][vO] : m_distance[vO][uO];
339 #else
340  const T dist = m_distance[uO][vO];
341 #endif
342  tree.newEdge(uC, vC, dist);
343  cost += dist;
344  }
345 
346  // recurse
347  for (const DWMData* subgraph : data.subgraphs) {
348  cost += getSteinerTreeFor(*subgraph, tree);
349  }
350  }
351 
352  OGDF_ASSERT(isAcyclicUndirected(tree));
353  return cost;
354  }

378  for (m_terminalSubset.begin(2, restricted-1); m_terminalSubset.valid(); m_terminalSubset.next()) {
