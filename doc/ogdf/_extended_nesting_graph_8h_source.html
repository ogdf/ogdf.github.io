100  if(m_cnClusters == cr.m_cnClusters)
101  return (m_cnEdges <= cr.m_cnEdges);
102  else
103  return (m_cnClusters <= cr.m_cnClusters);
104  }
105

107  if(m_cnClusters == cr.m_cnClusters)
108  return (m_cnEdges < cr.m_cnEdges);
109  else
110  return (m_cnClusters < cr.m_cnClusters);
111  }

115  }

120  }

123  {
