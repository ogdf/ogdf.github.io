157 inline double randomDoubleNormal(double m, double sd)
158 {
159  double x1, y1, w;
160 
161  do {
162  double rndVal = randomDouble(0,1);
163  x1 = 2.0 * rndVal - 1.0;
164  rndVal = randomDouble(0,1);
165  double x2 = 2.0 * rndVal - 1.0;
166  w = x1*x1 + x2*x2;
167  } while (w >= 1.0);
168 
169  w = sqrt((-2.0 * log(w))/w) ;
170  y1 = x1*w;
171 
172  return m + y1 * sd;
173 }
