165inline double randomDoubleNormal(double m, double sd) {
166 double x1, y1, w;
167
168 do {
169 double rndVal = randomDouble(0, 1);
170 x1 = 2.0 * rndVal - 1.0;
171 rndVal = randomDouble(0, 1);
172 double x2 = 2.0 * rndVal - 1.0;
173 w = x1 * x1 + x2 * x2;
174 } while (w >= 1.0);
175
176 w = sqrt((-2.0 * log(w)) / w);
177 y1 = x1 * w;
178
179 return m + y1 * sd;
180}
