248  inline int randomNumber(int low, int high) {
249 #if RAND_MAX == 32767
250  // We get only 15 random bits on some systems (Windows, Solaris)!
251  int r1 = (rand() & ((1 << 16) - 1));
252  int r2 = (rand() & ((1 << 16) - 1));
253  int r = (r1 << 15) | r2;
254 #else
255  int r = rand();
256 #endif
257  return low + (r % (high-low+1));
258  }

269  inline double randomDoubleNormal(double m, double sd)
270  {
271  double x1, y1, w;
272 
273  do {
274  double rndVal = randomDouble(0,1);
275  x1 = 2.0 * rndVal - 1.0;
276  rndVal = randomDouble(0,1);
277  double x2 = 2.0 * rndVal - 1.0;
278  w = x1*x1 + x2*x2;
279  } while (w >= 1.0);
280 
281  w = sqrt((-2.0 * log(w))/w) ;
282  y1 = x1*w;
283 
284  return(m + y1*sd);
285  }
