231  void findTriangle(GraphCopy& copy, edge currentEdge, const EdgeArray<TCost> *pCost,
232  DisjointSets<>& components, NodeArray<int>& set,
233  std::function<bool(node, edge, edge)> callback) {
234  node source = currentEdge->source();
235  node target = currentEdge->target();
236  auto sourceIt = source->adjEntries.begin();
237  auto targetIt = target->adjEntries.begin();
238  int sourceSet = components.find(set[source]);
239  int targetSet = components.find(set[target]);
240  // Our nodes cannot be in the same set.
241  if (sourceSet == targetSet) return;
242 
243  while (sourceIt != source->adjEntries.end() && targetIt != target->adjEntries.end()) {
244  if ((*sourceIt)->theEdge() == currentEdge) {
245  sourceIt++;
246  continue; // re-check loop condition
247  }
248  if ((*targetIt)->theEdge() == currentEdge) {
249  targetIt++;
250  continue; // re-check loop condition
251  }
252 
253  // Look for a triangle, starting with the edge with next highest weight
254  // If no weights are given, it does not matter which edge we start with
255  adjEntry potentialConnector;
256  node potentialConnection;
257  internal::GraphIterator<adjEntry> potentialConnectionIterator;
258  if (pCost == nullptr || (*pCost)[copy.original((*sourceIt)->theEdge())] > (*pCost)[copy.original((*targetIt)->theEdge())]) {
259  potentialConnector = *sourceIt;
260  potentialConnection = target;
261  potentialConnectionIterator = targetIt;
262  sourceIt++;
263  }
264  else {
265  potentialConnector = *targetIt;
266  potentialConnection = source;
267  potentialConnectionIterator = sourceIt;
268  targetIt++;
269  }
270  // Note: sourceIt and targetIt may be invalid until next loop
271 
272  node potentialNode = potentialConnector->twinNode();
273  int potentialSet = components.find(set[potentialNode]);
274 
275  // Only use this edge if it doesn't connect back to one of the components
276  if (potentialSet != sourceSet && potentialSet != targetSet) {
277  edge potentialEdge = searchEdge(potentialNode, potentialConnection, potentialConnectionIterator);
278  if (potentialEdge != nullptr) {
279  // We found a triangle.
280  // If our callback returns true, it's signalling that we're done and don't
281  // want to look for another triangle.
282  // If it returns false, we continue looking.
283  if (callback(potentialNode, potentialEdge, potentialConnector->theEdge())) {
