289  OGDF_ASSERT(v->graphOf() == &m_G);

371  edge addNewPath(DWMData& result, node curr, const NodeArray<edge>& pred) const {
372  edge e = nullptr;
373  while (curr != m_auxG.source()) {
374  e = pred[curr];
375  OGDF_ASSERT(e != nullptr);
376  node prev = e->opposite(curr);
377  OGDF_ASSERT(prev != curr);
378  if (prev != m_auxG.source()) {
379  edge eO = m_auxG.original(e);
380  OGDF_ASSERT(eO != nullptr);

382  }
383  curr = prev;
384  }

441  T getSteinerTreeFor(const DWMData& data, EdgeWeightedGraphCopy<T>& tree) const {
442  T cost(0);
443 
444  if (data.valid()) {
445  // add edges
446  for (edge e : data.edges) {
447  node uO = e->source();
448  node vO = e->target();
449  OGDF_ASSERT(uO != nullptr);
450  OGDF_ASSERT(vO != nullptr);
451  node uC = tree.copy(uO);
452  node vC = tree.copy(vO);
453  if (uC == nullptr) {
454  uC = tree.newNode(uO);
455  }
456  if (vC == nullptr) {
457  vC = tree.newNode(vO);
458  }
459  T dist = m_G.weight(e);
460  tree.newEdge(e, dist);
461  cost += dist;
462  }
463 
464  // recurse
465  for (const DWMData* subgraph : data.subgraphs) {
466  cost += getSteinerTreeFor(*subgraph, tree);
467  }
468 
469  OGDF_ASSERT(isAcyclicUndirected(tree));
470  }
471 
472  return cost;
473  }
