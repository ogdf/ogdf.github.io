79  /* witness set (for some component and specified K (core edge set))
80  *
81  * W(e) = { e } if e is a core edge
82  * W(e) =
83  * take the path P of loss edges from u to e in the component
84  * add every incident core edge of u
85  * that is: if we root the loss connected component at the terminal,
86  * W(e) = all core edges incident to the subtree below e
87  *
88  * Need fast lookups for:
89  * (1) |W(f)|
90  * (2) all loss edges f such that given core edge e is in W(f)
91  * (3) all loss edges f such that W(f) is a subset of a given basis -> for each e \in B: do (2)
92  *
93  * Data Structures:
94  * - witnessCard[e] = |W(e)|
95  * - witness[v_e] = { f | e \in W(f) }
96  * (Note that core edges are given as nodes.)
97  * Also note that we can save it all much sparser.
98  */
99  EdgeArray<int> m_witnessCard;
100  NodeArray<ArrayBuffer<edge>> m_witness;

146  node v = m_fullCompStore.original(start->theNode());
147  List<adjEntry> queueT;
148  List<node> queueC;
149  queueT.pushBack(start);
150  queueC.pushBack(copy[v]);
151  while (!queueT.empty()) {
152  const adjEntry inAdj = queueT.popFrontRet();
153  const node wT = inAdj->twinNode();
154  const node vC = queueC.popFrontRet();
155 
156  const node wO = m_fullCompStore.original(wT);
157  if (m_fullCompStore.isTerminal(wT)) {
158  newEdge(vC, copy[wO], m_fullCompStore.graph().weight(inAdj->theEdge()), cap);

161  newEdge(vC, wC, m_fullCompStore.graph().weight(inAdj->theEdge()), cap);
162  const adjEntry back = inAdj->twin();
163  for (adjEntry adj = back->cyclicSucc(); adj != back; adj = adj->cyclicSucc()) {
164  queueT.pushBack(adj);
165  queueC.pushBack(wC);

371  void initCoreWitness() {
372  m_witnessCard.init(m_graph, 0);
373  m_witness.init(m_graph);
374 
375  // compute set of core edges
376  EdgeArray<bool> isLossEdge;
377  m_ceModule.call(m_graph, terminals(), isLossEdge);
378 
379  // add nodes for core edges and be able to map them
380  EdgeArray<node> splitMap(m_graph, nullptr);
381  ArrayBuffer<edge> coreEdges;
382  for (edge e : m_graph.edges) {
383  if (!isLossEdge[e]) {
384  splitMap[e] = m_graph.newNode();
385  coreEdges.push(e);
386  }
387  }
388 
389  // traverse losses from all terminals to find witness sets
390  NodeArray<adjEntry> pred(m_graph, nullptr);
391  for (node t : terminals()) {
392  ArrayBuffer<node> stack;
393  stack.push(t);
394  while (!stack.empty()) {
395  // for each node v "below" an edge e in the traversal:
396  // add all incident core edges vw to W(e)
397  const node v = stack.popRet();
398  for (adjEntry adj : v->adjEntries) {
399  const edge e = adj->theEdge();
400  const node w = adj->twinNode();
401  if (!pred[v] || w != pred[v]->theNode()) { // do not look at backward arcs in the tree
402  if (isLossEdge[e]) {
403  stack.push(w);
404  pred[w] = adj;
405  } else {
406  for (node x = v; pred[x]; x = pred[x]->theNode()) {
407  addWitness(splitMap[e], pred[x]->theEdge());
408  }
409  }
410  }
411  }
412  }
413  }
414 
415  // finally replace core edges by nodes
416  for (edge e : coreEdges) {
417  const T w = getCost(e);
418  const node x = splitMap[e];
419  const int cap = getCapacity(e);
420  OGDF_ASSERT(x);
421  newEdge(e->source(), x, w, cap);
422  newEdge(x, e->target(), 0, cap);
423  // the cost of a core edge node is hence the weight of one incident edge;
424  // also keep capacity.
425  // Note that we cannot guarantee throughout the algorithm that the edge
426  // with the non-zero cost is the first one nor that it is the incoming one
427  // because both directions and adjacency orders can be changed.
428  m_graph.delEdge(e);
429  addCore(x);
430  }
431  }

735  List<node> origin;
736  HashArray<edge,edge> edgeMap;
737  todo.pushBack(origEdge);
738  origin.pushBack(origEdge->source());
739  while (!todo.empty()) {
740  edge eO = todo.popFrontRet();
741  node vC = origin.popFrontRet();
742  node wO = eO->target();
743  node wC = wO;
744  if (!isTerminal(wO)) {
745  wC = initNode(getOriginal(wO));
746  }
747  edge eC = newEdge(vC, wC, getCost(eO), copyCap);
748  setCapacity(eO, origCap);
749  edgeMap[eO] = eC;
750  if (!isTerminal(wO)) {
751  for (adjEntry adj = eO->adjTarget()->cyclicSucc(); adj != eO->adjTarget(); adj = adj->cyclicSucc()) {
752  OGDF_ASSERT(adj->theEdge()->target() != eO->target()); // outgoing edges
753  origin.pushBack(wC);
754  todo.pushBack(adj->theEdge());
755  }
756  }
757  }
758  makeCWCopy(edgeMap);
759  }
760 
763 
765  const List<node> &core() const {
766  return m_coreEdges;
767  }
768 
771  void delCore(node e) {
772  /* What happens when we remove a core edge?
773  * - loss edges are not affected
774  * - we have to remove a core edge e from W(f) for all f, which means:
775  * for all elements f of witness[v_e], decrease witnessCard[f], then remove witness[v_e]
776  */
777  for (edge f : m_witness[e]) {
778  --m_witnessCard[f];
779  }
780  // witness[e] is removed by removing the node from the graph
781  m_coreEdges.removeFirst(e);
782  }
783 
785  int numberOfWitnesses(edge e) const {
786  return m_witnessCard[e];
787  }
788 
790  const ArrayBuffer<edge> &witnessList(node e) const {
791  return m_witness[e];
792  }

795 };
