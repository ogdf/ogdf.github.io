1132 template<typename T>
1133 void SteinerTreePreprocessing<T>::computeClosestKTerminals(const int k, NodeArray<List<std::pair<node,T>>> &closestTerminals) const
1134 {
1135  using NodePairQueue = PrioritizedQueue<NodePair, T>;
1136 
1137  closestTerminals.init(m_copyGraph);
1138  NodePairQueue queue;
1139  std::unordered_map<NodePair, typename NodePairQueue::Handle, steiner_tree::UnorderedNodePairHasher, steiner_tree::UnorderedNodePairEquality> qpos;
1140 
1141  // initialization
1142  for (node v : m_copyTerminals) {
1143  closestTerminals[v].pushBack(std::make_pair(v, 0));
1144  qpos[NodePair(v, v)] = queue.push(NodePair(v, v), closestTerminals[v].front().second);
1145  }
1146 
1147  auto getCurrentDist = [&closestTerminals](const node currentNode, const node sourceTerminal) {
1148  for (std::pair<node,T> currentPair : closestTerminals[currentNode]) {
1149  if (currentPair.first == sourceTerminal) {
1150  return currentPair.second;
1151  }
1152  }
1153  return static_cast<T>(-1);
1154  };
1155 
1156  auto setNewDist = [&closestTerminals, k](const node currentNode, const node sourceTerminal, const T newDist) {
1157  List<std::pair<node, T>> &currentList = closestTerminals[currentNode];
1158 
1159  // delete the old distance
1160  for (ListIterator<std::pair<node, T>> it = currentList.begin(); it.valid(); ++it) {
1161  if ((*it).first == sourceTerminal) {
1162  currentList.del(it);
1163  break;
1164  }
1165  }
1166 
1167  if (currentList.size() == k) { // the list is full
1168  currentList.popBack(); // delete the largest cost element
1169  }
1170 
1171  // add the new distance such that the list remains sorted
1172  if (currentList.empty()
1173  || (*currentList.begin()).second >= newDist) { // if the list is empty
1174  currentList.pushFront(std::make_pair(sourceTerminal, newDist));
1175  return;
1176  }
1177 
1178  ListIterator<std::pair<node, T>> it = closestTerminals[currentNode].begin();
1179  while (it.succ() != closestTerminals[currentNode].end()
1180  && (*it.succ()).second < newDist) {
1181  ++it;
1182  }
1183  closestTerminals[currentNode].insertAfter(std::make_pair(sourceTerminal, newDist), it);
1184  };
1185 
1186  while (!queue.empty()) {
1187  NodePair minDistPair = queue.topElement();
1188  queue.pop();
1189  node currentNode = minDistPair.source;
1190  node sourceTerminal = minDistPair.target;
1191 
1192  T currentDist = getCurrentDist(currentNode, sourceTerminal);
1193  if (currentDist == -1) { // source terminal not found
1194  continue; // check if the current path needs to be expanded
1195  }
1196 
1197  for(adjEntry adj : currentNode->adjEntries) {
1198  edge e = adj->theEdge();
1199  node adjacentNode = e->opposite(currentNode);
1200 
1201  if (m_copyIsTerminal[adjacentNode]) {
1202  continue;
1203  }
1204 
1205  T possibleNewDistance = currentDist + m_copyGraph.weight(e);
1206 
1207  // try to insert the new path from sourceTerminal to adjacentNode
1208  T currentDistToAdjacentNode = getCurrentDist(adjacentNode, sourceTerminal);
1209  if (currentDistToAdjacentNode != -1) { // there is already one path from sourceTerminal to adjNode
1210  if (possibleNewDistance < currentDistToAdjacentNode) {
1211  queue.decrease(qpos[NodePair(adjacentNode, sourceTerminal)], possibleNewDistance);
1212  setNewDist(adjacentNode, sourceTerminal, possibleNewDistance);
1213  }
1214  } else {
1215  if (closestTerminals[adjacentNode].size() < k
1216  || closestTerminals[adjacentNode].back().second > possibleNewDistance) {
1217  qpos[NodePair(adjacentNode, sourceTerminal)] = queue.push(NodePair(adjacentNode, sourceTerminal), possibleNewDistance);
1218  setNewDist(adjacentNode, sourceTerminal, possibleNewDistance);
1219  }
1220  }
1221  }
1222  }
1223 }

1357 template<typename T>
1358 void SteinerTreePreprocessing<T>::findTwoMinimumCostEdges(node v, edge& first, edge& second) const {

1360  * in nearestVertexTest(): Imagine a triangle of three terminals where the three edges
1361  * have all equal costs. It can happen that each terminal chooses its first edge to
1362  * be unique, so in the end we add all three edges of the triangle except only two of it.
1363  * We tackle this problem at *this* stage by choosing the edge with smallest index
1364  * in case of a tie in the costs. It is sufficient to do that for the first only
1365  * because this is the edge that will be put into the solution. */
1366  for (adjEntry adj : v->adjEntries) {
1367  edge e = adj->theEdge();

1369  || m_eps.less(m_copyGraph.weight(e), m_copyGraph.weight(first)) // smaller cost or
1370  || (m_eps.equal(m_copyGraph.weight(e), m_copyGraph.weight(first)) // cost tie but...
1371  && e->index() < first->index())) { // cost tie but smaller index
1372  second = first;
1373  first = e;
1374  } else {
1375  if (second == nullptr
1376  || m_eps.less(m_copyGraph.weight(e), m_copyGraph.weight(second))) {
1377  second = e;
1378  }
1379  }
1380  }
1381  OGDF_ASSERT(first != second);

1578  computeClosestKTerminals(3, closestTerminals);
1579

1581  // as explained in [PV01, page 278, Observation 3.5]

1584  if (m_copyIsTerminal[v]) {

1600  EdgeArray<T> lowerBoundWithEdge(m_copyGraph, 0);
1601  for (edge e : m_copyGraph.edges) {
1602  node x = e->source(), y = e->target();
1603

1605  T distanceToClosestTerminalY = (*closestTerminals[y].begin()).second;

1610  // Update the lowerbound of the cost of a Steiner tree containing one particular node
1611  // as explained in [PV01, pages 279-280, Observation 3.8]
1612  Graph auxiliaryGraph;
1613  NodeArray<node> terminalInAuxiliaryGraph(m_copyGraph, nullptr);
1614  for (node terminal : m_copyTerminals) {
1615  node newAuxiliaryNode = auxiliaryGraph.newNode();

1617  }

1620  for (edge e : m_copyGraph.edges) {
1621  initialEdgeWeight[e] = m_copyGraph.weight(e);
1622  }
1623  Voronoi<T> voronoiRegions(m_copyGraph, initialEdgeWeight, m_copyTerminals);
1624 
1625  std::unordered_map<NodePair, edge, steiner_tree::UnorderedNodePairHasher, steiner_tree::UnorderedNodePairEquality> edgeBetweenNodes;
1626  EdgeArray<T> edgeWeight(auxiliaryGraph, std::numeric_limits<T>::max());
1627  for (edge e : m_copyGraph.edges) {
1628  node x = e->source(), y = e->target();
1629  node seedX = voronoiRegions.seed(x), seedY = voronoiRegions.seed(y);
1630  if (seedX == seedY) {
1631  continue;
1632  }
1633

1635  if (edgeBetweenNodes.find(pair) == edgeBetweenNodes.end()) {
1636  edgeBetweenNodes[pair] = auxiliaryGraph.newEdge(terminalInAuxiliaryGraph[seedX], terminalInAuxiliaryGraph[seedY]);
1637  }
1638  edge auxiliaryEdge = edgeBetweenNodes[pair];
1639  Math::updateMin(edgeWeight[auxiliaryEdge], min(voronoiRegions.distance(x), voronoiRegions.distance(y)) + m_copyGraph.weight(e));
1640  }
1641

1643  T minimumSpanningTreeCost = computeMinST(auxiliaryGraph, edgeWeight, isInTree);
1644  T longestEdgeCost = std::numeric_limits<T>::lowest();

1646  if (isInTree[e]) {
1647  Math::updateMax(longestEdgeCost, edgeWeight[e]);
1648  }
1649  }
1650 
1651  for (node v : m_copyGraph.nodes) {
1652  if (m_copyIsTerminal[v] || closestTerminals[v].size() < 2) {
1653  continue;
1654  }
1655 
1656  std::pair<node, T> closestTerminalPair1 = *(closestTerminals[v].get(0)), closestTerminalPair2 = *(closestTerminals[v].get(1));
1657  T distanceToClosestTerminal1 = closestTerminalPair1.second, distanceToClosestTerminal2 = closestTerminalPair2.second;
1658 
1659  Math::updateMax(lowerBoundWithNode[v], minimumSpanningTreeCost - longestEdgeCost + distanceToClosestTerminal1 + distanceToClosestTerminal2);
1660  }
1661 
1662  return deleteNodesAboveUpperBound(lowerBoundWithNode, upperBound)
1663  || deleteEdgesAboveUpperBound(lowerBoundWithEdge, upperBound);
1664 }
