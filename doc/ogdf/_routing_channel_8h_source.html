96  void computeRoutingChannels(const OrthoRep &OR, bool align = false)
97  {
98  const Graph &G = OR;
99 
100  node v;
101  forall_nodes(v,G)
102  {
103  const OrthoRep::VertexInfoUML *pInfo = OR.cageInfo(v);
104 
105  if (pInfo) {
106  const OrthoRep::SideInfoUML &sNorth = pInfo->m_side[odNorth];
107  const OrthoRep::SideInfoUML &sSouth = pInfo->m_side[odSouth];
108  const OrthoRep::SideInfoUML &sWest = pInfo->m_side[odWest];
109  const OrthoRep::SideInfoUML &sEast = pInfo->m_side[odEast];
110 
111  (*this)(v,odNorth) = computeRoutingChannel(sNorth,sSouth,align);
112  (*this)(v,odSouth) = computeRoutingChannel(sSouth,sNorth,align);
113  (*this)(v,odWest ) = computeRoutingChannel(sWest ,sEast ,align);
114  (*this)(v,odEast ) = computeRoutingChannel(sEast ,sWest ,align);
115  }
116  }
117  }

119 private:
120  // computes required size of routing channel at side si with opposite side siOpp
121  int computeRoutingChannel(
122  const OrthoRep::SideInfoUML &si,
123  const OrthoRep::SideInfoUML &siOpp,
124  bool align = false)
125  {
126  if (si.m_adjGen == 0)
127  {
128  int k = si.m_nAttached[0];
129  if (k == 0 ||
130  ((k == 1 && siOpp.totalAttached() == 0) && !align) )
131  return 0;
132  else
133  return (k+1)*m_separation;
134 
135  } else {
136  int m = max(si.m_nAttached[0],si.m_nAttached[1]);
137  if (m == 0)
138  return 0;
139  else
140  return (m+1)*m_separation;
141  }
142  }
