47template<typename TWeight>
48class SpannerBermanDisconnected : public SpannerModule<TWeight> {
49public:
51 virtual bool preconditionsOk(const GraphAttributes& GA, double stretch,
52 std::string& error) override {
53 if (!isSimple(GA.constGraph())) {
54 error = "The graph is not simple";
55 return false;
56 }
57 if (stretch < 1.0) {
58 error = "The stretch must be >= 1.0";
59 return false;
60 }
61 return true;
62 }
63
64private:
66 virtual typename SpannerModule<TWeight>::ReturnType execute() override {
67 const Graph& G = m_GA->constGraph();
68
69 if (G.numberOfNodes() == 0) {
70 return SpannerModule<TWeight>::ReturnType::Feasible;
71 }
72
73 NodeArray<int> components(G);
74 int amountComponenets = connectedComponents(G, components);
75
76 SpannerBerman<TWeight> spannerBerman;
77
78 for (int c = 0; c < amountComponenets; c++) {
79 assertTimeLeft();
80
81 List<node> nodes;
82 for (node n : G.nodes) {
83 if (components[n] == c) {
84 nodes.pushBack(n);
85 }
86 }
87
88 GraphCopySimple GC;
89 GC.createEmpty(G);
90 inducedSubGraph(G, nodes.begin(), GC);
91
92 GraphCopySimple spanner(GC);
93 EdgeArray<bool> inSpanner;
94 GraphAttributes GCA(GC, 0);
95 GCA.directed() = m_GA->directed();
96 if (m_GA->has(GraphAttributes::edgeIntWeight)) {
97 GCA.addAttributes(GraphAttributes::edgeIntWeight);
98 for (edge e : GC.edges) {
99 GCA.intWeight(e) = m_GA->intWeight(GC.original(e));
100 }
101 }
102 if (m_GA->has(GraphAttributes::edgeDoubleWeight)) {
103 GCA.addAttributes(GraphAttributes::edgeDoubleWeight);
104 for (edge e : GC.edges) {
105 GCA.doubleWeight(e) = m_GA->doubleWeight(GC.original(e));
106 }
107 }
108
109 if (isTimelimitEnabled()) {
110 int64_t timeLeft = max(getTimeLeft(), static_cast<int64_t>(0));
111 spannerBerman.setTimelimit(timeLeft);
112 }
113 typename SpannerModule<TWeight>::ReturnType r =
114 spannerBerman.call(GCA, m_stretch, spanner, inSpanner);
115 if (r != SpannerModule<TWeight>::ReturnType::Feasible) {
116 return r;
117 }
118
119 // copy used edges to m_spanner and m_inSpanner
120 for (edge e : spanner.edges) {
121 edge eOrig = GC.original(spanner.original(e));
122 (*m_inSpanner)[eOrig] = true;
123 m_spanner->newEdge(eOrig);
124 }
125 }
126
127 return SpannerModule<TWeight>::ReturnType::Feasible;
128 }
129
130 using SpannerModule<TWeight>::getWeight;
131 using SpannerModule<TWeight>::assertTimeLeft;
132 using SpannerModule<TWeight>::getTimeLeft;
133 using SpannerModule<TWeight>::isTimelimitEnabled;
134 using SpannerModule<TWeight>::m_GA;
135 using SpannerModule<TWeight>::m_stretch;
136 using SpannerModule<TWeight>::m_spanner;
137 using SpannerModule<TWeight>::m_inSpanner;
138};
