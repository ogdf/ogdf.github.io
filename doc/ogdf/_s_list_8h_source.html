932 // computes l and h value
933 template<class E>
934 void SListPure<E>::bucketSort(BucketFunc<E> &f)
935 {
936  // if less than two elements, nothing to do
937  if (m_head == m_tail) return;
938 
939  int l, h;
940  l = h = f.getBucket(m_head->m_x);
941 
942  SListElement<E> *pX;
943  for(pX = m_head->m_next; pX; pX = pX->m_next)
944  {
945  int i = f.getBucket(pX->m_x);
946  if (i < l) l = i;
947  if (i > h) h = i;
948  }
949 
950  bucketSort(l,h,f);
951 }

954 // sorts list L using bucket sort
955 template<class E>
956 void SListPure<E>::bucketSort(int l, int h, BucketFunc<E> &f)
957 {
958  // if less than two elements, nothing to do
959  if (m_head == m_tail) return;
960 
961  Array<SListElement<E> *> head(l,h,0), tail(l,h);
962 
963  SListElement<E> *pX;
964  for (pX = m_head; pX; pX = pX->m_next) {
965  int i = f.getBucket(pX->m_x);
966  if (head[i])
967  tail[i] = (tail[i]->m_next = pX);
968  else
969  head[i] = tail[i] = pX;
970  }
971 
972  SListElement<E> *pY = 0;
973  for (int i = l; i <= h; i++) {
974  pX = head[i];
975  if (pX) {
976  if (pY)
977  pY->m_next = pX;
978  else
979  m_head = pX;
980  pY = tail[i];
981  }
982  }
983 
984  m_tail = pY;
985  pY->m_next = 0;
986 }

990 template<class E>
991 void SListPure<E>::permute(const int n)
992 {
993  Array<SListElement<E> *> A(n+1);
994  A[n] = 0;
995 
996  int i = 0;
997  SListElement<E> *pX;
998  for (pX = m_head; pX; pX = pX->m_next)
999  A[i++] = pX;

1001  A.permute(0,n-1);
1002 
1003  for (i = 0; i < n; i++) {
