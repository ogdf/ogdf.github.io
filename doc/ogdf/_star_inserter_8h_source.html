108  node findLCAInInsertionPathTree(node w1, node w2, edge &parentOfLCA) const {
109  const NodeArray<edge> &preds1 {*m_predecessors.at(w1)};
110  const NodeArray<edge> &preds2 {*m_predecessors.at(w2)};
111 
112  // Go backwards from root until predecessor path diverges.
113  node lastDualNode1 {m_rootDualNode};
114  node lastDualNode2 {m_rootDualNode};
115  edge edgeToChild {nullptr};
116  node lca {nullptr};
117  while (lastDualNode1 == lastDualNode2) {
118  // Remember current lca.
119  parentOfLCA = edgeToChild;
120  lca = lastDualNode1;
121 
122  // Go further down.
123  edgeToChild = preds1[lastDualNode1];
124  if (preds1[lastDualNode1] == nullptr ||
125  preds2[lastDualNode2] == nullptr) {
126  // If either node has no child in the insertion path tree,
127  // the lca cannot be lower.
128  break;
129  } else {
130  lastDualNode1 = preds1[lastDualNode1]->opposite(lastDualNode1);
131  lastDualNode2 = preds2[lastDualNode2]->opposite(lastDualNode2);
132  }
133  }
134 
135  return lca;
136  }

144  int compare(const edge &e1, const edge &e2) const {
145  node w1 {m_graphCopy.copy(e1->opposite(m_origNode))};
146  node w2 {m_graphCopy.copy(e2->opposite(m_origNode))};
147  if (w1 == w2) {
148  return 0;
149  }
150 
151  // Find lowest common ancestor in predecessor tree.
152  edge parentOfLCA {nullptr};
153  node lcaDualNode {findLCAInInsertionPathTree(w1, w2, parentOfLCA)};
154 
155  // w1 and w2 must have a common ancestor.
156  OGDF_ASSERT(lcaDualNode != nullptr);
157 
158  // Get the adjEntry of the primal edge of parentOfLCA which is incident
159  // to the face corresponding to lcaDualNode.
160  // This adjEntry marks the start of the cyclic order around lcaDualNode.
161  face lcaFace {m_dualGraph->primalFace(lcaDualNode)};
162  adjEntry parentAdj {nullptr};
163  if (parentOfLCA == nullptr) {
164  OGDF_ASSERT(lcaDualNode == m_rootDualNode);
165  // The lca is the root of the insertion path tree, hence it has no
166  // predecessor. Just use any adjEntry to start the cyclic order.
167  parentAdj = lcaFace->firstAdj();
168  } else {
169  // The lca is not the root of the insertion path tree.
170  // Determine the correct adjEntry of the primal of its parent-edge.
171  edge primalParentOfLCA {m_dualGraph->primalEdge(parentOfLCA)};
172  adjEntry sourceAdj {primalParentOfLCA->adjSource()};
173  adjEntry targetAdj {primalParentOfLCA->adjTarget()};
174 
175  const ConstCombinatorialEmbedding &embedding =
176  m_dualGraph->getPrimalEmbedding();
177  if (embedding.rightFace(sourceAdj) == lcaFace) {
178  parentAdj = sourceAdj;
179  } else {
180  OGDF_ASSERT(embedding.rightFace(targetAdj) == lcaFace);
181  parentAdj = targetAdj;
182  }
183  }
184 
185  // Traverse adjEntries of face corrsponding to lcaDualNode clockwise,
186  // starting at parent edge of lcaDualNode in the insertion path tree.
187  auto cyclicNextAdj = [&parentAdj](adjEntry adj) {
188  adjEntry nextAdj = adj->faceCycleSucc();
189  return nextAdj == parentAdj ? nullptr : nextAdj;
190  };
191 
192  edge pred1 {(*m_predecessors.at(w1))[lcaDualNode]};
193  edge pred2 {(*m_predecessors.at(w2))[lcaDualNode]};
194  for (adjEntry adj {parentAdj}; adj != nullptr; adj = cyclicNextAdj(adj)) {
195  edge primalEdge {adj->theEdge()};
196  node primalNode {adj->theNode()};
197  // First, if lcaFace has no pred in the insertion path of w1/w2,
198  // w1/w2 is adjacent to lcaFace.
199  // Check whether the node of the current adj is w1/w2.
200  if (pred1 == nullptr && primalNode == w1) {
201  return -1;
202  }
203  if (pred2 == nullptr && primalNode == w2) {
204  return 1;
205  }
206 
207  // If w1/w2 is not adjacent to lcaFace, check whether the insertion
208  // path from lcaFace to the face adjacent to w1/w2 crosses the
209  // current edge.
210  edge dualEdge = m_dualGraph->dualEdge(primalEdge);
211  if (dualEdge == pred1) {
212  OGDF_ASSERT(dualEdge != pred2);
213  return -1;
214  }
215  if (dualEdge == pred2) {
216  OGDF_ASSERT(dualEdge != pred1);
217  return 1;
218  }
219  }
220 
221  // Something went terribly wrong: The LCA of the insertion paths of w1
222  // and w2 is not actually part of the insertion paths of w1 and w2?
223  OGDF_ASSERT(false);
224  return 0;
225  }
