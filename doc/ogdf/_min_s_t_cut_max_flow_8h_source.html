213  void markCut(node startNode, bool frontCut, std::function<node(node)> origNode) {
214  List<node> queue;
215  queue.pushBack(startNode);
216  m_nodeSet[origNode(startNode)] = (frontCut ? cutType::FRONT_CUT : cutType::BACK_CUT);
217  frontCut ? m_frontCutCount++ : m_backCutCount++;
218 
219  while (!queue.empty()) {
220  const node v = queue.popFrontRet();
221  for (adjEntry adj : v->adjEntries) {
222  const node w = adj->twinNode();
223  const edge e = adj->theEdge();
224  if (m_nodeSet[origNode(w)] == cutType::NO_CUT
225  && (((frontCut ? e->source() :e->target()) == v &&
226  m_et->less(m_flow[e], m_weight[e]))
227  || ((frontCut ? e->target() : e->source()) == v &&
228  m_et->greater(m_flow[e], TCost(0))))) {
229  queue.pushBack(w);
230  m_nodeSet[origNode(w)] = (frontCut ? cutType::FRONT_CUT : cutType::BACK_CUT);
231  frontCut ? m_frontCutCount++ : m_backCutCount++;
232  }
233  }
234  }
235  }

247  void computeCut(const Graph &graph, std::function<edge(edge)> origEdge, std::function<node(node)> origNode,
248  const node source, const node target, List<edge> &edgeList) {
249  m_frontCutCount = 0;
250  m_backCutCount = 0;
251  m_totalCount = graph.numberOfNodes();
252 
253 
254  List<node> queue;
255  m_nodeSet.init(graph, cutType::NO_CUT);
256 
257  if(m_primaryCut || m_calculateOtherCut) {
258  // front cut
259  markCut(source, true, origNode);
260  }
261 
262  if(!m_primaryCut || m_calculateOtherCut) {
263  // back cut
264  markCut(target, false, origNode);
265  }
266 
267  ArrayBuffer<edge> stack;
268  EdgeArray<bool> visited(graph, false);
269  node startNode = (m_primaryCut ? source : target);
270  adjEntry startAdj = startNode->firstAdj();
271  if (startAdj == nullptr) {
272  return;
273  }
274  if (startAdj->theEdge()->adjTarget() != startAdj) {
275  stack.push(startAdj->theEdge());
276  }
277  for (adjEntry adj = (m_primaryCut ? startAdj->cyclicSucc() : startAdj->cyclicPred());
278  adj != startAdj;
279  adj = (m_primaryCut ? adj->cyclicSucc() : adj->cyclicPred()))
280  {
281  if (adj->theEdge()->adjTarget() != adj) {
282  stack.push(adj->theEdge());
283  }
284  }
285  while (!stack.empty()) {
286  edge e = stack.popRet();
287  if (visited[origEdge(e)]) {
288  continue;
289  }
290  visited[origEdge(e)] = true;
291  if (m_nodeSet[origNode(e->source())] == cutType::FRONT_CUT &&
292  m_nodeSet[origNode(e->target())] != cutType::FRONT_CUT) {
293  edgeList.pushBack(origEdge(e));
294 
295  if(m_gc->numberOfEdges() != 0) {
296  MinSTCutModule<TCost>::m_direction[origEdge(e)] = m_gc->copy(origEdge(e)) == e;
297  }
298  } else {
299  startAdj = e->adjTarget();
300  for (adjEntry adj = (m_primaryCut ? startAdj->cyclicSucc() : startAdj->cyclicPred());
301  adj != startAdj;
302  adj = (m_primaryCut ? adj->cyclicSucc() : adj->cyclicPred()))
303  {
304  if (adj->theEdge()->adjTarget() != adj) {
305  stack.push(adj->theEdge());
306  }
307  }
308  }
309  }
310  }

314 bool MinSTCutMaxFlow<TCost>::call(const Graph &graph, const EdgeArray<TCost> &weight,
315  node source, node target, List<edge> &edgeList, edge e_st) {
316  MinSTCutModule<TCost>::m_direction.init(graph, -1);
317  delete m_gc;
318  m_gc = new GraphCopy(graph);
319 
320  if (e_st != nullptr) {
321  m_gc->delEdge(m_gc->copy(e_st));
322  }
323  node s = m_gc->copy(source);
324  node t = m_gc->copy(target);
325  List<edge> edges;
326  m_gc->allEdges(edges);
327  EdgeArray<edge> originalEdge(*m_gc, nullptr);
328  for (edge e : edges) {
329  if (m_treatAsUndirected) {
330  // a reversed edge is made and placed directly next to e
331  edge revEdge = m_gc->newEdge(e->target(), e->source());
332  m_gc->move(revEdge, e->adjTarget(), ogdf::Direction::before, e->adjSource(), ogdf::Direction::after);
333  originalEdge[revEdge] = m_gc->original(e);
334  }
335  originalEdge[e] = m_gc->original(e);
336  }
337 
338  m_flow.init(*m_gc, -1);
339  m_weight.init(*m_gc, 1);
340  for (edge e : m_gc->edges) {
341  edge origEdge = originalEdge[e];
342  m_weight[e] = weight[origEdge];
343  OGDF_ASSERT(m_weight[e] >= 0);
344  }
345 
346  m_mfModule->init(*m_gc);
347  m_mfModule->computeFlow(m_weight, s, t, m_flow);
348 
349  computeCut(graph,
350  [&](edge e) -> edge {return originalEdge[e];},
351  [&](node v) -> node {return m_gc->original(v);},
352  s, t, edgeList);
353 
354  return true;
355 }
