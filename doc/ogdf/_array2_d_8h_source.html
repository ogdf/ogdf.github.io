183 template<class E>
184 void Array2D<E>::construct(int a, int b, int c, int d)
185 {
186  m_a = a;
187  m_b = b;
188  m_c = c;
189  m_d = d;
190 
191  int lenDim1 = b-a+1;
192  m_lenDim2 = d-c+1;
193 
194  if (lenDim1 < 1 || m_lenDim2 < 1) {
195  m_pStart = m_vpStart = m_pStop = 0;
196 
197  } else {
198  int len = lenDim1*m_lenDim2;
199  m_pStart = (E *)malloc(len*sizeof(E));
200  if (m_pStart == 0)
201  OGDF_THROW(InsufficientMemoryException);
202 
203  m_vpStart = m_pStart - c;
204  m_pStop = m_pStart + len;
205  }
206 }

267 template<class E>
268 float Array2D<E>::det() const
269 {
270  // matrix must be quadratic
271  OGDF_ASSERT(size1() == size2());
272 
273  int a = m_a;
274  int b = m_b;
275  int c = m_c;
276  int d = m_d;
277  int n = m_lenDim2;
278 
279  int i, j;
280  int rem_i, rem_j, column;
281 
282  float determinant = 0.0;
283 
284  switch(n) {
285  case 0:
286  break;
287  case 1:
288  determinant = (float)((*this)(a, c));
289  break;
290  case 2:
291  determinant = (float)((*this)(a, c) * (*this)(b, d) - (*this)(a, d) * (*this)(b, c));
292  break;
293 
294  // Expanding along the first row (Laplace's Formula)
295  default:
296  Array2D<E> remMatrix(0, n-2, 0, n-2); // the remaining matrix
297  for(column = c; column <= d; column++) {
298  rem_i = 0;
299  rem_j = 0;
300  for(i = a; i <= b; i++) {
301  for(j = c; j <= d; j++) {
302  if(i != a && j != column) {
303  remMatrix(rem_i, rem_j) = (*this)(i, j);
304  if(rem_j < n-2) {
305  rem_j++;
306  }
307  else {
308  rem_i++;
309  rem_j = 0;
310  }
311  }
312  }
313  }
314  determinant += pow(-1.0,(a+column)) * (*this)(a,column) * remMatrix.det();
315  }
316  }
317 
318  return determinant;
319 }
