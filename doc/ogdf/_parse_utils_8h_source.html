84 template<class B>
85 static int parseInt(B& in) {
86  int val = 0;
87  bool neg = false;
88  skipWhitespace(in);
89  if (*in == '-') neg = true, ++in;
90  else if (*in == '+') ++in;
91  if (*in < '0' || *in > '9') fprintf(stderr, "PARSE ERROR! Unexpected char: %c\n", *in), exit(3);
92  while (*in >= '0' && *in <= '9')
93  val = val*10 + (*in - '0'),
94  ++in;
95  return neg ? -val : val; }

98 // String matching: in case of a match the input iterator will be advanced the corresponding
99 // number of characters.
100 template<class B>
101 static bool match(B& in, const char* str) {
102  int i;
103  for (i = 0; str[i] != '\0'; i++)
104  if (in[i] != str[i])
105  return false;
106 
107  in += i;
108 
109  return true;
110 }

112 // String matching: consumes characters eagerly, but does not require random access iterator.
113 template<class B>
114 static bool eagerMatch(B& in, const char* str) {
115  for (; *str != '\0'; ++str, ++in)
116  if (*str != *in)
117  return false;
118  return true; }
