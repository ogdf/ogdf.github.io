161  m_pOrigGraph->allEdges(edges);

166  m_diGraph.clear();
167  m_edgeSlacks.init(m_diGraph);
168  m_origMapping.init(m_diGraph);
169 
170  // create resulting Steiner tree
171  m_steinerGraph.createEmpty(m_diGraph);
172  m_steinerGraph.clear();
173  m_componentMapping.init(m_steinerGraph);
174 
175  // TODO: Better choices possible?
176  m_rootTerminal = m_pTerminals->chooseElement();
177 
178  for(node v : nodes) {
179  node w = m_diGraph.newNode(v);
180  m_steinerGraph.newNode(w);
181  }
182 
183  for(edge e : edges) {
184  node source = m_diGraph.copy(e->source());
185  node target = m_diGraph.copy(e->target());
186  edge copiedEdgeS = m_diGraph.newEdge(source, target);
187  edge copiedEdgeT = m_diGraph.newEdge(target, source);
188  m_edgeSlacks[copiedEdgeS] = m_edgeSlacks[copiedEdgeT] = m_pOrigGraph->weight(e);
189  m_origMapping[copiedEdgeS] = m_origMapping[copiedEdgeT] = e;
190  }
191  updateComponents();
192 
193 #ifdef OGDF_DUAL_ASCENT_LOGGING
194  std::cout << "directed graph has " << m_diGraph.numberOfNodes() << " nodes "
195  << "and " << m_diGraph.numberOfEdges() << " edges." << std::endl

261 template<typename T>
262 List<edge> *MinSteinerTreeDualAscent<T>::computeCutSet(const node root) const
263 {
264  OGDF_ASSERT(root->graphOf() == &m_steinerGraph);
265 
266  // establish "weakly connected" component of v (non-standard definition, see paper for details)
267  NodeArray<bool> visited(m_steinerGraph, false);
268  List<node> weakComp;
269  visited[root] = true;
270 
271  List<node> queue;
272  queue.pushBack(root);
273 
274  // determine all nodes connected to root
275  // meaning all nodes from which a directed path to root exists
276  while(!queue.empty()) {
277  node v = queue.popFrontRet();
278  weakComp.pushBack(v);
279  for(adjEntry adj : v->adjEntries) {
280  edge e = adj->theEdge();
281  node w = e->source();
282  if(!visited[w]) {
283  visited[w] = true;
284  queue.pushBack(w);
285  }
286  }
287  }
288 
289  // identify (incoming) cut edges
290  List<edge> *result = new List<edge>();
291 
292  for(node v : weakComp) {
293  node w = m_steinerGraph.original(v);
294  for(adjEntry adj : w->adjEntries) {
295  edge e = adj->theEdge();
296  if(!visited[m_steinerGraph.copy(e->source())]) {
297  result->pushBack(e);
298  OGDF_ASSERT(m_steinerGraph.copy(e) == nullptr);
299  }
300  }
301  }
302  return result;
303 }

305 template<typename T>
306 bool MinSteinerTreeDualAscent<T>::isActiveComponent(const node source) const
307 {
308  OGDF_ASSERT(source->graphOf() == &m_steinerGraph);
309 
310  int comp = findComponent(source);
311  bool danglingTerminalFound = false;
312  bool hasTerminal = false;
313 
314 #ifdef OGDF_DUAL_ASCENT_LOGGING
315  std::cout << " checking whether component of node " << source << " is active.. " << std::endl;
316  std::cout << " component has id " << comp << std::endl;
317 #endif
318 
319  List<node> queue;
320  NodeArray<bool> visited(m_steinerGraph, false);
321  queue.pushBack(source);
322  visited[source] = true;
323 #ifdef OGDF_DUAL_ASCENT_LOGGING
324  while(!queue.empty()) {
325 #else
326  while(!queue.empty() && !danglingTerminalFound) {
327 #endif
328  node v = queue.popFrontRet();
329  hasTerminal |= isTerminal(v, false) && findComponent(v) == comp;
330  for(adjEntry adj : v->adjEntries) {
331  edge e = adj->theEdge();
332  node w = e->source();
333  if(!visited[w]) {
334  danglingTerminalFound |= isTerminal(w, true) && findComponent(w) != comp;
335  visited[w] = true;
336  queue.pushBack(w);
337  }
338  }
339  }
340 
341 #ifdef OGDF_DUAL_ASCENT_LOGGING
342  if(hasTerminal) { std::cout << " component includes a terminal" << std::endl; }
343  if(danglingTerminalFound) { std::cout << " component has dangling terminal" << std::endl; }
344  if(hasTerminal && !danglingTerminalFound) { std::cout << " component is active!" << std::endl; }
345 #endif
346  return hasTerminal && !danglingTerminalFound;
347 }

358  std::cout << "terminals are: ";

360  for(node v : terminals) {
361  std::cout << v << " ";
362  }
363  std::cout << std::endl;
364 #endif

367  m_pIsTerminal = &isTerminal;
368 
369  init();
370 
371  // create resulting Steiner tree
372  finalSteinerTree = new EdgeWeightedGraphCopy<T>();
373  finalSteinerTree->createEmpty(*m_pOrigGraph);
374  T result = 0;
375 
376  int comp = -1;
377  node terminal = nullptr;
378 
379 #ifdef OGDF_DUAL_ASCENT_LOGGING
380  std::cout << "main loop starting.." << std::endl;
381 #endif
382  while((comp = findActiveComponent(&terminal)) != -1) {
383  // if active comonents exists we
384  // have one of its terminals
385  OGDF_ASSERT(terminal != nullptr);
386 
387  // find minimal cut edge
388  List<edge> *cutEdges = computeCutSet(m_steinerGraph.copy(m_diGraph.copy(terminal)));
389  edge minEdge = nullptr;
390  T minSlack = MAX_VALUE;
391 #ifdef OGDF_DUAL_ASCENT_LOGGING
392  std::cout << " cut edges:";
393 #endif
394  for(edge e : *cutEdges) {
395 #ifdef OGDF_DUAL_ASCENT_LOGGING
396  std::cout << " " << e;
397 #endif
398  if(m_edgeSlacks[e] < MAX_VALUE || minEdge == nullptr) {
399  minEdge = e;
400  minSlack = m_edgeSlacks[e];
401  }
402  }
403 #ifdef OGDF_DUAL_ASCENT_LOGGING
404  std::cout << std::endl << " next edge: " << minEdge << std::endl;
405 #endif
406  OGDF_ASSERT(minEdge != nullptr);
407 
408  // update slack variables
409  for(edge e : *cutEdges) {
410  m_edgeSlacks[e] -= minSlack;
411  }
412  delete cutEdges;
413 
414  // insert edge
415  m_steinerGraph.newEdge(minEdge);
416  updateComponents();
417 
418  // insert edge into final Steiner tree
419  edge origEdge = m_origMapping[minEdge];
420  T cost = m_pOrigGraph->weight(origEdge);
421  if (finalSteinerTree->copy(origEdge->source()) == nullptr) {
422  finalSteinerTree->newNode(origEdge->source());
423  }
424  if (finalSteinerTree->copy(origEdge->target()) == nullptr) {
425  finalSteinerTree->newNode(origEdge->target());
426  }
427  if (finalSteinerTree->copy(origEdge) == nullptr) {
428  finalSteinerTree->newEdge(origEdge, cost);
429  result += cost;
430  }
431  }
432 
433 #ifdef OGDF_DUAL_ASCENT_LOGGING
434  std::cout << "removing expendable edges" << std::endl;
435 #endif
436  result -= this->pruneAllDanglingSteinerPaths(*finalSteinerTree, *m_pIsTerminal);
437  result -= this->removeCyclesFrom(*finalSteinerTree, *m_pIsTerminal);
438 
439 #ifdef OGDF_DUAL_ASCENT_LOGGING
440  std::cout << "algorithm terminated." << std::endl;
441 #endif
442  return result;
443 }
