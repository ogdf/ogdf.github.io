320  void leftShift(ArrayBuffer<INDEX, INDEX> &ind) {
321  const INDEX nInd = ind.size();
322  if (nInd == 0) return;
323 
325  #ifdef OGDF_DEBUG
326  if(ind[0] < 0 || ind[0] >= num)
327  OGDF_THROW_PARAM(AlgorithmFailureException, afcIndexOutOfBounds);
328  #endif
329 
330  INDEX j, current = ind[0];
331  for (INDEX i = 0; i < nInd - 1; i++) {
332  #ifdef OGDF_DEBUG
333  if(ind[i+1] < 0 || ind[i+1] >= num)
334  OGDF_THROW_PARAM(AlgorithmFailureException, afcIndexOutOfBounds);
335  #endif
336 
337  const INDEX last = ind[i+1];
338  for(j = ind[i]+1; j < last; j++)
339  operator[](current++) = operator[](j);
340  }

343  for (j = ind[nInd - 1]+1; j < size(); j++)
344  operator[](current++) = operator[](j);

347  }
